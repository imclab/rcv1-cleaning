			$(".state_icons_wide").animate({left: '-=475'}, 500);
		}
	} else {

			$(".state_icons_wide").animate({left: '-4617'}, 500);
		} else {
			$(".state_icons_wide").animate({left: '+=475'}, 500);
		}
