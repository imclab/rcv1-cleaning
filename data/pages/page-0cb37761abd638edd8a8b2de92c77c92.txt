function init() {
	if (window.layoutInit) layoutInit();

	if ("en_US" != "es_MX") {
		for (var k=0;k < arrSiteMakes.length;k++) {
			if (!isGMPS(arrSiteMakes[k])) {
				allGM = false;
				break;
			}
		}
		if(allGM) document.getElementById('gmTools').style.display = 'block';
	}
}
function isGMPS(make) {
	var res = false;
	var len = arrGMMakes.length;
	for (var i=0;i < len;i++) {
		if(make == arrGMMakes[i]) {
			res = true;
			break;
		}
	}
	return res;
}
