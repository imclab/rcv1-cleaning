function sharetippopup(that, platform, url){
	var track_var = 'ShareTip: ' + platform + ': ' + unescape(that.getAttribute('data-message'));
	s.tl(that, 'o', track_var);
	if (url.search('www.facebook')) {
		share_window = window.open(url, 'Facebook', 'height=450,width=630,resizable=yes');
		if (window.focus) {share_window.focus()}
	} else {
		share_window = window.open(url, 'Twitter', 'height=355,width=560,resizable=yes');
		if (window.focus) {share_window.focus()}
	}
	return false;
}

(function($){

	/* Here we're defining, and setting defaults for, all of the params you can pass to the plugin */
	/* This makes it easy to simply call .shareTip( object ) and pass in an object literal with all of the settings */
	$.fn.shareTip = function(options) {  
		var settings = {
			'speed'				: 500, //Must be a number, or string 'fast' or 'slow'
			'twitter_account'	: 'latimes', //default via @account for Twitter message
			'img_dir'			: 'http://www.latimes.com/includes/projects/metro/img/', //Directory for the facebook / twitter images
			'fbook_img'			: 'FaceBook-icon.png', //Facebook image file name
			'twitter_img'		: 'Twitter-icon.png', // Twitter image file name
			'message_attr'		: 'data-shareTip' //use this attribute to specify a message for photos
		};

		/* Before we get to the meat of the function, we need to create the div that will pop up */
		var shareDiv = document.createElement('div');
		shareDiv.id = 'shareTip';
		/* here we're going to define the basic structure for the shareTip HTML, but leave off the <a> tags for the share functionality */
		/* now that we have this template, we can use it to wipe the old links when we refresh the div */

		var templateHTML = '<div id="sharetip_button_cont" align="center"><div id="shareTipFb"><img class="shareTipButton" src="' + settings.img_dir + settings.fbook_img + '" /></div><div id="shareTipTwttr"><img class="shareTipButton" src="' + settings.img_dir + settings.twitter_img + '" /></div><div style="clear:both;"></div></div>';
		shareDiv.innerHTML = templateHTML;

		/* append the new div to the end of the document, which is hidden already with CSS */
		document.body.appendChild(shareDiv);

		/* Store the div in both a regular JavaScript variable and as a jQuery object so we can reference them faster later */
		var shareTip = document.getElementById('shareTip'),
		    $shareTip = $('#shareTip');

		/* This extends our settings object with any user-defined settings passed to the function and returns the jQuery object shareTip
		 was called on */
		return this.each(function() {        
			if (options) { 
				$.extend(settings, options);
			}

			/* This is a hack to make sure the shareTip always fades back to 100% opacity */
			var checkOpacity = function (){
				if ( $shareTip.css('opacity') !== 1 ){
					$shareTip.css({'opacity': 1});
				}
			};

			/* Function that replaces the HTML in the shareTip with the template we defined at the top */
			/* It will wipe/reset the links on the social media buttons each time the function is called */
			var removeLinks = function (){
				shareTip.innerHTML = templateHTML;
			};

			/* This is the function that makes the links for the Tweet / Share functionality */
			
			var makeURLS = function (link, message){
				/* Here we construct the Tweet URL using an array, with values passed to the function */
				var tweetConstruct = [
						'http://twitter.com/share?url=', link, '&text=', message, '&via=', settings.twitter_account
					],
					/* Then join the array into one chunk of HTML */
				    tweetURL = tweetConstruct.join(''),

					/* Same story for Facebook */
				    fbConstruct = [
						'http://www.facebook.com/sharer.php?u=', link, '&src=sp'
					],

				    fbURL = fbConstruct.join(''),
			
				    newHTML = [
				    	'<div id="sharetip_button_cont" align="center">',
							'<div id="shareTipFb">',
								'<a target="_blank" data-message=\"', message, '\" href="', fbURL, '" onclick="sharetippopup(this, \'Facebook\', \'', fbURL, '\'); return false;">',
									'<img class="shareTipButton" src="', settings.img_dir, settings.fbook_img, '" />',
								'</a>',
							'</div>',
							'<div id="shareTipTwttr">',
								'<a target="_blank" data-message=\"', message, '\" href="', tweetURL, '" onclick="sharetippopup(this, \'Twitter\', \'', tweetURL, '\'); return false;">',
									'<img class="shareTipButton" src="', settings.img_dir, settings.twitter_img, '" />',
								'</a>',
							'</div>', 
						'<div style="clear:both;"></div>',
						'</div>'
					],
				    shareHTML = newHTML.join('');
				/* Load in our new HTML */
				shareTip.innerHTML = shareHTML;
			};

			/* Since the shareTip will automatically fade out when the user mouses out of an element */
			/* we have to specifically tell the shareTip we want it to stay put when the user mouses over it */
			/* This effectively gives the user a 500 ms (or whatever) window to mouse */
			/* from the element to the shareTip to prevent it from popping out */
			$shareTip.hover(function(){
					$shareTip.stop(true, true);
					$shareTip.show();
					checkOpacity();
				}, function(){
					$(this).fadeOut(settings.speed);
			});

			/* This function handles the hover action */
			$(this).hover(function(){
					/* remove the old links, so someone doesn't accidentally click on them */
					removeLinks();

					/* If there's already an animation running on the shareTip, stop it */
					$shareTip.stop(true, true);

					var eso = $(this),
					    message,
						/* Store the width and height of the shareTip and the offset of the element for our calculations */
					    height = eso.height(),
					    width = eso.width(),
					    offset = eso.offset(),
					    link;


						link = eso.children('a').attr('href');
						message = escape( eso.children('h3').children('a').text() ) || eso.attr(settings.message_attr);

					if (link.search('http://') === -1){
						link = 'http://www.latimes.com' + link;
					}
					link = encodeURIComponent(link);

					/* The first div is smaller, so we need to compensate for that.  */
					if (eso.index() < 1){
						$shareTip.css({
							'position': 'absolute',
							'top' : offset.top + height - 15,
							'left' : (offset.left + width) - $shareTip.width()
						}).fadeIn(settings.speed, checkOpacity());				
					/* Otherwise, the shareTip will pop up to the top right of the element */
					} else {
						$shareTip.css({
							'position': 'absolute',
							'top' : offset.top + height - 5,
							'left' : (offset.left + width) - $shareTip.width()
						}).fadeIn(settings.speed, checkOpacity());			
									
					}

					/* run our function to make the URLs and generate the HTML for the shareTip, passing in the link and text from the current element */
					makeURLS(link, message);

				}, function(){
					/* On mouseout, it will fade */
						$shareTip.fadeOut(settings.speed);
			});
		});
	};
}(jQuery));
</script>
