function onloadfunc() 
{
  var mydate=new Date()
  var curryear = mydate.getFullYear()
  var yr=new Array()
  var vyr=new Array()
  var ct=0;
  for (m=curryear+1;m>1950;m--) {
  yr[ct]= m;
  vyr[ct]=m;
  ct=ct+1;
  }

  document.info.year.options.length = 0
  var year = document.info.year
  year.options[year.options.length]=new Option("Year","Year",true,false)
  for (i=0; i<yr.length; i++){
  year.options[year.options.length]=new Option(yr[i],vyr[i])
  }

  document.info.make.disabled=true;
  document.info.model.disabled=true;
}


function FillMake() {
document.info.make.disabled=false;
document.info.make.options.length = 0
var make = document.info.make
for (i=0; i < mcar.length; i++){
    string=mcar[i].split(/,/)
    if (string[0] == document.info.year.options[document.info.year.selectedIndex].value) {
        make.options[make.options.length]=new Option(string[1],string[1]);
    }
}
FillModel()
}

function FillModel() {
	document.info.model.disabled=false;
	document.info.model.options.length = 0
	var model = document.info.model
	for (i=0; i < mcar.length; i++){
		string=mcar[i].split(/,/)
		if (string[0] == document.info.year.options[document.info.year.selectedIndex].value && string[1] == document.info.make.options[document.info.make.selectedIndex].value) {
			for (j=2; j < string.length; j++){
				model.options[model.options.length]=new Option(string[j],string[j]);}
		}
	}
}

//initiate the process
$(document).ready(function() {
onloadfunc();
