$(function() {

  var add_search_tag = function(el, val) {
    var text = val || $(el).data('label');
    if (typeof(text) != 'string') {
      text = $(text).text();
    }
    var tag = $('<div />', {
      'class': 'search-block',
      text: text
    });
    tag.data('feature-id', $(el).data('feature'));
    tag.appendTo('#search-block-list');
    $('#search-block-container').show();
  }
  $('input[data-feature]').each(function(index) {
    var type = $(this).attr('type');
    if (type == 'checkbox' && $(this).attr('checked')) {
      add_search_tag(this);
    }
    if ((type == 'text' || type == 'hidden') && $(this).val() != '') {
      add_search_tag(this);
    }
  });

  $('.search-block').click(function(event) {
    var id = $(this).data('feature-id');
    var linked = $('input[data-feature="' + id + '"]');
    $(this).remove();
    if (linked.attr('type') == 'checkbox') {
      linked.prop('checked', false);
    }
    if (linked.attr('id') == 'distance-value') {
      $('#distance').slider('values', 0, 0);
      $('#distance').slider('values', 1, 25);
    }
    if (linked.attr('type') == 'text' || linked.attr('type') == 'hidden') {
      linked.val('');
    }
    window.filter_find_it_list();
  });
});
