valid = form.data('validator');

widgetName: 'Footer',
eventName: 'Contact Us',
data: 'Submit'
});
},
error: function(response) {
alert("There was an error submitting the form. Please try again.");
},
complete: function () {
footerContact.find('.loading').remove();
form.find('button').attr('disabled', 'false');
}
});
return false;
}
