function count_increase(element) {
    var eso = jQuery(element);
    var count_div = jQuery("div.tweet_count", eso.parent());
    if(!count_div.data("has_clicked")) {
        var count = count_div.html();
        if(count) {
            count = parseInt(count, 10) + 1;
        }
        else {
            count = 1;
        }
        count_div.html(count);
        count_div.data("has_clicked", true);
    }
}

function font_size(toggle) {
	var post_body_selector = jQuery('.entry-content');
	var current_size = parseInt(post_body_selector.css('font-size').replace('px', ''));
	if (toggle === 'up'){
		post_body_selector.css({'font-size': (current_size + 1) + 'px'})
	}
	else if (toggle === 'down'){
		post_body_selector.css({'font-size': (current_size - 1) + 'px'})
	}
}

	<div class="entry-content">
	<div class="entry-body"><p><a class="asset-img-link" href="http://latimesblogs.latimes.com/.a/6a00d8341c630a53ef0168ea1f6ea1970c-pi" style="display: inline;"><img alt="Robin Gibb" class="asset  asset-image at-xid-6a00d8341c630a53ef0168ea1f6ea1970c" src="http://latimesblogs.latimes.com/.a/6a00d8341c630a53ef0168ea1f6ea1970c-600wi" style="width: 600px;" title="Robin Gibb" /></a></p>
<p>Robin Gibb is gravely ill with pneumonia, reportedly in a coma, on the very week that his first classical work, &quot;The Titanic Requiem,&quot; debuted in London.</p>
<p>At the 62-year-old singer-songwriter&#39;s hospital bedside Saturday, according to the Sun newspaper, were his wife, daughter and two sons, plus his brother Barry Gibb, who&#39;d famously been with Robin and Maurice Gibb in the Grammy-winning Bee Gees.</p>
<p>“Our prayers are with Robin,&quot; a family friend told the <a href="http://www.thesun.co.uk/sol/homepage/news/4256284/Coma-vigil-for-Bee-Gee-Robin-Gibb.html" target="_blank">Sun</a>. &quot;He has kept so positive and  always believed he could beat this.&quot; Though the paper reported the coma and possibly a second tumor, publicist Doug Wright has given no details beyond saying Gibb was hospitalized in stable condition.</p>
<p><strong><a href="http://www.latimes.com/entertainment/news/la-robin-gibb-20120414,0,4025356.photogallery" target="_self">PHOTOS: Robin Gibb</a></strong></p>
<p>Gibb as recently as February was reveling in what he called a <a href="http://latimesblogs.latimes.com/gossip/2012/02/robin-gibb-colon-cancer-spectacular-recovery.html" target="_blank">&quot;spectacular&quot; recovery</a> after having a growth removed from his colon and undergoing chemotherapy. Though it has been reported that he was battling cancer, and that it involved his liver as well, Gibb has not confirmed that publicly.</p>
<p>He had intestinal surgery last month, and was <a href="http://latimesblogs.latimes.com/gossip/2011/11/robin-gibb-liver-cancer-bee-gees.html" target="_blank">hospitalized</a> late last  year for stomach and colon problems.</p>
<p>In 2010, Robin had surgery for a twisted small intestine, repairing the same congenital condition that had resulted in the death at age 53 of his twin brother Maurice.</p>
<p>&quot;He sends all his love,&quot; Robin-John said ahead of the  performance his  father missed on Tuesday. &quot;We are all praying for him and hoping he has a   speedy recovery.&quot;</p>
<p>&quot;The Titanic Requiem,&quot; which Robin Gibb composed with son Robin-John to comemmorate the 100th anniversary of the legendary ship&#39;s sinking on April 15, 1912, is a far cry from the disco-era hits that brought the Bee Gees to fame in the 1970s.</p>
<p>Hear a taste of the non-disco side of Robin Gibb, below. &quot;It&#39;s a serious subject and it’s not a rock opera,&quot; he said in a recent <a href="http://www.telegraph.co.uk/culture/music/classicalconcertreviews/9198105/Titanic-Requiem-Central-Hall-Westminster-review.html" target="_blank">interview</a>. &quot;There are no backbeats.    This could have been written 300 years ago.&quot;</p>

	<div class="entry-body"><p><a href="http://latimesblogs.latimes.com/.a/6a00d8341c630a53ef016301bca998970d-pi" style="display: inline;"><img alt="Dame Judi Dench says she&#39;s not going blind" class="asset  asset-image at-xid-6a00d8341c630a53ef016301bca998970d" src="http://latimesblogs.latimes.com/.a/6a00d8341c630a53ef016301bca998970d-600wi" style="width: 600px;" title="Dame Judi Dench says she&#39;s not going blind" /></a></p>
<p>Dame Judi Dench is suffering from age-related macular degeneration, yes, and it has affected her vision significantly, she said recently, but she also clarified Monday that no, she&#39;s not going blind.</p>
<p>Dench, 77, revealed her condition Saturday to the Daily Mirror.</p>
<p>&quot;I can’t read scripts anymore because of the trouble with my eyes. And so somebody comes in and reads them to me, like telling me a story,&quot; she told the <a href="http://www.mirror.co.uk/news/uk-news/dame-judi-denchs-eyesight-battle-689976" target="_blank">Mirror</a>. &quot;It&#39;s  usually my daughter or my agent or a friend and actually I like that,  because I sit there and imagine the story in my mind.&quot;</p>
<p>It&#39;s a condition her mother had before her, she said, and she has lenses and glasses to help — those things plus good old bright light.</p>
<p>&quot;I do not wish for this to be overblown ...&quot; Dench said Monday in a <a href="http://www.reuters.com/article/2012/02/20/us-judidench-eyesight-idUSTRE81J0X720120220" target="_blank">statement</a> sent to Reuters. &quot;It&#39;s something that I have learnt to cope with and adapt to — and it will not lead to blindness.&quot;</p>
<p>To the Mirror, the James Bond-franchise actress and Oscar winner described a particularly irritating facet of the condition, however. &quot;The most distressing thing is in a restaurant in the evening I can’t see the person I’m having dinner with,&quot; she said, after explaining to the interviewer that while she could see his outline, his face remained a mystery.</p>

		function dateparse(pubDate){
				date = new Date(pubDate);
				m = date.getMonth() + 1; 
				d = date.getDate();
				y = date.getFullYear();
				h = date.getHours();
				if (h > 12){
				h -= 12;
					var ampm =  "p.m.";
				}
				else if (h == 12) {
					var ampm =  "p.m.";
				}

					var ampm = "a.m.";
				}
				mm = date.getMinutes();

					mm = "0" + mm; 
				}
				hhmm = ", " + h + ":" + mm + " " +ampm;
				if (h == 0){
					hhmm = "";
				}
				var timestamp = m + "/" + d + "/" + y + hhmm;
				return timestamp;
		}

var stickytooltip={
	tooltipoffsets: [0, -20], //additional x and y offset from mouse cursor for tooltips
	fadeinspeed: 250, //duration of fade effect in milliseconds

	//***** NO NEED TO EDIT BEYOND HERE

	isdocked: false,

	positiontooltip:function($, $tooltip, e){
		var x=e.pageX+this.tooltipoffsets[0], y=e.pageY+this.tooltipoffsets[1]
		var tipw=$tooltip.outerWidth(), tiph=$tooltip.outerHeight(),
		x=x-(tipw/2)
		//x=(x+tipw>$(document).scrollLeft()+$(window).width())? x-tipw-(stickytooltip.tooltipoffsets[0]*2) : x
		y=y-tiph
		$tooltip.css({left:x, top:y})
	},	
	
	showbox:function($, $tooltip, e){
		$tooltip.show()
		this.positiontooltip($, $tooltip, e)
	},

	hidebox:function($, $tooltip){
		if (jQuery('#googleplus').css('display') == 'block') {
			$tooltip.fadeOut(2000)
		}
		else {
			$tooltip.fadeOut(500)
		}
	},

	init:function(targetselector, tipid){
		jQuery(document).ready(function($){
			var $targets=$(targetselector)
			var $tooltip=$('#'+tipid).appendTo(document.body)
			if ($targets.length==0)
				return
			var $alltips=$tooltip.find('div.newtip')

			stickytooltip.hidebox($, $tooltip)
			
			$targets.hover(function(e){
				$alltips.hide().filter('#'+$(this).attr('data-tooltip')).show()
				$tooltip.stop(true, true)
				stickytooltip.showbox($, $tooltip, e)
			}, function(e){
				stickytooltip.hidebox($, $tooltip)
			})

			$tooltip.hover(function(){
				jQuery(this).stop(true, true)
				stickytooltip.showbox($, $tooltip)				
			}, function(){
				stickytooltip.hidebox($, $tooltip)								
			})

			$(this).bind("click", function(e){
				if (e.button==0){
					stickytooltip.hidebox($, $tooltip)
				}
			})

		}) //end dom ready
	}
}

function vidfix(){

	  var objs = placeholder.getElementsByTagName("object");
	  obj = objs[0];

	  obj.removeAttribute("data");
	  obj.removeAttribute("width");
	  obj.removeAttribute("height");

    	widthItem.setAttribute('name','width');
    	widthItem.setAttribute('value','620');

	  var heightItem = document.createElement('param');
    	heightItem.setAttribute('name','height');

	  myobject[i].parentNode.replaceChild(placeholder.firstChild, myobject[i]);

	}
