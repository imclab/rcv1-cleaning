$(function(){ 
	/* *************************************************************************************************
	Instantiate the following values
	comparevehiclelist - Sets list of vehicles from the cookie.comparevehiclelist.
	totalCompare - Total number of vehicles in cookie.comparevehiclelist
	savedvehiclelist - Sets list of vehicles from the cookie.savedvehiclelist.
	totalSaved - Total number of vehicles in cookie.savedvehiclelist
	**************************************************************************************************** */
	if (!$.cookie('COMPAREVEHICLELIST')) {
		comparevehiclelist = '';
	}
	else {
		comparevehiclelist = $.cookie('COMPAREVEHICLELIST');
	}
	
	if (!$.cookie('SAVEDVEHICLELIST')) {
		savedvehiclelist = '';
	}
	else {
		savedvehiclelist = $.cookie('SAVEDVEHICLELIST');
	}
	
	
	//Loop through all saveCheckBox input and check if the value matches the value in savedvehiclelist
	$("input[name=saveCheckBox]").each(function(x) {
		var savedArr = savedvehiclelist.split(',');
		
		if (parseInt(jQuery.inArray($(this).val(), savedArr)) > -1) {
			$(this).attr('checked',true);
		}
		else {
			$(this).attr('checked',false);
		}
	});
	
	//Loop through all compareCheckBox input and check if the value matches the value in comparevehiclelist
	$("input[name=compareCheckBox]").each(function(x) {
		var compareArr = comparevehiclelist.split(',');

		if (parseInt(jQuery.inArray($(this).val(), compareArr)) > -1) {
			$(this).attr('checked',true);
		}
		else {
			$(this).attr('checked',false);
		}
	});
	
	if ($.cookie('COMPAREVEHICLELIST') != '' && $.cookie('COMPAREVEHICLELIST') != null && $.cookie('COMPAREVEHICLELIST') != 0) {
		totalCompare = parseInt($.cookie('COMPAREVEHICLELIST').toString().split(',').length);
	}

		totalSaved = parseInt($.cookie('SAVEDVEHICLELIST').toString().split(',').length);
	}
	
	if (totalCompare > 0) {
		$("#comparedVehiclesCount").html(totalCompare);
	}
	else {
		$("#comparedVehiclesCount").html(0);
	}
		
	if (totalSaved > 0) {
		$("#savedVehiclesCount").html(totalSaved);
	}
	else {
		$("#savedVehiclesCount").html(0);
	}
