			//calculate points from dollars spent
			amountSpent = Math.round(amountSpent*100*100)/100;

			var points = Math.floor(amountSpent/baseVal)*SYWR_POINTS_OBTAINED;

			if(modVal >= roundUpFactor){
				points = points + parseInt(SYWR_POINTS_OBTAINED);
			}
			return points;
			}
