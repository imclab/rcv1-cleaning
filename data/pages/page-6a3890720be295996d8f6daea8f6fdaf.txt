function removeElement(parentDiv, childDiv) {
    if (childDiv == parentDiv) {
        return false;
    }
    else if (document.getElementById(childDiv)) {
        var child = document.getElementById(childDiv);
        var parent = document.getElementById(parentDiv);
        parent.removeChild(child);
    }
    else {
        return false;
    }
}
function addElement() {
		var dv = '<iframe allowtransparency="true" style="border: none; height: 530px; width: 930px; display: none;" id="frame-trick" frameborder="0" scrolling="no"></iframe>';
		jQuery('#iframeContainer').append(dv);
}

function playMeChild(e){
	if (!document.getElementById('frame-trick'))
		addElement();
	document.getElementById("frame-trick").src = "video-player.html?"+e;
	setTimeout(function() {
		document.getElementById("frame-trick").style.display = 'inline-block';
	},2000);
	childModal();
}

function childModal(){
	jQuery(document).ready(function($) {
		var id="#dialog";
		//Get the screen height and width
		var maskHeight = $(document).height();
		var maskWidth = $(window).width();
	
		//Set heigth and width to mask to fill up the whole screen
		$('#mask').css({'width':maskWidth,'height':maskHeight});
		
		//transition effect		
		$('#mask').fadeIn(1000);	
		$('#mask').fadeTo("slow",0.8);	
	
		//Get the window height and width
		var winH = $(window).height()+$(window).scrollTop();
		var winW = $(window).width();
              
		//Set the popup window to center
		$(id).css('top',  winH/2-$(id).height()/2 +'px');
		$(id).css('left', winW/2-$(id).width()/2 +'px');
	
		//transition effect
		$(id).fadeIn(2000); 
	});
}

function getCookie(c_name)
			{
			var i,x,y,ARRcookies=document.cookie.split(";");
			for (i=0;i<ARRcookies.length;i++)
			{
			  x=ARRcookies[i].substr(0,ARRcookies[i].indexOf("="));
			  y=ARRcookies[i].substr(ARRcookies[i].indexOf("=")+1);
			  x=x.replace(/^\s+|\s+$/g,"");
			  if (x==c_name)
				{
				return unescape(y);
				}
			  }
			}
