        function getHoverSectionContent(sectionName) {
            var xmlhttp;

            if (window.XMLHttpRequest) {// code for IE7+, Firefox, Chrome, Opera, Safari
                xmlhttp = new XMLHttpRequest();
            }
            xmlhttp.onreadystatechange = function () {
                if (xmlhttp.readyState == 4 && xmlhttp.status == 200) {
                    var i = 0;
                    var str = "";
                    var resultNode = xmlhttp.responseXML.documentElement;

                    str += "<div style=\"float:left; width:100%; clear:left;\">";
                    str += "<div style=\"float:left; position:relative; width:100%; right:370px; border-right: 1px solid #B0B0B0;\">";
                    str += "<div style=\"float:left; position:relative; width:100%; right:450px; border-right: 1px solid #B0B0B0;\">";

                    var tags = xmlhttp.responseXML.documentElement.getElementsByTagName("TAGS");
                    str += "<div class=\"hoverTags\" style=\"float:left; position:relative; width:150px; left:820px;\">";
                    if (tags.length > 0) str += "<h4>TOPICS</h4>";
                    for (var i = 0; i < tags.length; i++) {
                        //var TagElements = new Array();
                        var y = tags[i].getElementsByTagName("T");

                        for (var j = 0; j < y.length; j++) {
                            //TagElements.push(y[j])
                            var tagElements = y[j].textContent.split("~");
                            str += "<div class=\"sideTags\"><a href=\"/news/searchresults.aspx?searchKey=" + tagElements[0] + "&p=1&pub=D&tag=" + tagElements[1] + "\">" + tagElements[0] + "</a></div>";

                        }
                    }
                    str += "</div>";

                    var headlines = xmlhttp.responseXML.documentElement.getElementsByTagName("HEADLINES");
                    str += "<div class=\"hoverHeadlines\" style=\"float:left; position:relative; width:400px; left:810px;\">";
                    if (headlines.length > 0) str += "<h4>TOP HEADLINES</h4>";

                    for (var i = 0; i < headlines.length; i++) {
                        var y = headlines[i].getElementsByTagName("H");
                        for (var j = 0; j < y.length; j++) {
                            var headlineElements = y[j].textContent.split("~");
                            str += "<h5><a href=\"/news/story.aspx?id=" + headlineElements[0] + "\">" + headlineElements[1] + "</a></h5>";
                        }
                    }
                    str += "</div>";

                    var photos = xmlhttp.responseXML.documentElement.getElementsByTagName("MULTIMEDIA");

                    str += "<div class=\"hoverMultimedia\" style=\"float:left; position:relative; width:350px; left:840px;\">";
                    for (var i = 0; i < photos.length; i++) {
                        var y = photos[i].getElementsByTagName("M");
                        for (var j = 0; j < y.length; j++) {
                            var multimediaElements = y[j].textContent.split("~");
                            str += "<div class=\"hoverMultimediaItems\">";
                            str += "<a href=\"/news/story.aspx?id=" + multimediaElements[0] + "\" title=\"" + multimediaElements[1] + "\"><img src=\"/news" + multimediaElements[3] + "\"/></a>";
                            str += "<div class=\"hoverMultimediaHeadline\"><a href=\"/news/story.aspx?id=" + multimediaElements[0] + "\" title=\"" + multimediaElements[1] + "\">" + multimediaElements[1] + "</a></div>";
                            str += "<div class=\"hoverMultimediaSummary\"><a href=\"/news/story.aspx?id=" + multimediaElements[0] + "\">" + multimediaElements[2] + "</a></div>";
                            str += "</div>";
                        }
                    }
                    str += "</div>";

                    str += "</div>";
                    str += "</div>";
                    str += "</div>";


                    document.getElementById("sectionHover").innerHTML = str;
                }
                else if (xmlhttp.status == 404) {

                }
                else {

                }
            }
            xmlhttp.open("POST", "/news/ajax-source/SectionHoverDisplay.aspx?", true);
            xmlhttp.setRequestHeader("Content-type", "application/x-www-form-urlencoded");
            xmlhttp.send("sectionName=" + sectionName);

            return false;
        }

    function validatePasswordFunction(pw, errDisplay) {
        
        document.getElementById(errDisplay).value = "";
        
        var options = {
            length: [8, 50],
            lower: 0,
            upper: 0,
            numeric: 1,
            special: 0,
            badWords: ["password"],
            badSequenceLength: 5
        };

        // default options (allows any password)
        var o = {
            lower: 0,
            upper: 0,
            alpha: 0, /* lower + upper */
            numeric: 0,
            special: 0,
            length: [0, 40],
            custom: [ /* regexes and/or functions */],
            badWords: [],
            badSequenceLength: 0,
            noQwertySequences: false,
            noSequential: false
        };

        for (var property in options)
            o[property] = options[property];

        var re = {
            lower: /[a-z]/g,
            upper: /[A-Z]/g,
            alpha: /[A-Z]/gi,
            numeric: /[0-9]/g,
            special: /[\W_]/g
        },
		rule, i;

        // enforce min/max length
        if (pw.length < o.length[0] || pw.length > o.length[1]) {

           document.getElementById(errDisplay).value= "Password should be atleast 8 characters in length and less than 50 characters";
            return false;
        }

        // enforce lower/upper/alpha/numeric/special rules
        for (rule in re) {
            if ((pw.match(re[rule]) || []).length < o[rule]) {
                document.getElementById(errDisplay).value = "Password should have atleast one  numeric and special character";
                return false;
            }
        }

        // enforce word ban (case insensitive)
        for (i = 0; i < o.badWords.length; i++) {
            if (pw.toLowerCase().indexOf(o.badWords[i].toLowerCase()) > -1)
               {
                document.getElementById(errDisplay).value = "Password cannot contain the word 'password'";
                return false;
            }
        }

        // enforce the no sequential, identical characters rule
        if (o.noSequential && /([\S\s])\1/.test(pw)) {
            document.getElementById(errDisplay).value = "Password cannot contain alphanumeric sequence five or more characters in length ";
            return false;
        }

        // enforce alphanumeric/qwerty sequence ban rules
        if (o.badSequenceLength) {
            var lower = "abcdefghijklmnopqrstuvwxyz",
			upper = lower.toUpperCase(),
			numbers = "0123456789",
			qwerty = "qwertyuiopasdfghjklzxcvbnm",
			start = o.badSequenceLength - 1,
			seq = "_" + pw.slice(0, start);
            for (i = start; i < pw.length; i++) {
                seq = seq.slice(1) + pw.charAt(i);
                if (
				lower.indexOf(seq) > -1 ||
				upper.indexOf(seq) > -1 ||
				numbers.indexOf(seq) > -1 ||
				(o.noQwertySequences && qwerty.indexOf(seq) > -1)
			) {
                    document.getElementById(errDisplay).value = "Password cannot contain alphanumeric sequence five or more characters in length ";
                    return false;
                }
            }
        }

        // enforce custom regex/function rules
        for (i = 0; i < o.custom.length; i++) {
            rule = o.custom[i];
            if (rule instanceof RegExp) {
                if (!rule.test(pw))
                    return false;
            } else if (rule instanceof Function) {
                if (!rule(pw))
                    return false;
            }
        }

        // great success!
        document.getElementById(errDisplay).value = "";
        return true;
    }

    function createUUID() {

        var s = [];
        var hexDigits = "0123456789ABCDEF";
        for (var i = 0; i < 32; i++) {
            s[i] = hexDigits.substr(Math.floor(Math.random() * 0x10), 1);
        }
        s[12] = "4";  // bits 12-15 of the time_hi_and_version field to 0010
        s[16] = hexDigits.substr((s[16] & 0x3) | 0x8, 1);  // bits 6-7 of the clock_seq_hi_and_reserved to 01

        var uuid = s.join("");
        return uuid;
    }

    function getCurrDSSide() {

            case 'all': return allMediaDSSide;

            case 'video': return videosDSSide;

        function changePageSide() {

                currPageDSSide = spliceArray(getCurrDSSide(), currItemPageMultSide*itemsPerPageMultSide, currItemPageMultSide*itemsPerPageMultSide+itemsPerPageMultSide);

                    if(currPageDSSide[i] != null) {
                        headerSide = document.getElementById("sideHeader" + i);

                        descSide = document.getElementById("sideDesc" + i);

                            iconSide='<span id="cam" class="spanCameraClass" ></span>';
                        }
                        else if(currPageDSSide[i].item_type =='A')
                        {
                            iconSide = '<span id="aud" class="spanAudioClass"></span>';
                        }
                        else if(currPageDSSide[i].item_type =='V')
                        {
                            iconSide='<span id="vid" class="spanVideoClass"></span>';
                        }

                        {
                            iconSide='<span id="int" class="spanInteractiveClass"></span>';
                        }
                        if(headerSide == null || imageSide == null || descSide == null) return;
                        
                        if(currPageDSSide[i].height > currPageDSSide[i].width) {
                            imgPropertiesSide = 'width="' + 110 + 'px"';
                            imgStyleSide = 'style="margin: 0px 0px 0px 0px;"';

                    }
                 }
            }

         function spliceArray(arr, start, end) {

            for(var i=start; i<end; i++) {

            }
