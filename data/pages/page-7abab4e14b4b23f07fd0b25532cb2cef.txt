 function dynamicDataHasValue(data){
      if(data==null) return false; //data is null
      if(data.replace(/\s/g, '').length < 1) return false; //data is all whitespace or empty string
      if(data.substring(0,1)!="@"){ //data doesn't start with "@"
          return true;
      }else{
          return false;
      }
  }
