function restructure_sub_nav(li_id){
        var sub_width=0;
        var parent_width=$("#"+li_id).innerWidth();
        var nav_width=$("#"+li_id).parent().innerWidth();
        $("#"+li_id+" > div.subnav > ul.sub_col").each(function(i){
                sub_width+=$(this).innerWidth();
//              alert(sub_width);
        });
        if (sub_width < parent_width){
                sub_width = parent_width;
        }
		
		if (sub_width > nav_width)
		{
			remove_sub_nav_column(li_id);
			sub_width=restructure_sub_nav(li_id);
		}
		
        $("#"+li_id+" > div.subnav").width(sub_width+2);
        return sub_width;
}

function remove_sub_nav_column(li_id){

	var current_items=$("#"+li_id+" > div.subnav > ul.sub_col > li").length;

	$("#"+li_id+" > div.subnav > ul.sub_col").each(function(i){
		if (plus_cols>0)
		{
			var rows=min_rows+1;
			plus_cols--;
		}
		else
		{
			var rows=min_rows;
		}

//		alert('there are '+$(this).children('li').length+' rows, and we need '+rows);

					rows=0;
				}
			}
		}

			$(this).remove();

var adv_search_arguments='';

function set_value(criteria) {
	
	var post_data={criteria:criteria};
	
	var type=$('#adv_tp').val();
	if (type.length) {
		post_data.type=type;
	}
	var make=$('#adv_mk').val();
	if (make.length) {
		post_data.make=make;
	}
	var model=$('#adv_md').val();
	if (model.length) {
		post_data.model=model;
	}
	
	//var adv_search_arguments='{"makes":["ford","chevy"]}'; 

	post_data.arg_string=adv_search_arguments;
	
	$.post('/req/private/search/advancedsearch/filter_results/6b05b73a5806fccc5d4514972630ada5/', post_data, function(data) {
		switch (criteria)
		{
			case 'tp': // update the make select
				$('#adv_mk').html(data);
				$('#adv_mk').removeAttr('disabled');
				if ($('#adv_mk option').length==1) {
					set_value('mk');
				}
				break;
			case 'mk': // update the model select
				$('#adv_md').html(data);
				$('#adv_md').removeAttr('disabled');
				break;
		}
	});
}

function update_search_url() {
	var strings=Array();
	var ids=Array();
	var sort_order='';
	
	var type=$('#adv_tp').val();
	if (type.length) {
		strings.push(type);
		ids.push('tp');
	}
	var make=$('#adv_mk').val();
	if (make.length) {
		val_parts=make.split('-');
		strings.push(val_parts[0]);
		ids.push('mk'+val_parts[1]);
	}
	var model=$('#adv_md').val();
	if (model.length) {
		val_parts=model.split('-');
		strings.push(val_parts[0]);
		ids.push('md'+val_parts[1]);
	}
	var order=$('#adv_o').val();
	if (order.length) {
		sort_order='s:'+order+'/';
	}
	if (ids.length) {
		$('#advancedsearchform').attr('action', '/search/'+strings.join('-')+'/'+ids.join('-')+'/'+sort_order);
	} else {
		$('#advancedsearchform').attr('action', '/search/'+sort_order);
	}	
}
