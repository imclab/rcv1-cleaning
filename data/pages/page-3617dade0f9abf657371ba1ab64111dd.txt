			function _initSubmit(){
			    $("#submit-search").click(function(){
					if (GMCERTIFIED.search.validateSearchForm() == false) {
						return false;
					} else {
						var zipc = $.trim($("#zip").val());	
						if (isZipCode(zipc) == true) {
							GMCERTIFIED.PROFILECOOKIE.setZipcode(zipc);
						}	
						
						var oss = GMCERTIFIED.vehiclesearch.getSelectedMMTBOmnitureString();
						if (omniture == null) {
							omniture = new OmnitureWrapper();
						}
						if (context != ADVANCED && omniture != null) {
							
							var omni = new OmnitureWrapper().clear();
							
							var ch = omniture.getChannel();
							if (isEmpty(ch)) {
								ch = "Home";
							}
							var pg = omniture.getPage();
							if (isEmpty(pg)) {
								pg = "GM CERTIFIED | HOMEPAGE";
							}
							omni.channel(ch);
							omni.campaign(getQueryParam('s_cid'));
							omni.linkTrackVars('prop1,eVar1');
							omni.sprop(1, "Quick Searches");
							omni.evar(1, oss);
							omni.trackCustomLinkOnly(new Object(),'Quick Searches', null);
							
						} else if (context == ADVANCED && omniture != null) {
							var omni = new OmnitureWrapper().clear();
							omni.channel("Search");
							omni.campaign(getQueryParam('s_cid'));
							omni.linkTrackVars('prop2,eVar2');
							omni.sprop(2, "Advanced Searches");
							omni.evar(2, oss);
							omni.trackCustomLinkOnly(new Object(),"GM CERTIFIED | VEHICLE SEARCH", null);
						}
											
						var distance = $("#distance").val();	
						if (distance == GMCERTIFIED.search.UNLIMITED) {
							$("#distance").val("");	
						}
						if (context == QUICK) {
							var makeSel = $("#makeSel").val();
							if (makeSel == "Chevrolet"){
								$("#looking4Chevy").val("true")
							}
							if (makeSel == "Buick"){
								$("#looking4Buick").val("true")
							}
							if (makeSel == "GMC"){
								$("#looking4GMC").val("true")
							}
						} else if (context == ADVANCED) {
							if ($("#select-makes").attr("checked") == true) {
								checkIfSelectedMakeChevyBuickGMC("make");
							}	
							if ($("#select-trims").attr("checked") == true) {
								checkIfSelectedMakeChevyBuickGMC("trim");
							}
						}
						if ($("#select-models").attr("checked") == true) {
							checkIfSelectedMakeChevyBuickGMC("model");
						}	
						
						$("#searchForm").submit();
					}
			    });
			}

			function _initAdvancedSearch(){
				if (context != ADVANCED) return;	
				
				$("#makes-list").hide();
				$("#body-list").hide();
				
				$("input[type='radio']").click(function(){	
				
					var id = $(this).attr("id");
					var name = $(this).attr("name");
					
					if (id.indexOf("all") >= 0) {	

						var val = $(this).val();
						
						$("input[type='checkbox'][name='" + val + "']").each(function(){
							$(this).attr("checked", false);
							if (val != 'model' && val != 'trim') {
								$.uniform.update("input[type='checkbox'][name='" + val + "']");
							}	
						});
						
						if (val == "bodyStyle") {
							$("#body-list").hide();
						} else if (val == "make") {
							$("#makes-list").hide();
						}
						
						$("#" + name + "DivCar").hide();

						if (name.indexOf("body") >= 0) {
							if ($("#select-models").attr("checked") == true) {
								var results = GMCERTIFIED.MMTBDB.vehicles.get(GMCERTIFIED.search.getWhereClauseObj(GMCERTIFIED.search.getWhereClause()));	
								GMCERTIFIED.search.onSuccess(results,"modelRadioDivTemplate","modelRadioDivCar");
							}	
							if ($("#select-trims").attr("checked") == true) {
								GMCERTIFIED.search.hideTrims();
							}	
						} else if (name.indexOf("make") >= 0) {
							if ($("#select-models").attr("checked") == true) {
								var results = GMCERTIFIED.MMTBDB.vehicles.get(GMCERTIFIED.search.getWhereClauseObj(GMCERTIFIED.search.getWhereClause()));	
								GMCERTIFIED.search.onSuccess(results,"modelRadioDivTemplate","modelRadioDivCar");
							}	
							if ($("#select-trims").attr("checked") == true) {
								GMCERTIFIED.search.hideTrims();
							}	
						} else if (name.indexOf("model") >= 0) {
							if ($("#select-trims").attr("checked") == true) {
								GMCERTIFIED.search.hideTrims();
							}	
						} 
					} else {
						var showingModelOrTrim = name.indexOf("model") >= 0 || name.indexOf("trim") >= 0;
						var showingTrim = name.indexOf("trim") >= 0;
						var showingModel = name.indexOf("model") >= 0;
						var showingMake = name.indexOf("make") >= 0;
						var showingBS = name.indexOf("bodyStyle") >= 0;

						if (showingTrim) {
							if (GMCERTIFIED.search.doTrimWarn()) {
								return;
							}
						}
						
						if (showingBS) {
							$("#body-list").show();
						} else if (showingMake) {
							$("#makes-list").show();
						} else {
							$("#" + name + "DivCar").show();
							var whereClause = GMCERTIFIED.search.getWhereClause();		
							var results;
							if (!isEmpty(whereClause) && showingModelOrTrim == true) {
								var obj = GMCERTIFIED.search.getWhereClauseObj(whereClause);
								results = GMCERTIFIED.MMTBDB.vehicles.get(obj);
							}else {
								if (name.indexOf("model") >= 0) {
									results = GMCERTIFIED.MMTBDB.vehicles.get({mo:{length:{gt:0}}});
								} else if (name.indexOf("trims") >= 0) {
									results = GMCERTIFIED.MMTBDB.vehicles.get({tr:{length:{gt:0}}});
								}
							}
							GMCERTIFIED.search.onSuccess(results, name +"DivTemplate", name + "DivCar");
						}	
					}
				});	
			}

	function undoLastFilterSetting(){
		
		var lfs = lastFilterSetting;
		if (!filterSettings.isEmpty()) {
			lfs = filterSettings.pop();
		} 
	
		var arr = lfs.split("=");
		var fieldName = arr[0];
		var fieldValue = arr[1];
		
		if (fieldName == "yearMin"){
			$("#yearMin").val(fieldValue);
			$("#yearMin").attr("title", fieldValue);
			$("#slider-year-range").slider("values",0,fieldValue);
		} else if (fieldName == "yearMax"){
			$("#yearMax").val(fieldValue);
			$("#yearMax").attr("title", fieldValue);
			$("#slider-year-range").slider("values",1,fieldValue);
		} else if (fieldName == "priceMin"){
			$("#priceMin").val(fieldValue);
			$("#priceMin").attr("title", fieldValue);
			$("#slider-price-range").slider("values",0,fieldValue);
		} else if (fieldName == "priceMax"){
			$("#priceMax").val(fieldValue);
			$("#priceMax").attr("title", fieldValue);
			$("#slider-price-range").slider("values",1,fieldValue);
		}else if (fieldName == "mileageMin"){
			$("#mileageMin").val(fieldValue);
			$("#mileageMin").attr("title", fieldValue);
			$("#slider-mileage-range").slider("values",0,fieldValue);
		} else if (fieldName == "mileageMax"){
			$("#mileageMax").val(fieldValue);
			$("#mileageMax").attr("title", fieldValue);
			$("#slider-mileage-range").slider("values",1,fieldValue);
		}else {
			var id = "#" + fieldName;
			$(id).val(fieldValue);
			$(id).attr("title", fieldValue);
			if (fieldName == "distance") {
				$("#slider-distance").slider("value",fieldValue);	
				$("#distance").attr("class", "");
			}
			if (fieldName == "transmission" ||
				fieldName == "fuelType" ||
				fieldName == "engine" ||
				fieldName == "drivetrain" ||
				fieldName == "seats" ||
				fieldName == "exteriorcolor") {

				var tup = fieldValue.split("|");
				fieldValue = tup[0];				
				var fieldText = tup[1];				
				$(id).val(fieldValue);
				if (isEmpty(fieldValue)) {
				
					var txt = "Any ";
					if (fieldName == "fuelType") txt += "Incl. Hybrid";
					
					$("#uniform-" + fieldName).find("span").text(txt);
				} else {
			  		$("#uniform-" + fieldName).find("span").text(fieldText);
			  	}	
			}
			
			
			
			//See if its a checkbox
			if (id.indexOf("Only") >= 0) {
				$(id).attr("checked", lfs.indexOf("true") >= 0);
				if (lfs.indexOf("true") >= 0) {
					$("#uniform-" + fieldName).find("span").addClass("checked");
				} else {
					$("#uniform-" + fieldName).find("span").removeClass("checked");
				}	
			}
		}

		if (filterSettings.isEmpty()) {
			$("#undoLastLink").hide();
		}
		
		refineSearchResults();
		
		
	}
