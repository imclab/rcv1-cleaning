    function toggleTabs(element, activeClass, desiredTag,searchType,optionList) {
        var elements = element.parentNode.parentNode.getElementsByTagName(desiredTag);

        for (var i = 0; i < elements.length; i++)
            elements[i].className = "";

        element.className = activeClass;
        activeTabSearchType = searchType;
        activeOptionList = optionList;

        $('TabbedFSOptions').style.display = 'none';
        $('TabbedFLOptions').style.display = 'none';
        $('TabbedRSOptions').style.display = 'none';
        $('TabbedMPOptions').style.display = 'none';

        switch (searchType) {
            case "FS":
                setCorrespondingListItem(FSCategories, searchType, 'All properties for sale');
                break;
            case "FL":
                setCorrespondingListItem(FLCategories, searchType, 'All properties for lease');
                break;
            case "RS":
                setCorrespondingListItem(RSCategories, searchType, 'All sales comps');
                break;

            case "MP":
                setCorrespondingListItem(MPCategories, searchType, 'All property records');
            default:
        }
    }

    function setCorrespondingListItem(categoryIDArray, searchType, defaultCategoryName)
    //select an item based on previous list for Tabbed QuickSearch
    {
        var len = categoryIDArray.length;
        var itemFound = false;
        for (var i = 0; i < len; i++) {
            if (categoryIDArray[i] == UniversalQuickSearch.CategoryID) //found
            {
                itemFound = true;
                UniversalQuickSearch.SelectPropertyType(UniversalQuickSearch.CategoryName, categoryIDArray[i], searchType);
                break;
            }
        }

        if (!itemFound) {
            UniversalQuickSearch.SelectPropertyType(defaultCategoryName, '', searchType);
        } 
    }
