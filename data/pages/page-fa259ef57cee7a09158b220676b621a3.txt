		function getSearchTerm(searchTerm) {
			var search = $.trim(searchTerm),
				term = "",
				url = "http://" + window.location.hostname;
			
			if (search.length) {
				$.ajax({
					async: false, 
					type: "POST",
					url: url + "/search/searchTerm.cfm", 
					data: { term: search}, 
					// dataType: "html",
					success: function(msg){
						// create the regex object to get the term from the returned values
						var reg = /.*?(<term>.*?<\/term>).*/gim;
						// get the first element of the array that is returned by the exec function
						term = reg.exec(msg);
						// the exec() function returns an array || null. Make sure that the value is NOT null and proceed.
						// if the value is null then set the term to an empty string
						if (term) {
							term = term[0];
							// remove the HTML tags from the returned string
							term = term.replace(/.*?<term>(.*?)<\/term>.*/, "$1", "gim");
						} else {
							term = "";
						}
					}, 
					error: function(XMLreq, tS, err) {
						var fullError = "XMLHttpReq: " + XMLreq + "\ntextStatus: " + tS + "\nerrorThrown: " + err;
						window.location.href = url + "/term/" + encodeURI(search);
					}
				});
			}
			return term || encodeURI(search);
		}
		
		function fixedEncodeURIComponent (str) {
