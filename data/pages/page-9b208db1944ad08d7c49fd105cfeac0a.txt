	function isValid(sText, sType)
	{
		// Use this type for numbers with decimal points and commas
   		if(sType == 'numeric')
		{
			var ValidChars = "0123456789.,";
		}
		// Use this type for US zip codes
		else if(sType == 'zip')
		{
			var ValidChars = "0123456789-";
		}
		// Use this type for phone and fax numbers
		else if(sType == 'phone')
		{
			var ValidChars = "0123456789 .-()";
		}
		// Use this type for integers
		else
		{
			var ValidChars = "0123456789";
		}
   		var isValidStr=true;
   		var Char;
		// If the field is blank, return an error
		if(sText.length == 0)
		{
			isValidStr = false;
		}
		// Loop through all characters in the passed string for any invalid characters
   		for(i = 0; i < sText.length && isValidStr == true; i++)

      		Char = sText.charAt(i); 
      		if(ValidChars.indexOf(Char) == -1)

      	}

		// If the field is blank, return an error
		if(strValue.length == 0)

			validDate = false;

		var dateSplit = strValue.split('/');

		if(dateSplit.length < 3)

			if(dateSplit.length < 3)

			}
		}
		// If the date string hasn't been discounted yet, check for valid integer values
		if(validDate == true)

			// If there are at least three values in the array (month, day and year), test each string in array to make sure that it is a valid integer value

			else
			{
				validMonth = Number(dateSplit[0]);
			}
			if(!isValid(dateSplit[1], 'integer'))
			{
				validDate = false;
			}
			else
			{
				validDay = Number(dateSplit[1]);
			}
			if(!isValid(dateSplit[2], 'integer'))
			{
				validDate = false;
			}
			else
			{
				validYear = Number(dateSplit[2]);
			}
			// Months must be numeric from 1-12
			if(!validMonth || validMonth < 1 || validMonth > 12)
			{
				validDate = false;
			}
			// Days must be numeric from 1-31
			if(!validDay || validDay < 1 || validDay > 31)
			{
				validDate = false;
			}
			// Years must be numeric.  This range can be modified as needed
			if(!validYear || validYear < 1900 || validYear > 2100)
			{
				validDate = false;
			}
		}
		if(validDate == true)
		{
			return true;
		}
		else
		{
			return false;
		}
	}

		else

	}
	// Compare two dates to make sure that the second date is the same as or comes after the first
	function compareDate(laterDate, earlierDate)
	{
		// JS treats months as integers in an array, not as natural numbers.  So, January is "0" and December is "11"
		var laterMonth = laterDate[0]-1;
		var laterDay = laterDate[1];
		var laterYear = laterDate[2];
		// JS treats months as integers in an array, not as natural numbers.  So, January is "0" and December is "11"
		var earlierMonth = earlierDate[0]-1;
		var earlierDay = earlierDate[1];
		var earlierYear = earlierDate[2];
		var firstDate = new Date();
		var secondDate = new Date();
		firstDate.setFullYear(earlierYear,earlierMonth,earlierDay);
		secondDate.setFullYear(laterYear,laterMonth,laterDay);
		// If the second date field is actually set before the first date field, return false
		if(secondDate < firstDate)
		{
			return false;
		}
		else
		{
			return true;
		}
	}
