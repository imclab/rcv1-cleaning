    function PickupDateSelection(pkupDate) {
        ///<summary>PickupDateSelection function is being triggered by the selection of a date
        ///     in the pickup calendar Extender control.  this function sets the return
        ///     date textbox if the return date is empty or less then the pickup date selected</summary>
        ///<param name="pkupDate">string value of selected date</param>
        var pickupDate = new Date(pkupDate);
        if (pickupDate) {
            //set the new 'disable prior to' date in the return box
            var returnDateDisablePrior = $find('ctl03_LocationTime_ReturnDateCalendar_PopupCalendar');
            returnDateDisablePrior.set_disablePriorDatesDate(pkupDate);

            //get return date text field....
            var returnDateElement = $get('ctl03_LocationTime_ReturnDateCalendar_CalendarTextBox');
            if (returnDateElement) {
                // debug: alert('the current value of return date is: '+returnDateElement.value);
                var returnDate = new Date(returnDateElement.value).localeFormat('MMM dd, yyyy');

                if ((isNaN(returnDate.valueOf())) || (returnDate <= pickupDate)) {
                    //add 1 day to return date
                    var newDate = new Date(pickupDate);
                    newDate.setDate(newDate.getDate() + 1);

                    // debug:  alert('about to set return date to: '+ newDate.toString());
                    returnDateElement.value = newDate.localeFormat('MMM dd, yyyy');
                    // debug: alert('return has been set to: '+ returnDateElement.value);
                }
                else if (returnDate <= pickupDate) {
                    //add 1 day to return date
                    var newDate = new Date(pickupDate);
                    newDate.setDate(newDate.getDate() + 1);
                    //format date same as popupcalendar property format
                    returnDateElement.value = newDate.localeFormat('MMM dd, yyyy');
                }
            }
            else {
                alert('Error occurred: Return date is undefined.');
            }
        }
    }


    function PickupLocationSelection(locationObj) {
        /// <summary> Called from LocationSearch Extender control when location selected</summary>
        ///<param name="locationObj">delimited string value of location basic info, delimiter(:::)</param>
        ///<remarks>Code, Name, IsAirportLocation, IsLRALocation, Address1, Address2, City, State, PostalCode, Country</remarks>
        var locationArray = locationObj.split(':::');
        if (locationArray && locationArray.length > 0) {
            //set return location extender and call pre load
            var rtnLocationBehavior = $find('ReturnLocationsBehavior');
            if (rtnLocationBehavior) {
                rtnLocationBehavior.preLoad(locationArray[0]);
            }
        }
    }


    function ToggleReturnLocation(ckboxObj) {
        /// <summary> Toggle display of return location text box from check box object</summary>
        ///<param name="ckboxObj">check box element object</param>
        if (ckboxObj) {
            var rtnsection = $get('ctl03_LocationTime_ReturnLocationSection');
            if (rtnsection) {
                if (ckboxObj.checked) {
                    rtnsection.style.display = 'block';
                    var rtnTextbox = $get('ctl03_LocationTime_ReturnLocationTextBox');
                    if (rtnTextbox)
                        rtnTextbox.focus();
                }
                else {
                    rtnsection.style.display = 'none';
                    var rtnBehavior = $find('ReturnLocationsBehavior');
                    if (rtnBehavior)
                        rtnBehavior.clear();
                }
            }
        }
    }
