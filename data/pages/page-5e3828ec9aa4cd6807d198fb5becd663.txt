gaSettings.setDomainName : 'none'),
// setAllowLinker comes from the settings
googleAnalyticsSetAllowLinker =
(gaSettings.setAllowLinker === 'true');
// initialize
gaManager = new DDC.GoogleAnalyticsManager(analyticsAccounts);
gaManager
.add(gaSettings.globalId)
._setDomainName(googleAnalyticsSetDomainName);
if (googleAnalyticsSetAllowLinker) {
gaManager._setAllowLinker(true);
}
if (gaSettings.trackingEnabled === true) {
// Track page view on all normal tracker objects.
gaManager
.disable(gaSettings.globalId)
._trackPageview()
// Enable only the special DDC tracker object.
.disableAll()
.enable(gaSettings.globalId);
// Track a page view to the DDC tracker object using the generated tracking
// URI, if one is available.  Otherwise, track a normal page view.
if (gaSettings.trackingUri) {
gaManager._trackPageview(gaSettings.trackingUri);
} else {
gaManager._trackPageview();
}
}
// Reenable all tracker objects.
gaManager.enableAll();
// makes global variables available
// for backwards compatibility
DDC.gaManager = gaManager;
