function check_field(form) {

  var partvalue = form.obPartNumber.value;
  var allow = "false";

  if (partvalue == "" || partvalue.length < 4) {
    alert("Please enter at least 4 characters of a product number to submit this search");
    form.obPartNumber.focus(); 
    return false; 
  }

  wildcard = 0;

  for (i=0;i<partvalue.length;i++) {
    if (partvalue.charAt(i)=="*" || partvalue.charAt(i)=="_" || partvalue.charAt(i)=="%" )
      wildcard++;
  }

  if (partvalue.length-wildcard<4) {
    alert("Please enter at least 4 characters besides wildcard symbols");
    return false;
  }
  return true;    
}

function redir() {
  if (check_field(document.form1)) {
    document.ob.action = "/obsolete/active_result.jsp";  
    document.ob.submit();
  }
}

function AutocompleteLoadData(request, response) {
  var outputArray = new Array();
  if (request && request.term) {
    var requestRegex = $.trim(request.term).replace(/[-[\]{}()*+?.,\\^$|#\s]/g, "\\$&");
    var matcher = new RegExp(requestRegex, 'i');

    for (var j=0; (j < arrData.length) && (outputArray.length < 10); j++) {
      if (matcher.test(arrData[j])) {
        outputArray.push(arrData[j]);
      }
    }
    response(outputArray);
  }
}
