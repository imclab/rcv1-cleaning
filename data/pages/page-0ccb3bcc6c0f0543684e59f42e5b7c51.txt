function changeGalleryThumbs(direction){
	totalPhotos = parseInt(photosInGallery);
	thumbsPerTab = parseInt(jQuery('#thumbnailContainer li:first').attr('thumbs'));
	page = 1;
	if (direction=="prev") {
		p = parseInt(jQuery('.thumbList li a:first').attr('idx'));
		page = p-1;
		page = (page <= 0) ? totalPhotos : page; 
	} else {
		p = parseInt(jQuery('.thumbList li a:last').attr('idx'));
		page = p+1;
		page = (page >= totalPhotos) ? 1 : page;
	}
	mod = totalPhotos % thumbsPerTab;
	numArrays = (mod==0) ? totalPhotos/thumbsPerTab : ~~(totalPhotos/thumbsPerTab) + 1; 
	for (b=1; b<=numArrays; b++){
		tmp = 'p'+b;
		window[tmp] = new Array();
	
		for (c=1; c <= thumbsPerTab; c++){
			x = (b-1)*thumbsPerTab + c;
			if (x<=totalPhotos) {window[tmp].push(x);}
		}
	}	
	range = p1;
	for (d=2; d <= numArrays; d++){
		ary = 'p'+d;
		if (jQuery.inArray(page, window[ary]) > -1) {range = window[ary];}		
	}
	updateThumbs(range);
	jQuery('#thumbsCounter').html(range[0] + " to " + range[range.length-1] + " / " + totalPhotos ); 
	jQuery('.largeThumbsCounter').html(range[0] + " to " + range[range.length-1] + " / " + totalPhotos );
}

function updateThumbs(range){
	totalPhotos = parseInt(photosInGallery);
	thumbsPerTab = parseInt(jQuery('#thumbnailContainer li:first').attr('thumbs'));
	htmlStr = "";
	if (totalPhotos < thumbsPerTab ) {
		//do not include any arrows
		htmlStr += '<ul>';
	} else if ( (parseInt(range[0]) -1) <= 0) {
		//do not include the previous arrow
		htmlStr += '<div class="thumb-right-arrow pg-right-arrow"><a href="" onclick="changeGalleryThumbs(\'next\');return false;"><div></div></a></div><ul>';
	} else if ( (parseInt(range[range.length-1])+1) > totalPhotos) {
		//do not include the next arrow
		htmlStr += '<div class="thumb-left-arrow pg-left-arrow"><a href="" onclick="changeGalleryThumbs(\'prev\');return false;"><div></div></a></div></div><ul>';
	} else {
		htmlStr += '<div class="thumb-left-arrow pg-left-arrow"><a href="" onclick="changeGalleryThumbs(\'prev\');return false;"><div></div></a></div><div class="thumb-right-arrow pg-right-arrow"><a href="" onclick="changeGalleryThumbs(\'next\');return false;"><div></div></a></div><ul>';
	}

	for (a=0; a< range.length; a++){
		htmlStr += "<li class='thumbImg'>" + jQuery('#thumbnailContainer li#thumbrailimg'+range[a]+'').html() + "</li>";
	}
	htmlStr += "</ul>";
	jQuery('.thumbList').html("<a class='close'></a>"+htmlStr);
}

function getYieldExRevenueSciValues (adHTML)
{
	var revenueScienceStr = null;
	indexOfRevenueSciKey = adHTML.indexOf(";rs=");
	lastIndexOfRevenueSciKey = 0;
	
	lastIndexOfRevenueSciKey = adHTML.indexOf(";!category");
	if(lastIndexOfRevenueSciKey <= 0)
	{
		lastIndexOfRevenueSciKey = adHTML.indexOf(";ptype");
	}
	
	if(indexOfRevenueSciKey!=-1 && lastIndexOfRevenueSciKey!=-1)
	{
		// This should provide a substring that only gives revenue science values, something like this:
		// ;rs=B10210;rs=BB10011;rs=BB10015;rs=BB10017;rs=BB10035;rs=BB10039;rs=BB10046;rs=BB10203;
		// rs=BD08734_70008;rs=BD70012;rs=BD70015;rs=BD70021;rs=BD70028;    		
		revenueScienceStr = adHTML.substring(indexOfRevenueSciKey, lastIndexOfRevenueSciKey);

		revenueScienceStr = revenueScienceStr.replace("rs=", "");
		revenueScienceStr = revenueScienceStr.replace(/;rs=/g, "^");
		revenueScienceStr = revenueScienceStr.replace(/;/g, "");
	}
				
	var revenueSciSb = "";
	if(revenueScienceStr!=null)
	{
