        for (var i = 0; i < queryString.length; i++) {
            var parts = queryString[i].split('=', 2);
            result[decodeURIComponent(parts[0])] = (parts.length == 2) ? decodeURIComponent(parts[1]) : null;
        }
        return result;
    }
