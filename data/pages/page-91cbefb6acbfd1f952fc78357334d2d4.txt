		// Popular queries
		var limit = 5;
		$('#queries').each(function () {
			var $el = $(this);
			$.ajax({
				url: '//www.google.com/cse/api/017641786003145708416/cse/rb7nojjhpfk/queries/js',
				dataType: 'JSONP',
				success: function (data) {
					var $search_url = 'http://www.vic.gov.au/search.html?q=#QUERY#';

					$list = $('<ul></ul>').addClass('horizontal');
					$.each(data.popularQueries, function (index, value) {
						var $li = $('<li />');

						$('<a></a>').text(value.query)
								.attr('href', $search_url.replace('#QUERY#', encodeURI(value.query)))
								.appendTo($li);

						// limit number of queries shown
						if ( index > limit - 1 ) {
							return;
						}

						if ( (index < data.popularQueries.length - 1) && index < limit - 1 ) {
							$('<span>, </span>').appendTo($li);
						}

						$li.appendTo($list);


					});
					$el.html($list);
				}
			});
		});
	});
