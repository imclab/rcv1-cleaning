/**
 * Clears a text form element when it has the style 'clear-default'
 */
function clickClear() {
	if ($('input.clear-default').size() > 0) {
		$('input.clear-default').each(function(index) {
			this.defaultValue = $(this).val();
			$(this).click(function() {
				if ($(this).val() == this.defaultValue) {
					$(this).val('');
				};
			}).focus(function() {
				if ($(this).val() == this.defaultValue) {
					$(this).val('');
				};
			}).blur(function() {
				if ($(this).val() == "") {
					$(this).val(this.defaultValue);
				};
			});
		});
		
		// Clear out form defaults on submit
		$('form').submit(function(event) {
			$('input.clear-default').each(function(index) {
				if ($(this).val() == this.defaultValue) {
					$(this).val('');
				};
			});
		});
	};
}

 */
function buildPullQuote () {
	$('span.pullquote-left, span.pullquote-right').each(function(index) {
		var contents = $.trim($(this).html());
		var firstCharacterCode = contents.charCodeAt(0);
		if (firstCharacterCode < 65 || firstCharacterCode > 96) {
			contents = '&hellip; ' + contents;
		};
		
		var lastCharacter = contents.charAt(contents.length - 1);
		if ("?!.".search(lastCharacter) < 0) {
			contents = contents + ' &hellip;';
		};
		var $parent = $(this).parent();
		var $pullquote = $('<div>').attr('class', $(this).attr('class')).html(contents);
		$parent.before($pullquote);
	});
}

                <h1>Arts &amp; Entertainment</h1>
                <div class="content-inner clearfix">
                    <div class="callout-right"><h4>Related Links</h4>
<ul class="arrows LI">
<li><a href="http://webdb.montgomerycollege.org/Internet/calendar/daily.cfm">Calendar of Events</a></li>
<li><a href="/edu/secondary5.aspx?urlid=15">Maps &amp; Directions</a></li>
<li><a href="/edu/tertiary2.aspx?urlid=48">Visiting MC</a></li>
</ul>
</div><p>Montgomery College boasts artistic and cultural events that are open to all students, faculty, and members of the community for participation and enjoyment. Our theatres, galleries and concert halls are among the area’s finest venues offering performances and exhibitions of work from within the college community and beyond.</p>
<h4>Theatre <br />
 </h4>
<h4><img class="image-left" title="Cultural Arts Center" alt="Cultural Arts Center" src="/assets/0/2345/2747/3066/c1068012-ff7d-40eb-9599-afb5c01e3c0b.jpg" border="0" /><a title="Cultural Arts Center at Silver Spring" href="http://cms.montgomerycollege.edu/EDU/Altsub1.aspx?id=20490" target="_blank">Cultural Arts Center at Silver Spring</a></h4>
<p>Montgomery College's Cultural Arts Center is located on the Takoma Park/Silver Spring campus at 7995 Georgia Avenue in Silver Spring, Maryland. Featured series include: Arts Alive, Arts Alive Children, World Music, and Faculty Concert Series.<br /><br />
 </p>
<ul class="focus">
<li><img class="image-left" title="Robert E. Parilla Performing Arts Center" height="86" alt="Robert E. Parilla Performing Arts Center" src="/assets/0/2345/2747/3066/538d83d9-8e64-49f5-a766-dcbca432461e.jpg" width="86" border="0" /><h4><a href="http://www.montgomerycollege.edu/PAC/">Robert E. Parilla Performing Arts Center</a></h4>
<p>In addition to performances, art exhibits and concerts presented by talented students and faculty, the Performing Arts Center plays host to notable entertainers each year. From the wise and insightful comedy of Bill Cosby to the showcase of Branford Marsalis’ mastery of jazz and classical music, the Center boasts a distinguished list of performers.</p>
</li>
<li><img class="image-left" title="Black Box Theatre" height="86" alt="Black Box Theatre" src="/assets/0/2345/2747/3066/de3f7abe-d5a4-4ed0-9704-35886a96a9f2.jpg" width="86" border="0" /><h4><a href="http://www.montgomerycollege.edu/Departments/spdnth/theatre/black%20box%20players%20page.htm">Black Box Theatre</a></h4>
<p>Our black box theatre on the Takoma Park/Silver Spring campus features a wide variety of performances throughout the year with student and community actors.</p>
</li>
</ul>
<h4><img class="image-left" title="Summer Dinner Theatre" height="86" alt="Summer Dinner Theatre" src="/assets/0/2345/2747/3066/236c7757-a78d-4f0d-80a4-79f8884e8a9c.jpg" width="86" border="0" /> <a title="Summer Dinner Theatre" href="/edu/department2.aspx?id=5103">Summer Dinner Theatre</a></h4>
<p>Students accepted into the Summer Dinner Theatre course work on all aspects of musical theater production. Together with the artistic staff, they stage two engaging musical productions in a dinner theater format. The community is invited to join us for a delicious buffet dinner before the show with food service also provided by the students. <a title="Ticket information " href="/edu/department2.aspx?id=5103">Ticket information</a><br />
 <br /><br /></p>
<h3>Music</h3>
<ul class="focus">
<li><img class="image-left" title="College Music Concert Series" height="86" alt="College Music Concert Series" src="/assets/0/2345/2747/3066/5da88707-b90e-477d-b1d1-ee601661c3a9.jpg" width="86" border="0" /><h4><a href="http://www.montgomerycollege.edu/Departments/musicrv/consched.htm">College Music Concert Series</a></h4>
<p>Throughout the year, students showcase their musical talents as part of our College Music Concert Series. Our ensembles fill the hall with jazz, wind and flute, symphony and choral music. These performances are open to the public and admission is free.</p>
</li>
<li><img class="image-left" title="Music Ensembles" height="86" alt="Music Ensembles" src="/assets/0/2345/2747/3066/ac4e544a-089c-43f2-8116-bb995fd10eec.jpg" width="86" border="0" /><h4><a href="http://www.montgomerycollege.edu/Departments/musicrv/ensembles.htm">Music Ensembles</a></h4>
<p>Our music department welcomes participation from students and members of the community. Pursue your passion for music by auditioning for one of our ensembles.</p>
</li>
</ul>

<h3>Art Galleries</h3>
<p>Art students at Montgomery College find themselves well prepared as they move on to four-year schools from our programs. Nowhere is this more evident than the walls and display cases of the many art galleries on each of our campuses. These galleries are open to the public to enjoy and explore. We also bring in special shows throughout the year to compliment the work on regular display.</p>
<p>Find out more about the galleries on each campus.</p>
