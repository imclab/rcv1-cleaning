}

function validateForm(){
	swapIndex($('#endYearID').get(0),$('#startYearID').get(0));
	swapIndex($('#minPriceID').get(0),$('#maxPriceID').get(0));
	swapIndex($('#minMileageID').get(0),$('#maxMileageID').get(0));

	return true;
}
function swapIndex(a,b) {
	var temp;
	if (a.selectedIndex > b.selectedIndex && a.selectedIndex != 0  && b.selectedIndex != 0 ) {
		temp = b.selectedIndex;
		b.selectedIndex = a.selectedIndex;
		a.selectedIndex = temp;
	}
}

function setAdType(cb) {
	var adType = '';
	if ($('#allCb').length > 0) {
		if (cb) {
			if (cb.id != 'allCb') {
				$('#allCb').removeAttr('checked');
			}
		}
		if ($('#allCb:checked').length > 0) {
			adType = "dealer new,dealer used,liner,aoe_liner,bundle,dealer certified";
			$('#newCb').removeAttr('checked');
			$('#certifiedCb').removeAttr('checked');
			$('#usedCb').removeAttr('checked');
		}
		if ($('#newCb:checked').length > 0) {
			adType = 'dealer new';
		}
		if ($('#usedCb:checked').length > 0) {
			if (adType) adType += ',';
			adType += "dealer used,liner,aoe_liner,bundle";
		}
		if ($('#certifiedCb:checked').length > 0) {
			if (adType) adType += ',';
			adType += 'dealer certified';
		}
		if (!adType) adType = 'dealer new,dealer used,liner,aoe_liner,bundle,dealer certified';
	}
	$('#adTypeID').val(adType);

	changeHash('modify');
}

function get_models(arg) {

			var model_list = response.model_list;

				for (j=0; j < model_list.length; j++) {
					$('#modelID').append(

					);
