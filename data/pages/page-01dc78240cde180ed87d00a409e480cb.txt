// Add the Type over the menu items
$.widget( "custom.catcomplete", $.ui.autocomplete, {
	_renderMenu: function( ul, items ) {
		var self = this,
			currentType = "";
		$.each( items, function( index, item ) {
			if ( item.type != currentType ) {
				ul.append( "<li class='ui-autocomplete-type'>" + item.type + "</li>" );
				currentType = item.type;
			}
			self._renderItem( ul, item );
		});
	}
});

function keywords_go(href, search) {
	if(search == 1) {
		var action = $('#form_search_keywords').attr('action');
		href = action + href;
	}
	document.location.href = href;
	return false;
}

$(function() {
	$( "#keywords").catcomplete({
		source: "/search/suggest/",
		minLength: 2,
		select: function( event, ui ) {
			if(ui.item.href) {
				keywords_go( ui.item.href );
			} else {
				keywords_go( ui.item.label, 1 );
			}
		},
		delay: 300
	});
});
