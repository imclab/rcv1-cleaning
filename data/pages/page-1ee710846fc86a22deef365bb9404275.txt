          __squid_like.like_count = new_count;

          new_count = __squid_like.like_count + 1;
          $J('.sl_count_rt').html(new_count.toString());
          $J('li.social_like').addClass('social_liked');

          __squid_like.like_count = new_count;
          __squid_like.member_likes = true;
        }

        var previousState = __squid_like.member_likes;
        __squid_like.member_likes ? dec() : inc();

        SquidgamingJS.post(
          '/utility/like_lens/' + theLensName,
          // this logic is opposite because the state class is applied above 
          { 'action': $J('li.social_like').hasClass('social_liked') ? 'like' : 'unlike' },
          function (result) { // success
            var res = $J.parseJSON(result);
            if (res.status == 'like_limit') {
              $Sq.notify.error("Hey there! You've reached the maximum number of SquidLikes for today. Come back again tomorrow to SquidLike this lens.", { hideAfter: 5000 });
              previousState ? inc() : dec();
              _gaq.push(['_trackEvent', 'Errors', 'LikeLimit', theMember.username]);
            }
            allButtons.removeClass('disabled');
          },
          'text', 
          function() { // error
            previousState ? inc() : dec();
            allButtons.removeClass('disabled');
          }
        );
      }
    } else {
      // if not logged-in, redirect to login
      // this only happens if the Squidoo.login hasn't executed
      // by the time the user clicks the button
      document.location = '/member/login/'+theLensName;
    }
  }
