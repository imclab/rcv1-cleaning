function isEmail(string) {

   if (!string) return false;
   if (!string.match('@')) return false;
   var iChars = "*|,\":<>[]{}`\';()&$#%";

   for (var i = 0; i < string.length; i++) {
      if (iChars.indexOf(string.charAt(i)) != -1)
         return false;
   }
   return true;
}

function GetAll(category, subcategory, make, model) {
    $.post("/parts/allfacet",{ 
        ajax: 'true',
        category: category,
        sub_category: subcategory,
        makeFacet: make,
        modelFacet: model,
        view_type: "print"
           }, 
        function(data){
         var options = '';
         var j;
         // alert(data);
         if (data) {
        		var j = eval("(" + data + ")");
         }
         // alert(j);
         if (j['category']) {
             options += ' <option value="">-- All Categories --</option>';
             for (var i = 0; i < j['category'].length; i++) {
                 var selected_flag = '';
                 var disabled_flag  = '';
                 if (j['category'][i].selected) {
                     selected_flag = 'selected';
                 }
                 disabled_flag = ' class="enabled" ';
                 options += ' <option value="' + j['category'][i].key + '" ' + selected_flag + disabled_flag + '>' + j['category'][i].label  + ' (' + j['category'][i].count +  ')</option>';
             }
         } else {
             options += ' <option value="">Not Category Specific</option>';
         }
         if (options) {
        	  $('#category').html(options);
         }

         var subcatoptions = '';
         if (j['subcategory']) {
             subcatoptions += ' <option value="">-- All Subcategories --</option>';
             for (var i = 0; i < j['subcategory'].length; i++) {
                 var selected_flag = '';
                 var disabled_flag  = '';
                 if (j['subcategory'][i].selected) {
                     selected_flag = 'selected';
                 }
                 disabled_flag = ' class="enabled" ';
                 subcatoptions += ' <option value="' + j['subcategory'][i].key + '" ' + selected_flag + disabled_flag + '>' + j['subcategory'][i].label  + ' (' + j['subcategory'][i].count +  ')</option>';
             }
         } else {
        	 subcatoptions += ' <option value="">Not Sub-Category Specific</option>';
         }
         if (subcatoptions) {
        	  $('#sub_category').html(subcatoptions);
         }
         var makeoptions = '';
         if (j['make']) {
             makeoptions += ' <option value="">-- All Makes --</option>';
             for (var i = 0; i < j['make'].length; i++) {
                 var selected_flag = '';
                 var disabled_flag  = '';
                 if (j['make'][i].selected) {
                     selected_flag = 'selected';
                 }
                 disabled_flag = ' class="enabled" ';
                 makeoptions += ' <option value="' + j['make'][i].key + '" ' + selected_flag + disabled_flag + '>' + j['make'][i].key  + ' (' + j['make'][i].count +  ')</option>';
             }
         } else {
        	 makeoptions += ' <option value="">Not Make Specific</option>';
         }
         if (makeoptions) {
        	  $('#make').html(makeoptions);
         }
         
         var modeloptions = '';
         if (j['model']) {
			modeloptions += ' <option value="">-- All Models --</option>';
			for (var i = 0; i < j['model'].length; i++) {
			var selected_flag = '';
			var disabled_flag  = '';

			}
		} else {
			modeloptions += '<option value="">Other / Not Model Specific</option>';
		}
 		if (modeloptions) {
 			  $('#model').html(modeloptions);
 		}
     });

}
