	function hideAvailabilityMenu(e) {
		var availabilityHeaderSectionDiv = $("#availabilityHeaderSectionDiv");
		var availabilityMenuPadding = $("#availabilityMenuPadding");
		var availabilityMenuDiv = $("#availabilityMenuDiv");
		var isHide = true;

		var leftBegin1 = availabilityHeaderSectionDiv.offset().left;
		var leftEnd1 = leftBegin1 + availabilityHeaderSectionDiv.get(0).offsetWidth;
		var topBegin1 = availabilityHeaderSectionDiv.offset().top;
		var topEnd1 = topBegin1 + availabilityHeaderSectionDiv.get(0).offsetHeight;
		if ((e.pageX >= leftBegin1 && e.pageX <= leftEnd1) && (e.pageY >= topBegin1 && e.pageY <= topEnd1))
			isHide = false;

		var leftBegin2 = availabilityMenuDiv.offset().left;
		var leftEnd2 = leftBegin2 + availabilityMenuDiv.get(0).offsetWidth;
		var topBegin2 = availabilityMenuPadding.offset().top;
		var topEnd2 = topBegin2 + availabilityMenuPadding.get(0).offsetHeight + availabilityMenuDiv.get(0).offsetHeight + 40;
		if ((e.pageX >= leftBegin2 && e.pageX <= leftEnd2) && (e.pageY >= topBegin2 && e.pageY <= topEnd2))
			isHide = false;

		if (isHide) {
			var availabilityMenuPadding = $("#availabilityMenuPadding");
			availabilityMenuPadding.hide();

			var availabilityMenuDiv = $("#availabilityMenuDiv");
			availabilityMenuDiv.hide();

			$('body').unbind('mousemove');
		}
	}

			//The vertical scroll position is the same as the number of pixels that are hidden from view above the scrollable area. 
			//If the scroll bar is at the very top, or if the element is not scrollable, this number will be 0.
			var verticalScrollTopPosition = $(window).scrollTop();

			//check if our floating div is not fully visible
			if (verticalScrollTopPosition > floatingDivTopPosition) {
				//check if height of the floating div hit the bottom of the page
				var stopScrollAtPosition = leftColumn.height() - floatingDiv.height();

				if (verticalScrollTopPosition < stopScrollAtPosition) {
					//we are ok to move the floating div. do it.
					var diff = verticalScrollTopPosition - floatingDivTopPosition;
					floatingDiv.css("position", "absolute");
					floatingDiv.css("top", diff + 5);
				}
				else {
					floatingDiv.css("position", "absolute");
					floatingDiv.css("top", stopScrollAtPosition);
				}
			}
			else {
				floatingDiv.css("position", "absolute");
				floatingDiv.css("top", 0);
			}
		}
		else {
			floatingDiv.css("position", "absolute");
			floatingDiv.css("top", 0);
		}
