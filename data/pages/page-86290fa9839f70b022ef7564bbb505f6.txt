 	onMessage: function(msg, source, origin) {
		var t = this;
		if(t.debug) {
			t.log("onMessage - " + (source ? (origin ? "postMessage" : "IEChannel") : "relay.html") + " - " + msg);
		}
		var topic = t.getVal(msg, "topic");
		var response = {};
		response.topic = topic + ".complete";
		try {
			var id = t.getVal(msg, "id"), ad = t.R.ads[id];
			var i = id.lastIndexOf("_");
			if(i > -1) {
				id = id.substring(i + 1);
			}
			var st = t.getVal(msg, "st");
			var msgID = t.getVal(msg, "m");
			if(!id || !st || !msgID) {
				t.err(200, msgID);  // no id, st, or m
			}
			if(!ad) {
				t.err(202, msgID);  // no ad
			}
			if(st != ad.st) {
				t.err(203, msgID);  // token mismatch
			}
			var ms = (new Date()).getTime(), lt = ad.lastTime, lo = ad.lastTopic;
			ad.lastTime = ms;
			ad.lastTopic = topic;
			if((lo == topic) && ((ms - lt) < 500)) {
				//t.err(204, msgID);  // duplicate message
				return;  // duplicate message--ignore
			}
			t.onMsg(topic, id, msgID, ad, msg);
		} catch(e) {
			var s = "" + e;
			if(e.description) {
				s = e.description;
//			} else if(s.indexOf("Error: ") == 0) {
//				s = s.substring(7);
			}
			t.log(s);
			// causes a bug in IE where both IEChannel and relay.html will be used
//			if(t.debug) {
//				throw e;
//			}
			response.errors = s;
		}
		if(source) {
			msg = t.toMsg(response);
			if(origin) {
				source.postMessage(msg, origin);
			} else {
				return msg;
			}
		}

			t.log("postMessage channels created");

				}

			t.initTime = s;
//			}
		} catch(e) {
			s = "" + e;
			if(e.description) {
				s = e.description;
			}
			t.log(s);
			// ignore--the fallback to relay.html will handle this
		}
