scrub = function(key, v){
if (key && v) {
switch (key) {
case 'price':
if (parseInt(v.replace('$',''), 10) > 0) { // checks to make sure price isn't 0, .replace makes sure a number is returned if price begins with '$'
v = v.split('.')[0];
v = ((isIn('$', v)) ? '' : '$') + ((isIn(',', v)) ? v : addComma(v));  // TODO: add internationalized currency
} else {
v = "";
}
default: break;
}
return v;
}
