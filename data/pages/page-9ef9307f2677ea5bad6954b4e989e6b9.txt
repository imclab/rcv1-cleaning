    $('#excerptModule').load('/books/data/excerptsModule.html?blog=ca29');

    $('#eventModule').load('/books/data/eventsModule.html?category=ca29');

    $('#journalModule').load('/journals/data/journalsModule.html?category=ca29');

    $.ajax({
      url: "/books/data/gbsModule.html",
      data: "bibkey=ISBN:9780226020969,ISBN:9780226020976,ISBN:9780226923321",
      cache: false,
      async: false,
      success: function(response)
      {
        if ( response.length > 0 )
        {
          var span = document.getElementById( 'google_preview' );
          span.style.display = 'inline';

          var gbsScript = document.createElement('script');
          gbsScript.type = 'text/javascript';
          gbsScript.appendChild (document.createTextNode('GBS_insertPreviewButtonPopup(\''+response+'\');'));
          span.appendChild(gbsScript);
          $("#__GBS_Button0").appendTo("#google_preview");
        }
      }
    });

    $(function()
    {
      $(".jacket_image").click(function() {
      if ($("#dialog").length > 0) {
        $("#dialog").dialog({
              modal: true,
        width: 550
        });
        var largeJacketWidth = $(".large_jacket").width();
  $(".ui-dialog").css({
       width: largeJacketWidth,
       marginLeft: (550 - largeJacketWidth) / 2
        });
      }
      });

      $(".dialog-close").click(function() {
        $("#dialog").dialog( "close" );
        $(".ui-dialog").css({
      marginLeft: 0
  });
      });

      $('#tab-list').tabs(
      {
        show: function(event, ui)
        {
          $("#listwrap").removeClass("show").addClass("hide");
          $("#tab-list").removeClass("hide").addClass("show");
        }
      });

    });
  </script>
