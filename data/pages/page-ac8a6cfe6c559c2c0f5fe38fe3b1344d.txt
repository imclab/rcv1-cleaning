var HelpBubble = new Class({
	
	options: {
		triggerSelector: '.helpBubble',
		helpContentSuffix: 'helpcontent',
		injectIntoId: 'main',
		parentElId: 'top',
		showCloseIcon: true,
		closeIconId: 'help-close',
		openDelay: 300,
		closeDelay: 1000,
		arrowPositionSprite: 1022,
		arrowPositionPercent: 68,
		arrowHeight: 10,
		arrowWidth: 10,
		// the left size of the popup has a fixed width:
		fixedLeftWidth: 50,
		// margin to add to the right so that it does not overlap the shaddow:
		marginRight: 50,
		noEvents : false  //Setting true does not register any events on the trigger. Will be useful if the buble is shown only once
	},
	
	Implements: [Options, Events],
	
	initialize: function(options) {
		this.setOptions(options);
		this.openDelay = null;
		this.closeDelay = null;
		this.overBox = null;
		this.openBoxId = null;
		//this.overTrig = {};
		this.parentEl = $(this.options.parentElId);
		//this.triggers = parentEl.getElements(this.options.triggerSelector);
		this.trig = this.parentEl.getElement(this.options.triggerSelector);
		//this.registerEvents();
		this.persistant = null;
		//window.addEvent('registerPreviewTriggers', (function(newParent){ console.log('here'); this.registerEvents(newParent); }).bind(this))
		if(!this.options.noEvents) {
			this.registerEvents(this.parentEl);
		} else {
			this.persistant = true;
		}
		
	},
	
	registerEvents: function(parentEl) {
		parentEl = parentEl || this.parentEl;
		var self = this;
		var trigs = parentEl.getElements(this.options.triggerSelector);
		if(trigs) {
			trigs.each(function(trig, ind){
			
				trig.addEvents({
						'mouseenter': function(e){
							$clear(self.openDelay);
							self.trig = trig;
							//self.overTrig[ind] = true;
							//if(this.openBoxId != ind) {
								self.openDelay = self.showBubble.delay(self.options.openDelay, self, [trig, ind]);
							//}
							$clear(self.closeDelay);
						},
						'mouseleave': function(e){
							if(!self.persistant) {
								$clear(self.openDelay);
								//self.overTrig[ind] = false;
								self.trig = null;
								self.closeDelay = self.closeBubble.delay(self.options.closeDelay, self);
							}
						},
						'click': function(e){
							self.persistant = true;
							$clear(self.openDelay);
							self.trig = trig;
							//self.overTrig[ind] = true;
							//if(this.openBoxId != ind) {
								self.openDelay = self.showBubble.delay(self.options.openDelay, self, [trig, ind]);
							//}
							$clear(self.closeDelay);
						}
				});
			
			});
		}
	},
	
	closeBubble: function() {
		if(!this.overBox) {
			this.hideBox();
			this.openBoxId = null;
			this.persistant = null;
		}
	},
	
	showBubble: function() {
			trig = this.trig;
			this.openBoxId = 1;
			this.makeBox();
			this.showBox();
			this.showLoading();
			this.positionBox();

			var trigId = trig.get('id');
			var contentEle = this.parentEl.getElement('#'+trigId+'-'+this.options.helpContentSuffix);
			if(contentEle) {
				var content = contentEle.get('html');
				var tmp = new Element('div', {'html': content, 'styles':{'position':'absolute', 'bottom':0, 'visibility':'hidden', 'width': '300px'} }).inject(document.body);
				var size = tmp.getSize();
				tmp.destroy();
				this.setBoxWidth(size.x);
				this.setBoxHeight(size.y); // required to stop a race condition between inserting the content and the getSize method inside positionBox which adjusts the arrow- causing the box to render incorrectly.
				this.setBoxContents(content);
				this.positionBox();
				
			}

			if(this.options.showCloseIcon) {
				this.closeIcon = new Element('div', {'html':'x', 'id':this.options.closeIconId,'styles': {'padding-top': '15px', 'padding-left':'15px', 'cursor': 'pointer'}});
				this.topright.adopt(this.closeIcon);
			}

			
			//Help bubble should close on clicking anywhere on the window except the bubble itsef and the trigger
			window.addEvent('click', function(e){
				var targ = e.target,	
					id = targ.get('id'),
					closeId = null;
				if(this.closeIcon) {
					closeId = this.closeIcon.get('id');
				}
				if((id == closeId) || (!targ.hasClass(this.options.triggerSelector.replace('.', '')) && !(this.overBox))) {
					this.overBox = null;
					this.closeBubble();
					this.persistant = null;
					window.fireEvent('help_bubble_closed', this.closeIcon);
				}
			}.bind(this));
	},
	
	showBox: function() {
		if(this.box) {
			this.box.setStyle('display', '');
		}
	},
	
	hideBox: function() {
		if(this.box) {
			this.box.setStyle('display', 'none');
		}
	},
	
	handleResults: function(res) {
		if(res.status && res.html) {
			var tmp = new Element('div', {'html': res.html, 'styles':{'position':'absolute', 'bottom':0, 'visibility':'hidden'} }).inject(document.body);
			var size = tmp.getSize();
			tmp.destroy();
			this.setBoxWidth(size.x);
			this.setBoxHeight(size.y); // required to stop a race condition between inserting the content and the getSize method inside positionBox which adjusts the arrow- causing the box to render incorrectly.
			this.setBoxContents(res.html);
			this.positionBox();
		}
	},
	
	positionBox: function() {
		if(this.box && this.trig) {
			var coords = this.trig.getCoordinates($(this.options.injectIntoId)),
				windowCoords = this.trig.getCoordinates();
				//height = this.midleft.getSize().y,
				width = this.top.getSize().x,
				previewArrowOffset = (this.options.arrowPositionPercent/100) * width,
				arrowBackgroundOffset = Math.round(this.options.arrowPositionSprite - ( previewArrowOffset )),
				//previewTop = Math.round(coords.top + (coords.height/2) - (previewArrowOffset) - (this.options.arrowHeight/2)),
				previewTop = coords.bottom;// + this.options.arrowHeight
				//previewLeft = (coords.right - this.options.arrowWidth),
				previewLeft = Math.round(coords.left + (coords.width/2) - (previewArrowOffset) - (this.options.arrowWidth/2)),
				boxSize = this.box.getSize(),
				windowx = window.getWidth(),
				//default is top and to the right, see if we should flip it horizontally:
				arrowSide = 'topcenter',
				bgPosition = '-'+arrowBackgroundOffset+'px -25px';
			
			this[arrowSide].setStyle('background-position', bgPosition);
			this.box.setStyles({'top': previewTop+'px', 'left': previewLeft+'px'});
			// this is a hack: should fix if you can figure out why the height calculation periodically gets off:
		
		}
		
	},
	
	makeBox: function() {
		if(!this.box) {
			this.box = new Element('div', {'class':'help-bubble-box-wrap', 'styles':{'z-index':9999, 'position':'absolute'} });
			this.top = new Element('div', {'class':'help-bubble-top containme'});
			this.topleft = new Element('div', {'class':'help-bubble-box top-left'});

			<div id="content-current-page" style="display:inline;"><p>Literature is an important component of a total language arts program at all grade levels because of the many benefits it offers. Here are some reasons for integrating literature into your curriculum. </p><p><em>Literature provides pleasure to listeners and readers</em>. It is a relaxing escape from daily problems, and it fills leisure moments. Making time for recreational reading and using high-quality literature help to develop enthusiastic readers and improve achievement (Block &amp; Mangieri, 2002). According to Rosenblatt (1995, p. 175), &quot;The power of literature to offer entertainment and recreation is . . . still its prime reason for survival.&quot; Developing a love of literature as a recreational activity is possibly the most important outcome of a literature program. </p><p><em>Literature builds experience. </em>Children expand their horizons through vicarious experiences. They visit new places, gain new experiences, and meet new people. They learn about the past as well as the present and learn about a variety of cultures, including their own. They discover the common goals and similar emotions found in people of all times and places. Two examples of books that provide such experiences are <em>Nory Ryan's Song</em> by Patricia Reilly Giff, a harsh survival story set in Ireland during the potato famine of 1845, and Patricia Polacco's <em>The Butterfly</em>, dealing with Nazis, resistance, and Jewish persecution during World War II. </p><p><em>Literature provides a language model for those who hear and read it.</em> Good literature exposes children to correct sentence patterns, standard story structures, and varied word usage. Children for whom English is a second language can improve their English with the interesting context, and all children benefit from new vocabulary that is woven into the stories. </p><p><em>Literature develops thinking skills.</em> Discussions of literature bring out reasoning related to sequence; cause and effect; character motivation; predictions; visualization of actions, characters, and settings; critical analysis of the story; and creative responses. </p></div>
			<div style="display:none;" id="content-next-pages" class='expandable'> 
										<p><em>Literature supports all areas of the language arts curriculum</em>. The chapter-opening classroom vignette shows how literature brings together all of the language arts. Listening to stories provides opportunities for honing listening skills, and discussion allows children to express their thoughts, feelings, and reactions. When students read literature, they are practicing their comprehension strategies in meaningful situations. Young writers may use various genres of literature as models for their own writing, and literature can be the basis for creative dramatics. Children can find stories to read and puzzles to solve on the Internet, and the computer can serve as a word processor for creating stories of their own. </p><p><em>Literature helps children deal with their problems.</em> By finding out about the problems of others through books, children receive insights into dealing with their own problems, a process called <em>bibliotherapy</em>. Children might identify with Gilly, living resentfully in a foster home in Katherine Paterson's <em>The Great Gilly Hopkins</em>, or with Mary Alice, a city girl forced to live with her grandma in a &quot;hick town&quot; in Richard Peck's <em>A Year Down Yonder.  </em></p><p><em>Picture books develop visual literacy.</em> The carefully crafted, creative illustrations in picture books develop children's awareness of line, color, space, shape, and design. Some illustrations complement or reinforce the story, whereas others enhance or extend the text. Pictures convey meaning and open new opportunities for interpretation (Giorgis et al., 1999). </p><p><em>Multicultural literature helps readers value people from different races, ethnic groups, and cultures</em>. Excellent, well-illustrated books are available for many cultural groups. Children from such populations gain self-esteem by seeing themselves represented in books, and mainstream children begin to appreciate others from culturally diverse backgrounds. </p><p><em>Literature helps establish career concepts.</em> For children who have limited knowledge of occupations, literature expands their ideas for potential careers (Harkins, 2001). Peggy Rathman's <em>Officer Buckle and Gloria</em>, about a police officer who shares information, and Alexandra Day's <em>Frank and Ernest on the Road</em>, about truck driving, give insights into two career choices. </p><p><em>Literature integrates the curriculum</em>. <strong>Trade books</strong> (books of the trade, or library books) supplement and enrich any part of the curriculum. Instead of relying solely on textbooks, look for recent, brightly illustrated books on specific topics related to your theme or subject area. Remember that textbooks are assigned, but trade books are often chosen. </p><p><em>Literature improves reading ability and attitudes</em>. A study of thirty second-, third-, fourth-, and sixth-grade classrooms by Block, Reed, and deTuncq (2003) indicated that students benefited more from twenty minutes of daily trade book or short story reading instruction. The researchers claim that reading from trade books resulted in increased reading ability, improved attitudes toward reading, and increased reading rate.  </p>

		// MULTI SELECT:
	/* originally authored by: Blaž Maležič 
	- MIT-style license
	- modified education.com

	var MultiSelect = new Class({
		
		Implements: [Options, Events], 
		
		options: {
			/* events:
			onChange:
			
			*/
			boxes: 'input[type=checkbox]', 	// checkbox selector
			labels: 'label', 				// label selector
			monitorText: ' selected',		// monitor text (localization)
			containerClass: 'checkbox-list', 	// element container CSS class
			itemExtraClass: 'containme',
			monitorClass: 'input-text fake-select input-white-shadow input-blue-border iefix',		// monitor CSS class
			monitorActiveClass: 'active',	// monitor open CSS class
			monitorHoverClass: 'hover',
			itemSelectedClass: 'selected',	// list item selected CSS class
			itemHoverClass: 'hover'			// list item hover CSS class - usually we would use CSS :hover pseudo class, but we need this for keyboard navigation functionality
		}, 
		
		initialize: function(selector, options) {
			// set options
			this.setOptions(options);
			
			// set global action variables
			this.active = false;
			this.action = 'open';
			this.state = 'closed';
			
			// get elements array
			this.elements = document.getElements(selector);
			
			// off we go...
			this.elements.each(function(element) {
				this.buildMenu(element);
			}, this);
		},
		
		buildMenu: function(element) {
			// create closure
			var self = this;
			
			// add container class (for styling)
			element.addClass(self.options.containerClass);
			
			// create item instances
			var boxes = element.getElements(self.options.boxes);
			var labels = element.getElements(self.options.labels);
			
			// list container
			var list = new Element('ul', {
				'styles': { 'display': 'none' },
				'events': {
					'mouseenter': function() { self.action = 'open'; }, 
					'mouseleave': function() { 
						self.action = 'close';
						self.itemHover(this, 'none');
					},
					
					'mousedown': function(e) { e.stop(); }, // stop text selection
					'selectstart': function() { return false; }, // stop IE text selection
					
					'keydown': function(e) {
						if (e.key == 'esc') {
							self.toggleMenu('close', monitor, this);
						}
						else if (e.key == 'down' || e.key == 'up') {
							self.itemHover(this, e.key);
						}
					}
				}
			});
			// list items
			boxes.each(function(box, i) {
				box.addEvents({
					'click': function(e) {
						e.stop();
					},
					'keydown': function(e) {
						if (e.key == 'space') {
							self.active = true;
							self.changeItemState(this.getParent(), this, monitor);
						}
						if (self.active && (e.key == 'down' || e.key == 'up')) {
							self.changeItemState(this.getParent(), this, monitor);
						}
					},
					'keyup': function(e) {
						if (e.key == 'space') {
							self.active = false;
						}
					}
				});
				var label = labels[i];
				new Element('li', {
					'class': self.options.itemExtraClass+' '+(box.get('checked') ? self.options.itemSelectedClass : ''),
					'events': {
						'mouseenter': function() {
							if (self.active === true) {
								self.changeItemState(this, box, monitor);
							}
							self.itemHover(list, this);
						},
						'mousedown': function() {
							self.active = true;
							self.changeItemState(this, box, monitor);
						}
					}
				}).adopt([box, label]).inject(list);
			});
			// list monitor
			var monitor = this.monitor = new Element('div', {
				'class': self.options.monitorClass,
				'html': '<div>' + self.changeMonitorValue(list) + '</div>',
				'tabindex': 0,
				'events': {
					'mouseenter': function() { self.action = 'open'; monitor.addClass(self.options.monitorHoverClass); }, 
					'mouseleave': function() { self.action = 'close'; monitor.removeClass(self.options.monitorHoverClass); },
					'click': function() { 
						
						if (this.hasClass(self.options.monitorActiveClass)) {
							self.toggleMenu('close', monitor, list);
						}
						else {
							self.toggleMenu('open', monitor, list);
						}
					},
					'keydown': function(e) {
						if (e.key == 'space' || e.key == 'down' || e.key == 'up') {
							self.action = 'close';
							self.toggleMenu('open', monitor, list);
						}
					},
					
					'mousedown': function(e) { e.stop(); }, // stop text selection
					'selectstart': function() { return false; } // stop IE text selection
				}
			});
			// 'global' events
			document.addEvents({
				'mouseup': function() { self.active = false; },
				'click': function() {
					if (self.action == 'close') {
						self.toggleMenu('close', monitor, list);
					}
				},
				'keydown': function(e) {
					if (e.key == 'esc') {
						self.toggleMenu('close', monitor, list);
						self.itemHover(list, 'none');
					}
					if (self.state == 'opened' && (e.key == 'down' || e.key == 'up')) {
						e.stop();
					}
				}
			});
			// replace element content
			element.empty().adopt([monitor, list]);
			//IE6 hack:
			if(Browser.Engine.trident4) {
				element.setStyles({'border':'1px solid #e6f2f8'});
			}
		}, 
		
		append: function(selector) {
			var elements = document.getElements(selector);
			this.elements.combine(elements);
			
			elements.each(function(element) {
				this.buildMenu(element);
			}, this);
		}, 
		
		changeItemState: function(item, checkbox, monitor) {
			var checked = false;
			if (item.hasClass(this.options.itemSelectedClass)) {
				item.removeClass(this.options.itemSelectedClass);
				checkbox.set('checked', false).focus();
			}
			else {
				item.addClass(this.options.itemSelectedClass);
				checkbox.set('checked', true).focus();
				checked = true;
			}
			monitor.set('html', '<div>' + this.changeMonitorValue(item.getParent()) + '</div>');
			this.fireEvent('change', [item, checkbox, checked, monitor]);
		}, 
		
		changeMonitorValue: function(list) {
			var text = list.getElements(this.options.boxes).filter(function(box) {
				return box.get('checked');
			}).length + this.options.monitorText;
			var selected = [];
			list.getElements(this.options.boxes).each(function(box) {
				if(box.get('checked')) {
					selected.push(box.get('title').capitalize());
				}
			});
			//console.log(selected);
			var total = selected.length;
			if(total == 0) {
				return 'Select Grades';
			} 
			var last = selected.pop();
			if(total == 1) {
				return last;
			}
			var text = selected.join(', ')+' and '+last;
			return text;
		}, 
		
		itemHover: function(list, select) {
			var current = list.getElement('li.'+this.options.itemHoverClass);
			
			switch (select) {
				case 'down':
					if (current && (sibling = current.getNext())) current.removeClass(this.options.itemHoverClass);
					else this.itemHover(list, 'last');
					break;
	 			case 'up':
					if (current && (sibling = current.getPrevious())) current.removeClass(this.options.itemHoverClass);
					else this.itemHover(list, 'first');
					break;
				case 'none':
					list.getElements('li.'+this.options.itemHoverClass).removeClass(this.options.itemHoverClass);
					break;
				case 'first':
					var sibling = list.getFirst();
					break;
				case 'last':
					var sibling = list.getLast();
					break;
				default:
					if (current) current.removeClass(this.options.itemHoverClass);
					var sibling = select;
					break;
			}
			
			if (sibling) 
				sibling.addClass(this.options.itemHoverClass).getElement(this.options.boxes).focus();
		},
		
		toggleMenu: function(toggle, monitor, list) {
			if (toggle == 'open') {
				monitor.addClass(this.options.monitorActiveClass);
				list.setStyle('display', '');
				this.itemHover(list, 'first');
				
				this.state = 'opened';
			}
			else {
				// close all MultiSelect menus
				monitor.removeClass(this.options.monitorActiveClass);
				this.elements.getElement('ul').setStyle('display', 'none');

				this.action = 'open';
				this.state = 'closed';
			}
			
			//if (list.getScrollSize().y > (list.getStyle('max-height').toInt() ? list.getStyle('max-height').toInt() : list.getStyle('height').toInt())) {
			//	list.setStyle('overflow-y', 'scroll');
			//}
		}
	});

						el.setStyle('line-height', '11px');
					}
				}
			}
		});
