        playList+= '<li class="' + videoPath + '"><img src="' + videoThumb + '" alt="' + videoName + '" />' + videoName + '</li>';
      } else {
        continue;
      }
    };

    playList+= '<li class="viewAll"><a href="/VideosPage"><img src="' + ContextManager.getUrl() + 'HomePage_Cagney/videoPlayer/videos_viewAll.gif" alt="View All Videos" /></a></li>';

    playList += '</ul>';
    jQuery('div.playlist').append(playList);
    jQuery('div.playlist ul li').hover(function() {
        jQuery(this).addClass('hover');
      }, function() {
        jQuery(this).removeClass('hover');
    });
    jQuery('div.playlist ul li').click(function() {
      if (!(jQuery(this).hasClass('viewAll'))) {
        var videoURL;
        if ((jQuery(this).attr('class').split(' ')[0]).indexOf('http') == -1) {
          videoURL = videoSrcPrefix + jQuery(this).attr('class').split(' ')[0];
        } else {
          videoURL = jQuery(this).attr('class').split(' ')[0];
        }
        Playback.playVideo(videoURL);
        var videoTitle = jQuery('img', this).attr('alt');
        jQuery('#videoTitle').html('Now Playing: ' + videoTitle);
        Cobalt.lexus.omniture.HomePage.video.start(videoTitle);
      }
    });
  };
