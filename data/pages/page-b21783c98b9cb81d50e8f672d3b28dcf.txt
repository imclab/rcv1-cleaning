$(document).ready(function() {

  $("#geo").smartSuggest({

  });

    src: 'svc/geoQuery?keywords=&'

  $("#geo, #keywords").blur(function() {
    $(".error").hide();
  });

  $(".geo-select").on('click', function(e) {
    if (!$(this).hasClass('geo-select-expanded')) {
      $(this).addClass('geo-select-expanded');
      $('.geo-select-flyout').css({'height': 'auto'});

      // IE7 and IE8: do not add focus to geo field when it becomes visible
      // because placeholder text is removed
      if (!$('html').hasClass('ie')) {$('#geo').focus();}

      e.stopPropagation();
      $('html').on('click.geo', function(e) {
        if ($('.geo-select').find($(e.target)).length < 1 || $('.geo-select-bd').find($(e.target)).length == 1) {
          $('#geo').blur().val('');
          $('.geo-select').removeClass('geo-select-expanded');
          $('html').off('click.geo');
        }
      });
    }
  });


  // Trigger the submit handler
  $("#geobtn").keydown(function(event) {
    if (event.keyCode == '13') {
      $(".locform").submit();
    }
  });
  
});
