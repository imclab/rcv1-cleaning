<h1 id="articletitle" class="articletitle" style="margin-top: 0px; margin-left: 0px;">
Lucretius and Shakespeare
</h1>
<div id="articleauthor" class="articleauthor">
by <span class="authorByline"><a href="/thinkers/arts_and_entertainment/literature/poetry/al-shoaf" class="peer-link { guid:'1d8aefa0-5072-4f26-b4ff-f9bc2c29394f', role:'Thinker' }" rel="author">R Allen Shoaf</a></span>
</div>
<div style="clear:both;"></div>
<div class="articlefeatureddescription">
As part of my ongoing research into the influence of Titus Lucretius Caro, author of the magnificent Latin epic, DE RERUM NATURA, on Christopher Marlowe and William Shakespeare, I am starting a Topic on Barabas, Shylock, and the &quot;materialism&quot; of the signifier within poetic and legal discourses.
</div>

function updateSimpleSearchBox( option ) {
return;
// options are imperative, like 'HIDE MINIMUM RATING', 'SHOW SAVED SEARCHES'
var $ssbox = $('#simpleSearchForm');
switch( option ) {
case 'SHOW MINIMUM RATING':
if ($ssbox.hasClass('search-no-rating')) {
$.cookie('QUICKSEARCH-MINIMUM-RATING', 'SHOW', {path:'/'}); // session cookie
$ssbox.removeClass('search-no-rating');
setTimeout( decorateSlider, 50 );   // to safely add tic marks
}
break;
case 'HIDE MINIMUM RATING':
if ($ssbox.hasClass('search-no-rating') === false) {
$.cookie('QUICKSEARCH-MINIMUM-RATING', 'HIDE', {path:'/'}); // session cookie
$ssbox.addClass('search-no-rating');
}
break;
case 'SHOW REQUIRED CREDENTIALS':
if ($ssbox.hasClass('search-no-credentials')) {
$.cookie('QUICKSEARCH-REQUIRED-CREDENTIALS', 'SHOW', {path:'/'}); // session cookie
$ssbox.removeClass('search-no-credentials');
}
break;
case 'HIDE REQUIRED CREDENTIALS':
if ($ssbox.hasClass('search-no-credentials') === false) {
$.cookie('QUICKSEARCH-REQUIRED-CREDENTIALS', 'HIDE', {path:'/'}); // session cookie
$ssbox.addClass('search-no-credentials');
}
break;
case 'SHOW SAVED SEARCHES':
if ($ssbox.hasClass('search-no-saved')) {
$.cookie('QUICKSEARCH-SAVED-SEARCHES', 'SHOW', {path:'/'}); // session cookie
$ssbox.removeClass('search-no-saved');
}
break;
case 'HIDE SAVED SEARCHES':
if ($ssbox.hasClass('search-no-saved') === false) {
$.cookie('QUICKSEARCH-SAVED-SEARCHES', 'HIDE', {path:'/'}); // session cookie
$ssbox.addClass('search-no-saved');
}
break;
default:
// proceed, if only to update the backdrop
}
if ($ssbox.hasClass('search-no-rating') && $ssbox.hasClass('search-no-credentials') && $ssbox.hasClass('search-no-saved')) {
$ssbox.removeClass('search-expanded').addClass('search-collapsed');
$ssbox.prev('.backdrop').removeClass('expanded-backdrop')
.height(0)
.hide();
} else {
$ssbox.removeClass('search-collapsed').addClass('search-expanded');
$ssbox.prev('.backdrop').addClass('expanded-backdrop')
.height($ssbox.height() + 4)
.show();
}
}
function showSimpleSearchOptions() {
return;
var $menu = $('#simpleSearchOptions');
if ($menu.is(':hidden')) {
if ($('#search-ed-menu').length === 0) {
// user is anonymous; remove Saved Search options!
$('li.saved-search-action').remove();
}
$menu.show();
$menu.unbind('mouseleave').mouseleave(hideSimpleSearchOptions);
}
}
function hideSimpleSearchOptions() {
return;
var $menu = $('#simpleSearchOptions');
$menu.unbind('mouseleave');
$menu.hide();
}
function toggleSimpleSearchOptions() {
return;
// show (or hide) a little popup menu
var $menu = $('#simpleSearchOptions');
if ($menu.is(':hidden')) {
showSimpleSearchOptions();
} else {
hideSimpleSearchOptions();
}
}
