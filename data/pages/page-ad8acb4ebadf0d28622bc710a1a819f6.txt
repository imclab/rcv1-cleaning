function truncate_list() {
   // rel tag on a list 'truncate-129' will show the first 129 elements and hide the rest
   lists = jQuery('[rel|="truncate"]');
   //console.log(lists);
   jQuery.each(lists,function(i,el) {
      which_list = i;
      too_long = false;
      l = jQuery(this);
      q = jQuery(this).attr('rel').replace(/truncate-/g,'') - 1;
      jQuery.each(jQuery(this).children(), function(i,el) {
         hidden_children = (i == q) ? "start hiding" : "" ;
         //console.log("%s : %s", i, hidden_children);
         if (i > q) {
            jQuery(this).hide().addClass("hide-me-" + which_list);
            too_long = true;
         }
      });
      if (too_long) {
         jQuery(this).append("<small><a href='javascript:toggle(\"hide-me-" + which_list + "\"," + which_list + ")' id=\"list-control-" + which_list + "\">More...</a></small>");
      }
   });
}

function truncateContent() {

   /// vars ///////////////////////////////////////////
   s          = new Array();
   s['old']   = new Array();
   s['new']   = new Array();
   s['hide']  = new Array();
   ////////////////////////////////////////////////////

   /// treat divs /////////////////////////////////////
   jQuery.each(jQuery('.scrollout'), function(i) {
      cutoff_num = jQuery(this).attr('rel');
      var l = "<small id='cnspan-"+i+"' style='font-weight:normal;font-style:normal;display:inline;'>... (<a href='javascript:void(0)' class='read-more-content' rel='" + i + "'>read more.</a>)</small>";
      s['old'][i]  = jQuery(this).html();
      s['hide'][i] = "<div id='hidden"+i+"' class='hide-me'>" + jQuery(this).html().substring(cutoff_num) + "</div>";
      s['new'][i]  = (jQuery(this).html().length > cutoff_num) ? jQuery(this).html().substring(0,cutoff_num) + l: s['old'][i];
      jQuery(this).html(s['new'][i]).attr('id','cn'+i);
   });

   /// set behavior for links ////////////////////////
   jQuery.each(jQuery('.read-more-content'), function(i) {
      jQuery(this).bind('click',function() {
         n = jQuery(this).attr('rel');
         jQuery('#cn'+n).html(s['old'][n]).css({'display':'none'}).animate({
               "height": "toggle", 
               "opacity": "toggle"
         }, "slow" );
      });
   });

   /// aaaaand out ////////////////////////////////////
   return false;
}
