  function ieValueChange(pEle, pEleParent, pHidParent, pId){
     var ele = document.getElementsByTagName(pEle);
     var hid = document.getElementsByTagName('input');
     
     for(var i =0; i<ele.length; i++){

       var selectId = ele[i].getAttribute('id')
       var parentId = ele[i].parentNode.getAttribute('id')

       if(selectId == pId && parentId == pEleParent){
         for(var k=0; k<hid.length;k++){
            var hidId = hid[k].getAttribute('id')
            parentId = hid[k].parentNode.getAttribute('id')

            if(hidId == pId && parentId == pHidParent && (hid[k].getAttribute('type') == 'hidden')){
              hid[k].setAttribute('value', ele[i].getAttribute('value'))
            }
         }
       }
     }
  }

  var success=true;
  function validatePopupPostalCode(lang){
    jQuery("#error_fr").fadeOut();
    jQuery("#error_en").fadeOut();
    //alert('validatePopupPostalCode in ' + lang);  
    var form = document.getElementById("form_recherche_popup_" + lang);
    //alert('postalcode=' + form.postal_code.value)
    if(!isValidPopupPostalCode(form.postal_code.value, lang)){
      displayError(form.postal_code, lang);
    }
    //alert('success=' + success);
    return success;
  }


  function displayError(element, lang){
    //alert(2);
    jQuery("#error_" + lang).fadeIn();
    success=false;
    //alert('2success=' + success);
  }


  function isValidPopupPostalCode(postal_code, lang)
  {
    //alert(1);
    postal_code = postal_code.replace(/\s/g,"");
    //alert('length=' + postal_code.length)
    if (postal_code.length == 6){
      pattern = new RegExp("^[a-zA-Z][0-9][a-zA-Z][0-9][a-zA-Z][0-9]$");
      if(pattern.test(postal_code)) {
        success=true;
        return true;
      } else {
        //alert('regexp failed')
        return false;
      }
    } else {
      //alert('length failed')
      return false;
    }
  }
</script>
