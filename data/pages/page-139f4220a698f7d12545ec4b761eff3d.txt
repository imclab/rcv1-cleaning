	$("#stdNavSubmit").click(function(event) {
		
		var startDate = $("input#n_start_date").val();
		var endDate = $("input#n_end_date").val();
		
		//parse dates and create date objects
		var startDateSplits = startDate.split("/");
		var firstDate = new Date(startDateSplits[2],(startDateSplits[0]-1),startDateSplits[1]);
		
		var endDateSplits = endDate.split("/");
		var secondDate = new Date(endDateSplits[2],(endDateSplits[0]-1),endDateSplits[1]);
		
		// time from today
		var today = new Date();
		today.setHours(0);
		today.setMinutes(0);
		today.setSeconds(0);
		
		// days in advance from today to firstDate
		var days = dateDiff(today, firstDate);
		
		//start and end date formate
		var start = firstDate.getFullYear() + "-" + (firstDate.getMonth()+1) + "-" + firstDate.getDate();
		var end = secondDate.getFullYear() + "-" + (secondDate.getMonth()+1) + "-" + secondDate.getDate();
		
		// date range being searched
		//alert(start + " to " + end);
		trackSTDateRange("Save the Date: Nav", start, end, days);
	});

	            					<p>
	Summer in New York City is paradise for the stingy cinephile. In addition to new releases at multiplexes and art houses throughout the five boroughs, the warmer months bring free outdoor screenings of huge hits and cult classics, shown in some of the City&#39;s most striking settings. These include the foot of the Brooklyn Bridge, a 30-foot-high plaza in the thick of the Lower Manhattan skyline and even&mdash;why not?&mdash;the deck of an aircraft carrier.</p>
