var imgRotator = {
	autoSlideShow : true,
	currentSlide : 0,
	dateToday : new Date().setHours(0, 0, 0, 0),
	imgArr : [],
	promoArr : [],
	overrideArr : [],
	assetsUrl : "http://assets.cobaltnitra.com/teams/repository/export/bf5/88f00fb2b100484b410145edef087/bf588f00fb2b100484b410145edef087.js",

	beginShow : function () {
		"use strict";
		jQuery.extend(true, {}, this).init(1301438021);
	},
	init : function (uniqueId) {
		"use strict";
		var assetRotator = this;
		jQuery("img.lex_imgRot_loader1301438021").fadeIn(500);
		jQuery.getScript(this.assetsUrl, function () {
			assetRotator.trimArray();
		});
	},
	trimArray : function () {
		"use strict";
		var assetRotator = this,
			promoList = ContextManager.getPromotions(),
			groupList = ContextManager.getSiteGroups(),
			inventoryCheck = ContextManager.isNewInventory;

		/* Get a copy of the config object */
		assetRotator.config = jQuery.extend(true, {}, imgRotatorConfig1);

		jQuery.each(assetRotator.config.promos, function (i, item) {

			var showThisSlide = assetRotator.checkDate(item.startDate, item.endDate),
				selection = item.newOrUsed;

			/* Check to see if this is a "new inventory" promo */
			if (!assetRotator.isEmpty(item.newOrUsed)) {

				if (inventoryCheck === "true" && selection === "new") {
					showThisSlide = true;
				} else if (inventoryCheck === "false" && selection === "used") {
					showThisSlide = true;
				} else {
					showThisSlide = false;
				}
			}

			/* Check if promo contains override, if so add to the override array */

			if (!assetRotator.isEmpty(item.override) && (showThisSlide === true)) {
				if (item.override instanceof Array) {
					assetRotator.overrideArr.concat(item.override);
				} else {
					assetRotator.overrideArr.push(item.override);
				}
			}

			/* Check if this slide is in ContextManager.promotions */

			if (item.promotion) {
				showThisSlide = false;
				if (!assetRotator.isEmpty(promoList)) {
					jQuery.each(promoList, function (e, pList) {
						if (pList.title === item.promotion) {
							showThisSlide = true;
							return false;
						} else {
							showThisSlide = false;
						}
					});
				}
			}

			/* Check if this slide is in ContextManager.siteGroups */

			if (item.siteGroup) {
				showThisSlide = false;
				if (!assetRotator.isEmpty(groupList)) {
					jQuery.each(groupList, function (g, gList) {
						if (gList === item.siteGroup) {
							showThisSlide = true;
							return false;
						} else {
							showThisSlide = false;
						}
					});
				}
			}

			if (showThisSlide === false) {
				assetRotator.config.promos[i] = null;
			} else {
				assetRotator.promoArr.push(item);
			}
		});

		/* Check for overrides */

		if (assetRotator.overrideArr.length > 0) {
			jQuery.each(assetRotator.config.promos, function (i, item1) {
				if (jQuery.inArray(assetRotator.overrideArr, item1) > -1) {
					item1 = null;
				}
			});
		}
		assetRotator.loadHtml();
	},
	checkDate : function (start_date, end_date) {
		"use strict";

		if (this.isEmpty(start_date)) {
			start_date = this.dateToday;
		} else {
			start_date = this.scrubDate(start_date);
		}

		if (this.isEmpty(end_date)) {
			end_date = this.dateToday;
		} else {
			end_date = this.scrubDate(end_date);
		}

		if ((this.dateToday < start_date) || (this.dateToday > end_date)) {
			return false;
		}
		return true;
	},
	scrubDate : function (d) {
		"use strict";
		var year = d.split("/"),
			date;

		if (year[2].length < 4) {
			year[2] = "20" + year[2];
		}

		date = new Date(Number(year[2]), Number(year[0]) - 1, Number(year[1]));
		return (date);

	},
	loadHtml : function () {
		"use strict";
		var promoString = "",
			assetRotator = this;

		jQuery.each(assetRotator.promoArr, function (i, item) {
			var urlString = "",
				fallBackHtml;

			if (item === null) {return; }

			if (!assetRotator.isEmpty(item.url)) {
				if (item.tagNgo) {
					if (!assetRotator.isEmpty(item.tb)) {
						urlString = "style='cursor:pointer' class='lslp_stop_show' onclick='imgRotator.tag(\"" + item.pxTag + "\"); imgRotator.tb(\"" + item.url + "\"," + item.tb.width + "," + item.tb.height + ")'";
					} else {
						urlString = "href='" + item.url + "' target='_blank' onclick='imgRotator.tag(\"" + item.pxTag + "\")'";
					}
				} else {
					if (item.url.search("\\?") === -1) {
						urlString = "href=" + item.url + "?cs:a:i=" + item.pxTag;
					} else {
						urlString = "href=" + item.url + "&cs:a:i=" + item.pxTag;
					}
				}
			}
			if (item.img.indexOf(".swf") !== -1) {

				fallBackHtml = "<div><img src='" + item.swfFallBack + "' alt='" + item.imgAlt + "' /></div>";

				if (assetRotator.isEmpty(item.swfHeight)) {
					item.swfHeight = '123';
				}
				promoString += "<div class='lex_imgRot_slide_container" + 1301438021 + "' rel='" + item.duration + "' style='background:" + item.swfBg + ";'><object title='" + item.imgAlt + "' codebase='http://download.macromedia.com/pub/shockwave/cabs/flash/swflash.cab#version=6,0,40,0' width='" + item.swfWidth + "' height='" + item.swfHeight + "' data='" + item.img + "' type='application/x-shockwave-flash'><param name='allowScriptAccess' value='always' /><param name='allowFullScreen' value='false' /><param name='wmode' value='transparent' /><param name='movie' value='" + item.img + "' /><param name='quality' value='high' />" + fallBackHtml + "</object></div>";

			} else if (assetRotator.isEmpty(item.url)) {

				promoString += "<div class='lex_imgRot_slide_container" + 1301438021 + "' rel='" + item.duration + "'><img src='" + item.img + "' alt='" + item.imgAlt + "' /></div>";

			} else {

				promoString += "<div class='lex_imgRot_slide_container" + 1301438021 + "' rel='" + item.duration + "'><a " + urlString + "><img src='" + item.img + "' alt='" + item.imgAlt + "' /></a></div>";
			}

			if (!assetRotator.isEmpty(item.img)) {
				assetRotator.imgArr.push(item.img);
			}
		});

		if (assetRotator.promoArr.length === 1) {
			assetRotator.autoSlideShow = false;
		}

		jQuery("div.lex_imgRot_slides" + 1301438021).html(promoString);

		if (assetRotator.imgArr > 1) {
			assetRotator.preLoadImages(assetRotator.imgArr);
		} else {
			assetRotator.style();
		}
	},
	preLoadImages : function (imgs) {
		"use strict";
		var pic = [],
			loaded = 0,
			total = imgs.length,
			assetRotator = this;

		function onLoadCheck() {
			loaded += 1;
			if (loaded === total) {
				assetRotator.style();
			}
		}

		jQuery.each(imgs, function (i, img) {
			if (img.indexOf(".swf") !== -1) {
				/* Skip the preload if the asset is a SWF */
				total -= 1;
			} else {
				pic[i] = new Image();
				pic[i].onload = onLoadCheck;
				pic[i].src = img;
			}
		});

		if (total === 0) { assetRotator.style() };
	},
	style : function () {
		"use strict";

		jQuery("img.lex_imgRot_loader1301438021").css('display', 'none');
		this.fadeIn();

	},
	fadeIn : function () {
		"use strict";

		this.pause();

	},
	fadeOut : function () {

		if (this.autoSlideShow) {
			this.currentSlide += 1;
		}

		if (this.currentSlide === this.config.promos.length) {
			this.currentSlide = 0;
		}

		this.fadeIn();
	},

			assetRotator = this;

		if (this.autoSlideShow) {

			jQuery("div#lex_imgRot_wrapper1301438021").animate({'opacity': '100'}, dur, function () {
				assetRotator.fadeOut();
			});
		}
	},
	isEmpty : function (testVar) {
		"use strict";
		return ((testVar === undefined)
			|| (testVar === null)
			|| (testVar === "null")
			|| (testVar === "")
			|| (testVar === "undefined"));
	},
	tag : function (pixelTagId) {
		"use strict";
		var promoPixelTag = {

		};

			publisherData: promoPixelTag
		});
	},

		jQuery("#lslpIframe").css({'height': '100%', 'width': '100%'});
	},
	tb_destroy : function () {
		"use strict";
		jQuery('#lslpTbOverlay, #lslpTbContainer').fadeOut(250, function () {
			jQuery(this).remove();
		});
	}
};
