  function has_quicktime()
  {
	  var haveqt = false;
	  try
	  {
		  if (navigator.plugins) 
		  {
		      for (i=0; i < navigator.plugins.length; i++ ) 
			  {
		          if (navigator.plugins[i] && 
				        navigator.plugins[i].name && 
				          navigator.plugins[i].name.indexOf("QuickTime") >= 0)
		          { 
			          haveqt = true; 
			      }
		      }
		  }
	
		  if ((navigator.appVersion.indexOf("Mac") > 0)
		      && (navigator.appName.substring(0,9) == "Microsoft")
		      && (parseInt(navigator.appVersion) < 5) )
		  { haveqt = true; }
	  }
	  catch (err)
	  {
		  // got this once: "Error: TypeError: navigator.plugins[i] is undefined" (??)
	  }
	  
	  return haveqt;
  }
  
  function wizard_validation_xv(submitting)
  {
	  try
	  {
		    var us_based = false;
		  
		    for (var q = 0; q < document.forms["fyaswizard"].elements.length; q++)
			{

  function check_xv (param, value)
  {
	  // we do this since we can't have a synchronous cross-domain JSONp call
	  var xmlHttp = new XMLHttpRequest();
	  xmlHttp.open("GET", "/xv.php?p=" + param + "&v=" + encodeURIComponent(value), false);
	  xmlHttp.send(null);
	  
	  xmlDoc = xmlHttp.responseXML;

	  if (xmlDoc == null)   // Firefox
	  {
		  var parser = new DOMParser();
		  var xmlDoc = parser.parseFromString(xmlHttp.responseText, "application/xml");
	  }

	  //console.log(xmlHttp.responseText);
	  
	  var error = xmlDoc.getElementsByTagName("error")[0].textContent;
	  var message = xmlDoc.getElementsByTagName("message")[0].textContent;

	  if (error == null)
	  {
		  // probably IE8, use the IE-only "text" property
		  error = xmlDoc.getElementsByTagName("error")[0].text;
		  message = xmlDoc.getElementsByTagName("message")[0].text;
	  }

	  if (error == "1")
	  {
		  return message;
	  }

	  
	  return null;
  	
  }
  
  function wizard_validation(submitting)
  {
	  try
	  {
			var i = 0;
			
			for (i = 0; i < document.forms["fyaswizard"].elements.length; i++)
			{
				var element = document.forms["fyaswizard"].elements[i];
				
				if (element.type == 'text' || element.type == 'select-one')
				{
					var value = element.value; 
					value = value.trim(); 
					var required = element.getAttribute("req");
					
					if (value.length == 0 && required == 1)  
					{
						var label = get_label(element.getAttribute("id"));
						alert("The \"" + label + "\" field is required."); 
						if (!submitting)
						{
							// A hack, to deal with Drupal's incredibly annoying form input-validation problems
						    //document.forms["fyaswizard"]["validation"].value = 0;
						    //document.forms["fyaswizard"].elements["validation"].value = 0;
							var validation_element = document.forms["fyaswizard"].elements["validation"]; //document.forms["fyaswizard"]["validation"];
							validation_element.setAttribute('value', '0');
						}
						
						return false;
						
					}
					
					if (value.length == 0 && required == 0)  
					{
						continue;
					}
					
					var validation = element.getAttribute("validate");
					
					if (validation != null)
					{
					
						var pattern = new Array();
						switch(validation)
						{
							case "user@host.com":
							  pattern[0] = "^[-\._	A-Za-z0-9]+@[-A-Za-z0-9]+\.[A-Za-z]{2,4}$";
							  break;
							case "mm/yyyy":
							  pattern[0] = "^(0[1-9]|1[012])[- /.](19|20)\d\d$";
							  break;
							case "mm/dd/yyyy":
							  pattern[0] = "^(0[1-9]|1[012])[- /.](0[1-9]|[12][0-9]|3[01])[- /.](19|20)\d\d$";
							  break;
							case "xxx-xxx-xxxx":
							  pattern[0] = "^[0-9]{3}-[0-9]{3}-[0-9]{4}$";
							  pattern[1] = "^[0-9]{10}$";
							  break;
							default:
							  // we should not get here
							  continue;
						}
	
						var passed = false;
						
						for (var n = 0; n < pattern.length; ++n)
						{
							var regexp = new RegExp(pattern[n], "");
							
							if (regexp.test(value) == true)
							{
								passed = true;
							}
						}
	
						// extra phone validation:
						var custom_err_msg = null;
						if (passed && validation == "xxx-xxx-xxxx")
						{
							var exchIdx = 3;
							if (isNaN(value.substring(3,4)))
							{
								exchIdx = 4;
							}
	
							if (value.substring(0,1) == value.substring(1,2) 
									&& value.substring(1,2) == value.substring(2,3)
									&& value.substring(0,3) != "888")
							{
								passed = false;
								custom_err_msg = "The area code entered appears to be invalid.";
							}
							else if (value.substring(exchIdx,exchIdx + 1) == value.substring(exchIdx + 1, exchIdx + 2) 
									&& value.substring(exchIdx + 1, exchIdx + 2) == value.substring(exchIdx + 2, exchIdx + 3))
							{
								passed = false;
								custom_err_msg = "The phone number entered appears to be invalid.";
							}
							
						}
	
						
						if (!passed)
						{
							var label = get_label(element.getAttribute("id"));
							if (custom_err_msg)
								alert(custom_err_msg);
							else
								alert("The \"" +  label + "\" field must be entered in the following format: " + validation + ".");
	
							if (!submitting)
							{
								// A hack, to deal with Drupal's incredibly annoying form input-validation problems
								var validation_element = document.forms["fyaswizard"].elements["validation"]; //document.forms["fyaswizard"]["validation"];
								validation_element.setAttribute('value', '0');
							}
							return false;
						}	
					}
				}
			}
	  }
	  catch (err)
	  {
		  var xmlHttp = new XMLHttpRequest();
		  xmlHttp.open("GET", "/msg?e=" + encodeURIComponent(err.toString()), false );
		  xmlHttp.send(null);
	  }
	  
	  return true;
  }
	
  function get_label(element_name)

  	var labels = document.getElementsByTagName('LABEL');

		    }
		}
		return null;
	}

	<script type="text/javascript" >
		
		var curStart = 1;
		
		var sortProp = null;
		var sortAsc = true;
		
		function changePage (start)
		{
		
			if (start < 1 || start > totSchools)
				return;
				
			curStart = start;
				
			filterSchools();
			
			//var table = document.getElementById('schools_table');
			//table.src = "/resource/widget/inner?start=" + curStart + "&count=50&seed=2Fep%2Fperforming-arts-schools";
			
			//table.outerHTML = table.outerHTML.replace(/data="(.+?)"/, 'data="' + url + '"');
		}
		
		function filterSchools()
		{
			//var table = document.getElementById('schools_table');
			document.body.style.cursor = 'wait';
		
			var seed = encodeURIComponent('2Fep%2Fperforming-arts-schools');
		
			var url = "/resource/widget/inner?start=" + curStart + "&count=10&seed=" + seed;
			
						
				url = url + "&program=Performing Arts";
			
						
						
			
			var props = new Array('programs','offering_cat','state_abbr','tot_students','locale_cat','admin_cat');
			
			for (var i = 0; i < props.length; ++i)
			{
				var prop_name = props[i];
				
				var q = 1;
				var z = 0;
				var option = document.getElementById(prop_name + q);
				while (option != null)
				{
					if (option.selected)  
					{
						url += ("&" + prop_name + "[" + z + "]=" + encodeURIComponent(option.value));
						++z;
					}
					
					++q;
					option = document.getElementById(prop_name + q);
				}
				
			}
			
			// sort if enabled
			if (sortProp != null)
			{
				url += ('&sort=' + encodeURIComponent(sortProp));
				if (sortAsc)
					url += '&dir=asc';
				else
					url += '&dir=desc';
			}
			
			//console.log(url);
			
			//var table = document.getElementById('schools_table');
			//table.src = url;
			
			$("#schools_table").load(url);
		}
		
		function sortSchools (prop)
		{
			if (sortProp != null)
			{
				// the previous sort
				
				if (sortProp != prop)
				{
					var icon_txt = '►';
					var e_orig = document.getElementById(sortProp + '_sort_icon');
					var e_new = e_orig.cloneNode(false);
					
					var e_txt = document.createTextNode(icon_txt);
					e_new.appendChild(e_txt);
					
					e_orig.parentNode.replaceChild(e_new, e_orig);
				}
				
			}
			
			sortProp = prop;
			
			var e_orig = document.getElementById(prop + '_sort_icon');
			
			var icon_txt = '▲';
			sortAsc = false;
			
			if (e_orig.firstChild.nodeValue == '▲' || e_orig.firstChild.nodeValue == '►')
			{
				icon_txt = '▼';
				sortAsc = true;
			}
				
				
			var e_new = e_orig.cloneNode(false);
			var e_txt = document.createTextNode(icon_txt);
			e_new.appendChild(e_txt);
			e_orig.parentNode.replaceChild(e_new, e_orig);
			
			filterSchools();
		}
		
		function tableLoaded (tot)
		{
		
			var table = document.getElementById('schools_table');
			totSchools = tot; //table.contentWindow.getTotSchools();
			
			
			var update = document.getElementById('set_school_count_descr');
			if (totSchools == 0)
			{
				update.innerHTML = "No schools found for the filters selected.";
			}
			else
			{
				var curEnd = curStart + 9;
				if (curEnd > totSchools)
				{
					curEnd = totSchools;
					//curStart = totSchools - 49;
					if (curStart < 0)
						curStart = 1;
				}
				update.innerHTML = "Viewing schools " + curStart +  " to " + curEnd + " out of " + totSchools + " total";
			}
			
			// add city filter
			var stateSelected = false;
			var q = 1;
			var z = 0;
			var option = document.getElementById('state_abbr' + q);
			while (option != null)
			{
				if (option.selected)  
				{
					stateSelected = true;
					break;
				}
				
				++q;
				option = document.getElementById('state_abbr' + q);
			}
			
			var citySelector = document.getElementById('city_filter');
			
			if (stateSelected)
			{
				// add the city selector
				/*var cities = $("#schools_table").contentWindow.getSchoolCities();
				for (var i = 0; i < cities.length; ++i)
				{
				   var option = document.createElement("option");
				   option.id = "city" + i;
				   option.name = "city" + i;
				   option.value = cities[i];
				   option.text = cities[i];
				   citySelector.add(option, null);
				}
				*/
				document.getElementById('city_filter_div').style.visibility = 'visible';
			}
			else
			{
				// remove the city selector
				for (var i = 0; i < citySelector.size; ++i)
					citySelector.remove(i);
				document.getElementById('city_filter_div').style.visibility = 'hidden';
			}
			
			document.body.style.cursor = 'default';
		}
