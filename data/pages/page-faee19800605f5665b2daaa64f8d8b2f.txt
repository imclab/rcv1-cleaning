<script type="text/javascript">
$(document).ready(function(){
  var currentPosition = 0;
  var slideWidth = 143;
  var slides = $('.slide');
  var numberOfSlides = 20;
  var visible = Math.floor($('#carousel .videoCarousel').width() / slideWidth);
  var step_by = 4;

  // Remove scrollbar in JS
  $('#carousel').css('overflow', 'hidden');

  // Set #slideInner width equal to total width of all slides
  $('.new').css('width', slideWidth * numberOfSlides);

  // Insert left and right arrow controls in the DOM
  $('#carousel')
    .prepend('<span class="control" id="leftControl">Move left</span>')
    .append('<span class="control" id="rightControl">Move right</span>');

  // Move the carousel
  function move_carousel(){
      $('.new').animate({
            'marginLeft' : slideWidth*(-currentPosition)
      });
      // Set arrow status
      if(currentPosition <= 0){
        $('#leftControl').addClass('inactive');
      }else{
        $('#leftControl').removeClass('inactive');
      }
      if(currentPosition+visible >= numberOfSlides){
        $('#rightControl').addClass('inactive');
      }else{
        $('#rightControl').removeClass('inactive');
      }
  };

  // Create event listeners for .controls clicks
  $('.control')
    .bind('click', function(){
        // Determine new position
        if($(this).attr('id') === 'rightControl'){
            currentPosition = Math.min(numberOfSlides-visible, currentPosition + step_by);
        }else{
            currentPosition = Math.max(0, currentPosition - step_by);
        }
        // Move slideInner using margin-left
        move_carousel();
    });

  // see if there's a video in the carousel that matches the active video
  // if yes, change the li class to be current
  var currentVideoTitle = $.trim($('#natgeov-vtitle').text()),
      $vidthumbTitle = $(".videoCarousel li").find('a[title="'+ currentVideoTitle + '"]');
  if($vidthumbTitle.length){
      $vidthumbTitle.parent('li').addClass('current');
  }

  // Start carousel with the current video in the center
  currentPosition = Math.max(0, $(".videoCarousel li").index($("li.current"))-Math.floor(visible/2));
  move_carousel();

  });
</script>
