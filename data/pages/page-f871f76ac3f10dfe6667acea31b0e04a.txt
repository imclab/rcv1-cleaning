  function str_replace (search, replace, subject, count) {
    var i = 0,
        j = 0,
        temp = '',

		// there is no any specific way to detect chrome browser.
		//var url = Get_Cookie("bookmark");
		del_cookie("bookmark", '/', '');
		parent.window.opener.location.href = url;
		
	}else if($.browser.msie){
		// no need to do any thing already event is attached in popup function.
	}else{
		try{
			//window.parent.location.href=document.referrer;
			//var url = Get_Cookie("bookmark");
			del_cookie("bookmark", '/', '');
			parent.window.opener.location.href = url;
		}catch(e){
			//alert(e.message);
			parent.window.opener.location.href = 'http://www.zigwheels.com';	
		}
	}self.close(); 
}
function redirectUrl(url){
	if($.browser.msie){
		Set_Cookie("bookmark",url,'','/','','');
		self.close();
	}else{
		if(checkZWLoginStatus()==1){window.parent.location.href=url;}
		self.close();
	}
}

function ifTwitterUserExit(tw_user_id){  
	jQuery.post('http://www.zigwheels.com/2db/validateuserReview.php',{"twuserid":tw_user_id,"submittype":"IsUserExist"},function(data){  
		dataarray=data.split('_@@@_');
		if(dataarray[0]==1){
		   //ALREADY LINKED TWITTER ACCOUNT WITH ZIGWHEELS DO LOGIN
			doAutoLogin()
			//getTwitterLoginScreen(tw_user_id,dataarray[5],'1');			
		}else if(dataarray[0]==0){
		  /// REGISTER USER SCREEN
			getTwitterLoginScreen(tw_user_id,dataarray[5]);
		}else if(dataarray[0]==3){
				jQuery('#msgholder').removeClass("error");
				jQuery('#msgholder').html("You are already registered with zigwheels.").show();	
				setTimeout(function() {jQuery('#msgholder').hide();jQuery('#msgholder').removeClass("info");jQuery('#msgholder').addClass("error");}, 5000);				
				return false;
		}else{
			getTwitterLoginScreen(tw_user_id,'');
		}
	});
} 

function doAutoLogin(reviewid){
	var fromIFrame=1;
	if(reviewid==0 || jQuery.trim(reviewid)=='undefined'){ 
		var url = Get_Cookie("bookmark");

	var flag=0;
	if( loginid == "") {
		message="Please enter email ";
		jQuery('#msgholder').html(message).show();
		jQuery('#login').focus();
		flag=1;
		return false;
	}
	else if(!isValidEmail(loginid))	{
		message="Please enter a valid Email Address.";
		jQuery('#msgholder').html(message).show();
		jQuery('#login').focus();
		flag=1;
		return false;

		message="Please enter your password .";
		jQuery('#msgholder').html(message).show();
		jQuery('#passwd').focus();
		flag=1;
		return false;
	}
	jQuery('#msgholder').hide();
	
	return true;
}

function signOut(){
	if(Get_Cookie("fb_user_id") != ''){
		deleteFacebookCookies();
		FB.getLoginStatus(function(response) {
			if (response.status === 'connected'){
				FB.logout(function(){});
			}
			else{
				FB.logout(function(){});
				}
			});
			setTimeout(function() {
					window.location.href='http://www.zigwheels.com/logout.php';
			}, 1000); 
			//showWelcome()
	}else{
		deleteFacebookCookies();

		window.location.href='http://www.zigwheels.com/logout.php';
	}
}

function getTimeline(val, offset){
	var placeholder_id;
	if(val == 'me'){
		placeholder_id =  'yourfeed';
		var id_next_offset = 'your_feed_next';
		var id_prev_offset = 'your_feed_prev';
		var id_user_name = 'user_name_holder';
	}else{
		placeholder_id =  'friendfeed';
		var id_next_offset = 'friend_next_offset';
		var id_prev_offset = 'friend_prev_offset';
		var id_user_name = 'friend_user_name_holder';
	}
			
		$('#friend_user_name_holder').html('');
		$('#' + placeholder_id).html("<div align=\"center\"><img src=\"http://static.zigwheels.com/images/fb-loader.gif\"/></div>");	
	
		$.post('/submit_to_fb_timeline.php', { 'action' : 'read', 'offset' : offset, 'val' : val }, function(data){
		
		///remove del class for other
		if(val != 'me'){
			$(data).remove('.del');			
		}
				
		$('#' + placeholder_id).html(data);
		var user_name = $('#' + placeholder_id + ' #user_name_fb').val();
		if(val != 'me' && user_name != "" && user_name != undefined){
			$('#' + id_user_name).html(user_name + ',');
		}

		var next_offset = $('#next_offset').val(); //alert(next_offset);
		if(next_offset != ""){
			$('#' + id_next_offset).attr('href', 'javascript:getTimeline("'+ val +'", '+ next_offset +')');
			$('#' + id_next_offset).removeClass('disabled');
		}else{
			$('#' + id_prev_offset).attr('href', 'javascript:void(0)');
			$('#' + id_prev_offset).addClass('disabled');
		}

		var prev_offset = $('#prev_offset').val(); //alert(prev_offset);
		if(prev_offset != ""){
			$('#' + id_prev_offset).attr('href', 'javascript:getTimeline("'+ val +'", '+ prev_offset +')');
			$('#' + id_prev_offset).removeClass('disabled');
		}else{
			$('#' + id_prev_offset ).attr('href', 'javascript:void(0)');
			$('#' + id_prev_offset).addClass('disabled');
		}
	});
}

function isAlphabeticForFreePriceQuote(str) {
	var bReturn = true;
	var valid=" abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ_-";
	var invalidfirst = "_-";
	var invalidlast = "_-";
	for (var i=0; i<str.length; i++) 
	{
		if ( i == 0 && (invalidfirst.indexOf(str.charAt(i)) > 0))
		{
			bReturn = false;
			break;
		}
		else if ( i == (str.length-1) && (invalidlast.indexOf(str.charAt(i)) > 0))
		{
			bReturn = false;
			break;
		}
		else if (valid.indexOf(str.charAt(i)) < 0)
		{
			bReturn = false;
			break;
		}
	}
	return(bReturn);
}
