    function UpdateZipOverlay() {
        if (!KBB.cookies.enabled()) { return; }

        var zip = $("#enterzipinput input[id='selectedZipCode']").val();

        if (!KBB.zip.validateFormat(zip)) {
            $('#enterzip #enterziperror').html('The 5-digit ZIP code format was incorrect.');
            $('#enterzip #enterziperror').show();
            return;
        }

        var isZipCodeSet = KBB.zip.set(zip);
        
        if (!isZipCodeSet) {
            $('#enterzip #enterziperror').html('Please enter a valid ZIP Code.');
            $('#enterzip #enterziperror').show();
            return;
        }

        var redirectUrl = ''.replace(/zipcode=\d+/i, 'zipcode=' + zip);
        // randomizer only needed for overlay so that when page reloads w/o overlay, it is not cached.  
        // overlay does not pass redirectUrl
        if (redirectUrl == "") {
            //window.location.reload(true);
            if (isZipCodeSet) {
                var cacheRandomizer = Math.random() * 1000000000000000000;
                location.href = jQuery.format("{0}{1}{2}r={3}", location.pathname, location.search, location.search == "" ? "?" : "&", cacheRandomizer);
            }
            else {
                location.search = location.search.replace(/zipcode=\d+/i, 'zipcode=' + zip);
                location.href = jQuery.format("{0}{1}", location.pathname, location.search);
            }
        } else {
            if (isZipCodeSet)
                location.href = decodeURIComponent(redirectUrl.replace("JSZIPCODEREPLACE", zip));
            else {
                if (redirectUrl.indexOf('/sitemap/') == -1)
                    location.href = redirectUrl;
            }
        }
    }
