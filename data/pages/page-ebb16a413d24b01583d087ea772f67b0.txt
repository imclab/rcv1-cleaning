<script type="text/javascript">
(function($) {
	var menu = $('#navbar');
	var over = 0;
	menu.find('> .menu_item > .menu_tab').bind('click mouseover', function(e) {
		if ($(this).siblings('.submenu').length == 0) return;
		menu.find('.submenu').hide();
		if (over > 0) {
			over = 0;
		} else {
			$(this).parent().find('.submenu').show();
			over++;
		}
		$(this).parent().addClass('active');
	});
	
	menu.find('> .menu_item > .menu_tab').bind('mouseout', function(e) {
		if (over > 0) {
			over = 0;
		}
		$(this).parent().removeClass('active');
		setTimeout(function() { if (over == 0) menu.find('.submenu').hide(); }, 0);
	});
	
	menu.find('.submenu').bind('mouseover', function(e) {
		over++;
		$(this).parent().addClass('active');
	});
	
	menu.find('.submenu').bind('mouseout', function(e) {
		if (over > 0) {
			over = 0;
		}
		$(this).parent().removeClass('active');
		setTimeout(function() { if (over == 0) menu.find('.submenu').hide(); }, 0);
	});
})(jQuery);

<div align="justify">The Lyric Theatre has been the center for arts and entertainment on the Treasure Coast since it opened its doors in 1926. &nbsp;This historic, fully renovated showplace is an intimate performance hall with 500 seats and whisper-perfect acoustics. &nbsp;The Lyric stage hosts a year-round schedule of world-class performances from national and internationally renowned artists. &nbsp;It is also a venue for local productions, and civic and charitable events.<br /><br />For concerts, comedy, musicals, drama, dance, family fun -- and much more! -- come to the Lyric!</div>
