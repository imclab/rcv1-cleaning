(function( $ ){
/**
 * This plugin came about due to a need of a custom carasoul 
 * design in a project for work that i could not achieve with 
 * any other jquery carasoul plugins i found (30+ etc...)
 *
 * After a day or so of testing/tweaking/modifying other 
 * carousel plugins, i just decided to write my own.
 *
 */

$.fn.article_scroller = function( settings ) {

	var $this = $(this), // set a scoped-global reference to the parent object

		init = function() {

			$this.version = '1.0.3';

			var defaults = {
				/**
				 * Initial set of defauts
				 *
				 * @since 1.0.2
				 */
				currentSlide:1,
				animationSpeed:'fast',
				showButtons:true,
				showCaption:true,
				useViewSpace:true,

				/**
				 * new subset of defaults
				 *
				 * Adding these in after additional testing with the 
				 * plugin outside of its default layout to to add a 
				 * little more flexibility to the css
				 *
				 * @since 1.0.3
				 */
				btnNextClass:'next',
				btnPrevClass:'prev',
				captionClass:'caption',
				viewClass:'view',
				containerClass:'slideshow',
				opacityValue:0.420,
				showCategory:true,
				categoryClass:'category'
			}

			/** 
			 * extend the parent element with our default settings 
			 * and apply any user defined settings
			 * 
			 * @since 1.0.2
			 */
			$.extend( $this, defaults, settings || {} );

			setup(); // setup scroller defaults and build new DOM layout

		},

		setup = function() {

			$this.scrolling = false;

			var width = 0,

				container = $('<div>').addClass( $this.containerClass ),
				/**
				 * @since 1.0.2
				 *
				 * The view space is now optional, just to reduce 
				 * the DOM element count when its just not needed..
				 */
				view = $('<div>').addClass( $this.viewClass ),

				btnNext = $('<div class="control">').addClass( $this.btnNextClass ).append($('<span>')),

				btnPrev = $('<div class="control">').addClass( $this.btnPrevClass ).append($('<span>')),

				caption = $('<div>').addClass( $this.captionClass ), 

				category = $('<div>').addClass( $this.categoryClass );

			/**
			 * Set the default layout of the HTMLList element
			 *
			 * This loops through all the child elements, alters element.style.float 
			 * and calculates width to resize the parent element width.
			 *
			 * @since 1.0.1
			 *
			 * Note: In the future i plan to set this up so it builds and works off 
			 * a json object and only displays three DomElements on screen at once
			 */
			$this.children().each(function(i, item){

				width += $(item).outerWidth();
				$(item).css({
					float:'left',
					opacity: $this.opacityValue

				});
			});
			$this.css('width',width);


			/**
			 * Start building out the modified DOM layout
			 *
			 * @since 1.0.1
			 *
			 * Optional use of viewSpace
			 * @since 1.0.2
			 */
			if ($this.useViewSpace) container.append(view);
			if ($this.showCaption) container.append(caption);
			if ($this.showCategory) container.append(category);
			if ($this.showButtons)
			{
				container.append(btnNext);
				container.append(btnPrev);
			}

			$this.container = container;
			$this.before(container);
			
			if ($this.useViewSpace) view.append($this);
			else container.append($this);



			/**
			 * Setup mouseevents.
			 *
			 * @since 1.0.1
			 *
			 * Note: touch events to come...
			 *
			 */
			btnNext.bind('click', function() { $this.btnClick( 'right' ) });

			btnPrev.bind('click', function() { $this.btnClick( 'left' ) });
			


			/**
			 * Extend core functions
			 *
			 */
			$this.extend({

				setStartPosition: function() { 

					$this.currentSlide++;

					$this.children('li:first-child').before( $this.children('li:last') );
					$this.css({
						left: ( Math.abs($this.children(':nth-child(' + $this.currentSlide + ')').position().left) * -1 )
					});
					$this.setSlideCaption();
				},

				setSlideCategory: function() {

				},

				setSlideCaption: function() {

					var current = $this.getCurrentSlide(),
						caption = $(current).attr('data-caption'),
						anchor = $('<a>').attr('href', $(current).attr('data-url')).text(caption),
						category = $(current).attr('data-category');

						$($this.container).find('.' + $this.captionClass ).html( anchor );

						if ( $this.showCategory ) {

							$(this.container).find( '.'+$this.categoryClass ).html( category );

							// reset category left position to center
							$(this.container).find( '.'+$this.categoryClass ).css({
								left: ( ( $($this.container).outerWidth() / 2 ) - ( $(this.container).find( '.'+$this.categoryClass ).outerWidth() / 2 ) ),
								display:'inline-block'
							})
						}
				},

				setOpaqueSlide: function() {

					$this.children('li:nth-child(' + $this.currentSlide + ')').css({
						opacity:1
					});
				},

				getCurrentSlide: function() {

					return $this.find('li:nth-child(' + $this.currentSlide + ')');
				},

				btnClick: function( direction ) {

					if ( $this.scrolling )
						return;

					$this.scrolling = true;

					$( $this.container ).find( 'li' ).css( 'opacity', $this.opacityValue ); 

					var current,next,prev;

					switch( direction ) {

						case 'left':case 'up':

							current = $this.getCurrentSlide();

							prev = $this.find('li:nth-child(' + ($this.currentSlide -1) + ')').prev();

							if ( ! $(prev).prev().size() ) {

								$(current).prev().before( $this.find('li:last') );

								$this.currentSlide++;

								$this.css('left', Math.abs( $this.find('li:nth-child(' + $this.currentSlide + ')').position().left ) * -1 )
							}


							$this.animate({
								left: Math.abs( $this.find('li:nth-child(' + $this.currentSlide + ')').prev().position().left ) * -1
							}, $this.animationSpeed, function(){
								
								$this.scrolling=false;
								$this.currentSlide--;
								$this.setSlideCaption();
								$this.setOpaqueSlide();
							});

							break;

						case 'right':case 'down':

							current = $this.getCurrentSlide();

							next = $this.find('li:nth-child(' + ($this.currentSlide +1) + ')').next();

							if ( ! $(next).next().size() ) {
								$(current).next().after( $this.find('li:first') );

								$this.currentSlide--;

								$this.css('left', Math.abs( $this.find('li:nth-child(' + $this.currentSlide + ')').position().left ) * -1 )
							}

							$this.animate({
								//left: Math.abs( $this.find('li:nth-child(' + currentSlide + ')').next().position().left ) * -1
								left: Math.abs( $this.find('li:nth-child(' + $this.currentSlide + ')').next().position().left ) * -1
							}, $this.animationSpeed, function(){
								
								$this.scrolling=false;
								$this.currentSlide++;
								$this.setSlideCaption();
								$this.setOpaqueSlide();
							});

							break;
					}

				}
			});

			// once everything is setup, show all slides ( set to display none in css to prevent any content shifting on low memeory systems )
			$this.children('li').show();
			
			// set the starting postion
			$this.setStartPosition();

			// make the active slide opaque
			$this.children('li:nth-child(' + $this.currentSlide + ')').css({
				opacity:1
			})

		}

	init()  // kick start
}

})( jQuery );
