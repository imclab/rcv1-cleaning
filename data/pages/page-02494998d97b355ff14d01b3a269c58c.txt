	function mdl_StoreSearchSubmit(form) {
		var zipValu = $.trim($('#mdl_zipCode').val());
		//CQPRO-224 START
		//var regex = new RegExp("^\d*\.?\d*$");
		if( isNaN(zipValu))
		{
		  return alert('A 5-digit zip code required for radius search.');
		   }
		 //END
		if ( zipValu == '' ) {
			var selCity = $.trim($('#mdl_selectCity').val());
			var selProv = $.trim($('#mdl_selectState option:selected').val());
			if ( selCity == '' && selProv == '' ) {
				return false;
			}
			zipValu = selCity + "," + selProv;
		}
		
		var r = $('#mdl_radius').val();
		var geocd = new google.maps.Geocoder();
		// CQPRO-1388 START
		zipValu = zipValu +"," + 'US';
		//console.info(zipValu);
		// END
		geocd.geocode( {'address': zipValu}, function(results, status) {
			if (status == google.maps.GeocoderStatus.OK) {
				var lat = results[0].geometry.location.lat();
				var lng = results[0].geometry.location.lng();
				ajaxGetStoreSearchList(lat, lng, r);
			} else if (status == 'ZERO_RESULTS'){
				alert("There are 0 stores that match your search criteria.");
			} else {
				alert("An error occurred while fetching the result.");
			}
		});
		return false;
	}

	function ajaxGetStoreSearchList(lat, lng, r) {
		var url = "AjaxStoreSearchListDisplayView?storeId=" + "10151"

		$.get(url, function(data) {
			$('#mdl-stores-list').html(data);
		});
	}
