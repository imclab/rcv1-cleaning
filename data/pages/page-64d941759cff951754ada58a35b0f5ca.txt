	$.fn.setGridItemHeight = function () {
		var maxHeight = 0;
		var itemContainer = $(this);

		// get max height;
		var heights = itemContainer.children().map(function () {
			return $(this).height();

	function setDataWidthElements() {

				// set the element width to the natural parent width

				var unitPrice = window.TrackingPixelData.MinSalePrice ? window.TrackingPixelData.MinSalePrice : '';
				var listPrice = window.TrackingPixelData.MinRegularPrice ? window.TrackingPixelData.MinRegularPrice : ''; ;

				$('<span class="runa_command" type="product_detail"' +
					' product_id="' + window.PRODUCT_ID +
					'" name="' + window.PRODUCT_NAME +
					'" original_price="' + originalPrice +
					'" unit_price="' + unitPrice +
					'" list_price="' + listPrice +
					'" product_attributes=""' +
					' categories="' + categories + '"/>').appendTo('body');

			} else if (canonicalPageType == 'cart' || canonicalPageType == 'checkout') {
				if (typeof (window.CART) !== 'undefined' && window.CART && window.CART.Payload && window.CART.Payload.CartItems) {
					var cart = window.CART.Payload;
					for (var ch = 0; ch < window.CART.Payload.CartItems.length; ch++) {

						$('<span class="runa_command"' +
							' type="line_item"' +
							' line_item_type="product"' +

							' name="' + cart.CartItems[ch].ProductName + '"' +
							' unit_price="' + cart.CartItems[ch].UnitSalePrice.Amount + '"' +

							' categories=" "/>').appendTo('body');
					}

					if (cart.DiscountedTotalAmount) {
						var discountAmount = Math.round(cart.DiscountedTotalAmount.Amount * 100) / 100;

						if (discountAmount > 0) {
							$('<span class="runa_command" type="line_item"' +
								' line_item_type="charge" ' +
								'charge_type="coupon" ' +
								'amount="-' + discountAmount + '" />').appendTo('body');
						}
					}

					$('<span class="runa_command" type="coupon_code" value="' + cart.KeyCodeValue + '"/>').appendTo('body');

					if (cart.GrandtotalAmount) {
						$('<span class="runa_command" type="line_item" ' +
							'line_item_type="total"' +
							' total_price="' + Math.round(cart.GrandtotalAmount.Amount * 100) / 100 + '"/>').appendTo('body');
					}

					if (cart.TaxAmount) {
						$('<span class="runa_command" type="line_item" ' +
							'line_item_type="tax"' +
							' cost="' + Math.round(cart.TaxAmount.Amount * 100) / 100 + '"/>').appendTo('body');
					}

					if (cart.CartItems && cart.CartItems[0] && cart.CartItems[0].ShippingAmount) {
						$('<span class="runa_command"' +
							' type="line_item"' +
							' line_item_type="shipping"' +
							' method="' + cart.CartItems[0].ShipMethodName + '"' +
							' method_code="' + cart.CartItems[0].ShipMethodId + '"' +
							' method_code_cost="' + Math.round(cart.CartItems[0].ShippingAmount.Amount * 100) / 100 + '"' +
							' state="' + cart.ShippingAddress.StateCode + '"' +
							' zip_code="' + cart.ShippingAddress.ZipCode + '"' +
							' country="' + cart.ShippingAddress.CountryISO + '"/>').appendTo('body');
					}
				}

			} else if (canonicalPageType == 'purchase_confirmation') {

				if (typeof (window.ORDER) !== 'undefined' && window.ORDER && window.ORDER.Payload && window.ORDER.Payload.OrderItems) {
					var order = window.ORDER.Payload;

					switch (order.LastMoneyTransationMethodId) {
					case 'CCARD':
						paymentMethod = paymentMethods[0];
						break;
					case 'PP':
						paymentMethod = paymentMethods[2];
						break;
					case 'BML':
						paymentMethod = paymentMethods[1];
						break;
					case 'SCRDT':
						paymentMethod = paymentMethods[3];
						break;
					}

					for (var oi = 0; oi < order.OrderItems.length; oi++) {

						$('<span class="runa_command"' +
							' type="line_item"' +
							' line_item_type="product"' +
							' product_id="' + order.OrderItems[oi].ProductId + '"' +
							' name="' + order.OrderItems[oi].ProductName + '"' +
							' unit_price="' + order.OrderItems[oi].UnitSalePrice + '"' +
							' quantity="' + order.OrderItems[oi].Quantity + '"' +
							' list_price="' + order.OrderItems[oi].UnitSalePrice + '"' +
							' product_attributes=""' +
							' categories="" />').appendTo('body');
					}

					$('<span class="runa_command" type="line_item" ' +
						'line_item_type="total"' +
						' total_price="' + Math.round(order.GrandTotalAmount * 100) / 100 + '"/>').appendTo('body');


					$('<span class="runa_command" type="line_item" ' +
						'line_item_type="order_id"' +
						' order_id="' + window.ORDER_ID + '"/>').appendTo('body');

					$('<span class="runa_command" type="line_item" ' +
						'line_item_type="tax"' +
						' cost="' + Math.round(order.TaxAmount * 100) / 100 + '"/>').appendTo('body');

					$('<span class="runa_command" type="line_item" ' +
						'line_item_type="payment_type"' +
						' method="' + paymentMethod + '" ' +
						'method_code="' + order.LastMoneyTransationMethodId + '"/>').appendTo('body');


					$('<span class="runa_command"' +
						' type="line_item"' +
						' line_item_type="shipping"' +
						' method="' + order.OrderItems[0].ShipMethodName + '"' +
						' method_code="' + order.OrderItems[0].ShipmethodId + '"' +
						' method_code_cost="' + (Math.round(order.OrderItems[0].ShippingAmount * 100) / 100) + '"' +
						' state="' + order.ShippingAddress.StateCode + '"' +
						' zip_code="' + order.ShippingAddress.ZipCode + '"' +
						' country="' + order.ShippingAddress.CountryISO + '"/>').appendTo('body');

					if (order.DiscountedTotalAmount) {
						var orderDiscountAmount = Math.round(order.DiscountedTotalAmount.Amount * 100) / 100;

						if (orderDiscountAmount > 0) {
							$('<span class="runa_command" type="line_item"' +
								' line_item_type="charge" ' +
								'charge_type="coupon" ' +
								'amount="-' + orderDiscountAmount + '" />').appendTo('body');
						}
					}

					$('<span class="runa_command" type="coupon_code" value="' + order.KeyCodeValue + '"/>').appendTo('body');
				}
			}
		}
