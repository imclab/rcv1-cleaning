	function searchfilter_parsevar( url_to_parse, var_to_parse_out ) {

		var start_point = url_to_parse.indexOf(var_to_parse_out);
		if (start_point == -1) {
			return "";
		}
		start_point += var_to_parse_out.length + 1;
		var end_point = url_to_parse.indexOf("&",start_point);
		if (end_point == -1) {
				end_point = url_to_parse.length;
		}
		var parsed_var = url_to_parse.substring(start_point, end_point);
		return parsed_var;
	}

	function searchfilter_replacevar( url_to_parse, var_to_replace, replace_value ) {

		var start_point = url_to_parse.indexOf(var_to_replace);
		if (start_point == -1) {
			return "";
		}
		start_point += var_to_replace.length + 1;
		var end_point = url_to_parse.indexOf("&",start_point);
		if (end_point == -1) {
				end_point = url_to_parse.length;
		}
		var replaced_url = url_to_parse.substring(0, start_point) + replace_value + url_to_parse.substring(end_point, url_to_parse.length);

		return replaced_url;
	}

	function formholder_merge_other_params ( form1, form2 ) {
		var field_separator = "%3B";
		var params_separator = "%7C";

		var primary_op = document.getElementById(form1+"_other_params");
		var secondary_op = document.getElementById(form2+"_other_params");
		var other_params1 = primary_op.value;
		var other_params2 = secondary_op.value;

		//we scan other_params1 for the various values
		//we then scan other_params2, if there is a new
		//value, we concatenate it to other_params1
		//if there's an existing value that's different
		//other_params1 value takes priority
		//we then set everything in final_other_params

		var associative_params = new Array();
		var indiv_params1 = other_params1.split(params_separator);

		for (var i=0; i<indiv_params1.length; ++i) {
			var param = indiv_params1[i];
			if (param == "") {
				continue;
			}
			var param_components = param.split(field_separator);
			//alert ("param is '"+param+"'. setting associative_params["+param_components[1]+"] = " +param_components[0]);
			associative_params[param_components[1]] = param_components[0];
		}
		var indiv_params2 = other_params2.split(params_separator);
		for (var i=0; i<indiv_params2.length; ++i) {
			var param = indiv_params2[i];
			if (param == "") {
				continue;
			}
			var param_components = param.split(field_separator);
			if (associative_params[param_components[1]]) {
				//potential conflict
				//since params1 has priority
				//we do nothing for now
			} else {
				//alert ("param is '"+param+"'. adding associative_params["+param_components[1]+"] = " +param_components[0]);
				associative_params[param_components[1]] = param_components[0];
			}
		}
		//now we reassemble a merged otherparams via associative_params
		var assembled_params = new Array();
		var count = 0;
		for (var key in associative_params) {
			assembled_params[count++] = associative_params[key]+field_separator+key;
		}
		var merged_other_params = assembled_params.join(params_separator);
		//where should we put our other_params?
		//back where we found them I guess!
		secondary_op.value = merged_other_params;
		primary_op.value = merged_other_params;
		
	}

	function formholder_merge_breadcrumb ( form1, form2 ) {
		var breadcrumb_separator = "%7C";

		var primary_op = document.getElementById(form1+"_breadcrumb");

		var bread2 = secondary_op.value;

		primary_op.value = merged_breadcrumb;
		
	}

function getElementID(element) {
    if (element !=null) {
        if (element.attributes != null) {
            var element_attributes = element.attributes;
            for (i=0; i<element_attributes.length; i++){
                if (element_attributes[i].name == "id") {

                }
            }
        }
    }
    return null;

function elementIsUnderID(element, desired_id) {
    var current_id = getElementID(element);
    if (current_id == desired_id) {
        return true;
    }
    if (element.parentNode == null) {
        return false;
    } else {
        return elementIsUnderID(element.parentNode, desired_id);
    }
    return false;
