<script type="text/javascript">
    var eo_user_group_percent = 10;
(function($) {
    var defaults = {
        debug: false, // 'watch' | true | false
        enabled: true, // start tracking with initialization

        selector: '.post, .post-shared-headline', // which elements to track
        post_id: '',
        scroll_interval: 50, // how often to check for scroll
        noscroll_trigger: 2000, // how much time is considered "stopped scrolling" (i.e. start measuring time)

        result_mode: 'interval', // 'immediate|interval'
        result_interval: 500, // how often post results

        stat_url: 'http://' + location.hostname + '/services/engage-o-meter/ajax.php',
        stat_random_hash_length: 5,
        stat_ph_paged: ph_paged,
        stat_user_group_percent: 100
    },
    initialized = false,
            options,
            positions = [],
            is_scrolling = false,
            scroll_timeout,
            results = {},
            posted = [];

    $.fn.engageOMeterFindElement = function(scroll_position) {
        scroll_position += $(window).height() / 3;
        var min = {
            i: -1,
            diff: $(document).height()
        };
        for (var i = 0; item = positions[i]; i++) {
            var diff = Math.abs(item.top - scroll_position);
            if (diff < min.diff) {
                min = {
                    i: i,
                    diff: diff
                };
            }
        }

        var selected = positions[min.i];

        if (options.debug) {
            options.$targets.removeClass('selected');
            $(selected.elem).addClass('selected');
        }

        return selected;
    }

    var alphabet = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';

    $.fn.engageOMeterRandomUrlPart = function() {
        var hash = '';
        for (var i = 0; i < options.stat_random_hash_length; i++) {
            hash += alphabet.charAt(Math.floor(Math.random() * alphabet.length));
        }
        return '&rand=' + hash;
    }

    $.fn.engageOMeterPostResults = function() {
        var results = [];
        var stat_src = '';

        for (var i = 0; item = positions[i]; i++) {
            if (item.time > 0 && $.inArray(i, posted) === -1) {
                posted.push(i);
                results.push({
                    post_id: item.post_id,
                    time: item.time
                });
            }
            item.time = 0;
        }

        for (var i = 0; item = results[i]; i++) {
            //$.fn.engageOMeterWatchStat('Hit: ' + item.post_id + ': ' + item.time / 1000);
            $.fn.engageOMeterWatchStat('Hit: ' + item.post_id);
        }

        for (var i = 0; item = results[i]; i++) {
            item.post_id = item.post_id.replace("post-", "");
            stat_src = options.stat_url + "?action=count_post&post_id=" + item.post_id + "&page_number=" + options.stat_ph_paged + $.fn.engageOMeterRandomUrlPart() + "&site_of_the_post=" + $.fn.engageOMeterGetSite(item.post_id) + "&site_where_showing_up=" + ph_site;
            $("body").prepend('<div id="engage-o-meter-' + item.post_id + '" class="top_absolute_1x1"><img src="' + stat_src + '" style="width: 1px; height: 1px; border: none;"/></div>');
        }
    }

    $.fn.engageOMeterInitialize = function() {
        var last_scroll = $(window).scrollTop(),
                was_scrolling = false;

        window.setInterval(function() {
            if (!options.enabled) {
                return;
            }

            var current_scroll = $(window).scrollTop();
            is_scrolling = current_scroll != last_scroll;
            last_scroll = current_scroll;
            if (was_scrolling && !is_scrolling) {
                $(document).trigger('scrollStop', [current_scroll]);
            }
            if (is_scrolling && !was_scrolling) {
                $(document).trigger('scrollStart');
            }
            was_scrolling = is_scrolling;
        }, options.scroll_interval);

        var last_selected,
                reading_started;

        $(document).bind('scrollStop', function(e, scroll_position) {
            window.clearTimeout(scroll_timeout);
            scroll_timeout = window.setTimeout(function() {
                last_selected = $.fn.engageOMeterFindElement(scroll_position);
                reading_started = new Date();
            }, options.noscroll_trigger);
        });

        $(document).bind('scrollStart', function(e) {
            window.clearTimeout(scroll_timeout);
            if (options.debug) {
                options.$targets.removeClass('selected');
            }
            if (last_selected) {
                last_selected.time += (new Date() - reading_started);
            }
            last_selected = null;
        });

        $(document).bind('enableScrollTracking', function() {
            options.enabled = true;
            if (!initialized) {
                $.fn.engageOMeterInitialize();
            }
        });
        $(document).bind('disableScrollTracking', function() {
            options.enabled = false;
            last_selected = null;
            $(document).trigger('scrollStart');
        });

        if (options.result_mode === 'interval') {
            window.setInterval($.fn.engageOMeterPostResults, options.result_interval);
        } else {
            $(document).bind('scrollStart', $.fn.engageOMeterPostResults);
        }

        $(window).bind('beforeunload', $.fn.engageOMeterPostResults);

        initialized = true;
    }

    $.fn.engageOMeterPotentialHitsStatistics = function() {
        var stat_src = '?action=count_potential_hits';
        var watch_row = 'Potential hits: ';

        for (var i = 0; item = positions[i]; i++) {
            item.post_id = item.post_id.replace("post-", "");
            stat_src += '&post_ids[]=' + item.post_id;
            stat_src += '&sites_of_the_posts[]=' + $.fn.engageOMeterGetSite(item.post_id);
            watch_row += item.post_id + ', ';
        }

        watch_row = watch_row.substring(0, watch_row.length - 2);
        $.fn.engageOMeterWatchStat(watch_row);
        stat_src = options.stat_url + stat_src + "&page_number=" + options.stat_ph_paged + $.fn.engageOMeterRandomUrlPart() + "&site_where_showing_up=" + ph_site;

        $("body").prepend('<div id="engage-o-meter-potential-hits" class="top_absolute_1x1"><img src="' + stat_src + '" style="width: 1px; height: 1px; border: none;"/></div>');
    }

    $.fn.engageOMeterWatchStat = function(row) {
        if (options.debug !== 'watch')
            return;

        if (!$("#engage_watch").width()) {
            $("body").append('<div id="engage_watch"><b>Engage-o-meter stat</b><br/><br/></div>');
        }

        $("#engage_watch").html($("#engage_watch").html() + row + "<br/>");
    }

    $.fn.engageOMeterGetSite = function(id) {
        var url = '';
        var site = 'perezhilton';

        if ($("#permalink-" + id).attr("href"))
            url = $("#permalink-" + id).attr("href");

        if (url == '' && $("#post-" + id + " h2 a").attr("href"))
            url = $("#post-" + id + " h2 a").attr("href");

        if (url == '' && $("#post-" + id + " .perez-recommends").attr("href"))
            url = $("#post-" + id + " .perez-recommends").attr("href");

        if (url !== '') {
            if (url.search(/\/cocoperez/i) !== -1)
                site = 'cocoperez';
            else if (url.search(/\/perezitos/i) !== -1)
                site = 'perezitos';
            else if (url.search(/\/fitperez/i) !== -1)
                site = 'fitperez';
            else if (url.search(/\/teddyhilton/i) !== -1)
                site = 'teddyhilton';
        }

        return site;
    }

    $.fn.engageOMeterGetPositions = function() {
        var i;
        var offset;
        var top;

        for (i = 0; i < positions.length; i++) {
            if (typeof(positions[i]['post_id']) !== 'undefined' &&
                    $("#post-" + positions[i]['post_id']).offset()) {
                positions[i]['top'] = $("#post-" + positions[i]['post_id']).offset().top;
            }
        }

        setTimeout("$.fn.engageOMeterGetPositions()", 1000);
    }

    $.fn.engageOMeter = function(settings) {
        var sov = Math.floor(100 * Math.random()) + 1;
        options = $.extend({}, defaults, settings);
        options.$parent = this;
        options.$targets = this.find(options.selector);

        options.$targets.each(function() {
            var $this = $(this);

            positions.push({
                elem: this,
                post_id: $this.attr("id"),
                top: $this.offset().top,
                time: 0
            });
        });

        if (options.stat_ph_paged == 0)
            options.stat_ph_paged = 1;

        setTimeout("$.fn.engageOMeterGetPositions()", (ph_share_btn_delay + 1000));

        if (options.enabled && (sov <= options.user_group_percent)) {
            $.fn.engageOMeterInitialize();
            $.fn.engageOMeterPotentialHitsStatistics();
        }

        return this;
    }
})(jQuery);
