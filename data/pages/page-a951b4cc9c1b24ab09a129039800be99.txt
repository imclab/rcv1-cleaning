    function EventCalendarScrollTicker () {

      var currentIndex = EventCalendarFeedPosition++;
      var maxElements = 2;

      if (currentIndex >= EventCalendarTabItems[EventCalendarFeedSource].size()) {
        currentIndex = 0;
        EventCalendarFeedPosition = 1;
      }
      if (EventCalendarTabItems[EventCalendarFeedSource][currentIndex]) {
        newItem = new Element('li', {'class': "l0 eventItem animatedEventItem"}).hide().update('<div>' + '<div class="EventCalendarEventSearchResultsItemHead">' + EventCalendarTabItems[EventCalendarFeedSource][currentIndex].title + '</div>' + '<div class="EventCalendarEventSearchResultsItemBody">' + EventCalendarTabItems[EventCalendarFeedSource][currentIndex].body + '</div>' + '</div>');

        // Add new item.
        new Insertion.Top($('EventCalendarEventSearchResults'), newItem);

        // Poof!  New item!
        new Effect.SlideDown($$('#EventCalendarEventSearchResults>li').first());


        // Remove old item(s).
        var numElements = $$("#EventCalendarEventSearchResults>li").size();
        if (numElements > maxElements) {
          for (var i = maxElements; i < numElements; i++) {
            var deleteMe = $$("#EventCalendarEventSearchResults>li")[i];
            // Remove.
            new Effect.Fade(deleteMe, {afterFinish: function() {deleteMe.remove(); return false;}});
          }
        }
      }

      return false;
    }
