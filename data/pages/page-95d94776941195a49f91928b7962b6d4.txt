function boxCatPart(args, _k){
	_class = (_k<2)?' border-bottom-grey ':'';
	str = 		'<div class="white '+_class+' ">';
	str = str + '<div class="category-title clearfix"><span></span><a class="category-title-link" href="javascript:void(0);">'+args.label+'</a></div>';
	str = str + '</div>';
	return str;
}

function boxChild(v1ChildLink, args){
	_strV2 = '';
	$.each(v1ChildLink, function(k2,v2) {
		_label = (isArrMultiExist(v2, ['key']) && isNaN(v2.key) )?v2.key: (isArrMultiExist(v2, ['category']))?v2.category:'';
		_link = (isArrMultiExist(v2, ['frontEndLink', 'url']))?v2.frontEndLink.url:'';
		_strV2 = _strV2 + '<li><a href="'+_link+'">'+_label+'</a></li>';
	});
	_html  = '<div class="category-list" style="display: none;">';
	_html += '	<div>';
	_html += '		<ol>';
	_html += _strV2;
	_html += '		</ol>';
	if (args.href!=''){
	_html += '		<a class="left-nav-see-more" href="'+args.href+'"><strong>See All</strong></a>';
	}
	_html += '	</div>';
	_html += '</div>';
	return _html;
}

function constructCatPart(arrData, level){
	var allStr = '';
	if (level==1){
		if (isArrMultiExist(arrData, ['popular'])){
			_count = arrData.popular.link.length;
			$.each(arrData.popular.link, function(k1,v1) {
				var _class = ( (_count-1)==k1)?' class="last" ':'';
				_boxCatPart = boxCatPart({'label':v1.category}, k1);
				if (isArrMultiExist(v1, ['Child', 'link'])){
					_href = ( isArrMultiExist(v1, ['frontEndLink', 'url']) ) ?v1.frontEndLink.url:'';
					allStr += '<li '+_class+'>' + _boxCatPart + boxChild(v1.Child.link, {'href':_href}) + '</li>';
				}
			});
		}
	}
	return allStr 
}

function openUL(level){
	switch (level){
		case 1:
			return '<ul class="level1" style="padding-left:5px;">';
		break;
		case 2:
			return '<ul class="level2" style="padding-left:10px;">';
		break;
		case 3:
			return '<ul class="level3" style="padding-left:15px;">';
		break;
	}	
}
function closeUL(level){
	return '</ul>';
}

function constructCatPartLeftNav(arrData, level){
	var allStr = '';

	if (isArrMultiExist(arrData, ['goBack_LevelOne', 'frontEndLink', 'url']) ) {		
		allStr += '<a href="' + arrData.goBack_LevelOne.frontEndLink.url + '">View All Categories</a>';
	}

	if (isArrMultiExist(arrData, ['link'])){
		$.each(arrData.link, function(k,v) {
			if (isArrMultiExist(v, ['frontEndLink', 'url'])){
				_label = (isArrMultiExist(v, ['category']))?v.category:'';
				_label = ( _label=='' && isArrMultiExist(v, ['key']) )?v.key:_label;
				_href = v.frontEndLink.url;
				allStr += '<li>';

				if (  _href!='no-link' && isArrMultiExist(v, ['Child'])){					
						_label = '< ' + _label;
				}

				allStr += (_href!='no-link')?'<a href="'+_href+'">'+_label+'</a>':'<span style="font-weight:bold;color:black;cursor:default;">'+_label+'</span>';
				
				if (isArrMultiExist(v, ['Child', 'link'])){
					allStr += openUL((level+1));
					allStr += constructCatPartLeftNav(v.Child, (level+1));
					allStr += closeUL((level+1));
				}
				allStr += '</li>';
			}
			else if (isArrMultiExist(v, ['type'])){
				_label = (isArrMultiExist(v, ['value']))?v.value:'';
				allStr += '<li>'+ _label +'</li>';
			}
		});
	}
	return allStr;
}
