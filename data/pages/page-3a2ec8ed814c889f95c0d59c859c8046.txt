		var mapName = 'listings';

		//Initialize Map
		function initWidgetMap(){	
			maps['listings'] = new Map('listings','map_listings','0','0',10);
			initOverlayCollections();
			maps['listings'].create();
		
			//Make sure they are initialized with filter flags
			preparePlacemarks();
			
			initStreetView();
		
			//Set view to map
			toggleView('map');
		}
		
		//Adds div to show street view panoramas
		function initStreetView(){
			var mapdiv = $P('map_listings');
			var d = document.createElement('DIV');
			
			d.id = 'street_listings';
			d.width = mapdiv.width;
			d.height = mapdiv.height;
			d.style.width = '686px';
			d.style.height = '400px';	
			d.style.display = 'none';
			$P('viewstack_listings').appendChild(d);
		}
		
		function showStreetView(){
			var p = placemarks['listings'][0];
			var point = p.point;
		
			var mypano = new GStreetviewPanorama($P('street_listings'));
			GEvent.addListener(mypano, "error", handleError); 
			mypano.setLocationAndPOV(point);
		}
		
		function initOverlayCollections(){
			//Initialize Variables and Event Handlers
			overlays.map = maps['listings'];
			overlays.pagesize = 99;
			overlays.addHandler('showhide',overlays.markPlacemarkListings,true);
			overlays.addHandler('openinfobubble',overlays.updateBubble,true);

			//Add Main overlay collection and set as default
			maps['listings'].addCollection(overlays,true);

			overlays.getInfoHTML = function(placemark){
				if (currView == 'mini') 
					return null;
				else	
					return parsePlacemarkData('\r\n\t<div class=\"infoWindowContainer\" id=\"infoWindowContainer_placemark:prikey\">\r\n\t\t<div class=\"infoWindowTitle\">placemark:name</div>\r\n\t\t<div class=\"infoWindowImage\"></div>\r\n\t\t<div style=\"float: left\">\r\n\t\t\t<div class=\"infoWindowAddress\">placemark:address</div>\r\n\t\t\t<div class=\"infoWindowLinks\" style=\"clear: both;\">\r\n\t\t\t\t\r\n\t\t\t\t\r\n\t\t\t</div>\r\n\t\t</div>\r\n\t</div>\r\n\t',placemark);					
			}
		}

		//Go to Marker and Open InfoWindow
		function goToPlacemark(prikey){
			if(currView == 'mini')
				toggleView('map');
			if ($('.gmapBox').css('display') == "none")
				$('.handle').click();
			$P('map_listings').scrollTo();
			var p = findPlacemark('listings',prikey);
			p.marker.openInfoWindowHtml(p.marker.myInfoHTML);
			p.collection.execHandler('openinfobubble');
		}

		//Find a placemark in the placemarks collection
		function findPlacemark(mapname,placemarkID){
			for (var i = 0; i < placemarks[mapname].length; i++){
				if(placemarks[mapname][i].prikey == placemarkID)
					return placemarks[mapname][i];
			}
			return null;
		}		
		
		//This function sets up the filter fields on each placemark
		function preparePlacemarks(){
			var itinLookup = '';
		
			for (var i in placemarks){
				for (var j = 0; j < placemarks[i].length; j++){
					if(placemarks[i][j].latitude.length == 0 || placemarks[i][j].longitude.length == 0){
						placemarks[i].splice(j,1);
						j = j - 1;
					}
					else{
						placemarks[i][j].filteredBy = new Object();
						placemarks[i][j].filtered = false;
						placemarks[i][j].active = true;
						placemarks[i][j].point = new GLatLng(placemarks[i][j].latitude,placemarks[i][j].longitude);
						if(itinLookup.indexOf('|' + placemarks[i][j].prikey) >= 0)
							placemarks[i][j].itinerary = true;
						else
							placemarks[i][j].itinerary = false;
					}
				}
			}
		}

		//Runs whenever the filters are updated
		function updatedFilters(unfiltered){
			maps['listings'].removePlacemarks(maps['listings'].defaultCollection,true);
			maps['listings'].addPlacemarks(unfiltered,maps['listings'].defaultCollection);
			maps['listings'].showPlacemarks(maps['listings'].defaultCollection,1);
			maps['listings'].setBoundsCenterAndZoom();
		}

		function toggleView(view){
			var miniLoc = $P('map_listings_mini');
			var fullLoc = $P('map_listings_full');
			var containerdiv = $P('map_listings_container');		
			//alert(view);
			if(view == 'street'){
				maps['listings'].cancelCentering = false;
				$P('street_listings').style.display = '';	
				$P('street_listings').style.width = '686px';
				$P('street_listings').style.height = '400px';						
				
				if(miniLoc && fullLoc && containerdiv.parentNode == miniLoc){
					miniLoc.removeChild(containerdiv);
					fullLoc.appendChild(containerdiv);
				}
				
				maps['listings'].gmap.checkResize();
				maps['listings'].gmap.addControl(maps['listings'].control);
				maps['listings'].gmap.addControl(maps['listings'].typeControl);
				var largeLink;
				var smallLink;
				if((largeLink = $P('showLargeLink')) && (smallLink = $P('showSmallLink'))){
					largeLink.style.display = 'none';
					smallLink.style.display = '';
				}
				currView = 'street';
				
				
				
				$P('street_listings').style.display = '';
				$P('map_listings').style.display = 'none';
				currView = 'street';
				showStreetView();
				$P('maptabhead_1').className = 'tab-off';
				$P('maptabhead_2').className = 'tab-on';
			} else if (view == 'map'){
				maps['listings'].cancelCentering = false;
				$P('map_listings').style.display = '';	
				$P('map_listings').style.width = '686px';
				$P('map_listings').style.height = '400px';						

				if(miniLoc && fullLoc && containerdiv.parentNode == miniLoc){
					miniLoc.removeChild(containerdiv);
					fullLoc.appendChild(containerdiv);
				}
				
				maps['listings'].gmap.checkResize();
				maps['listings'].gmap.addControl(maps['listings'].control);
				maps['listings'].gmap.addControl(maps['listings'].typeControl);
				var largeLink;
				var smallLink;
				if((largeLink = $P('showLargeLink')) && (smallLink = $P('showSmallLink'))){
					largeLink.style.display = 'none';
					smallLink.style.display = '';
				}
				currView = 'map';
			}
			else if (view == 'mini'){
				maps['listings'].cancelCentering = false;
				$P('map_listings').style.display = '';	
				$P('map_listings').style.width = '100px';
				$P('map_listings').style.height = '100px';
				if(miniLoc && fullLoc && containerdiv.parentNode == fullLoc){
					fullLoc.removeChild(containerdiv);
					miniLoc.appendChild(containerdiv);
				}
				maps['listings'].gmap.checkResize();				
				maps['listings'].gmap.removeControl(maps['listings'].control);
				maps['listings'].gmap.removeControl(maps['listings'].typeControl);
				currView = 'mini';
				var largeLink;
				var smallLink;
				if((largeLink = $P('showLargeLink')) && (smallLink = $P('showSmallLink'))){
					largeLink.style.display = '';
					smallLink.style.display = 'none';
				}
			}
		
			updatedFilters(placemarks['listings']);
		}

			return !($P('map_listings_full').style.display == 'none');
