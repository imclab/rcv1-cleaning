  $(".submit").click(function() {  									//Handles Submitting the form

	var email = $("input#email").val();
	var topic = $("select#topic").val();

      $("textarea#message").focus();  
      return false;  
    }
	var dataString = 'name='+ name + '&email=' + email + '&topic=' + topic + '&message=' + message + '&subject=' + document.URL + '&res=' + window.screen.width + 'x' + window.screen.height;

        // Use Microsoft XDR for IE9 since ajax method below is not supported by IE9

        xdr.onload = function() {
            //XDomainRequest has no response method so the below is used
            var dom = new ActiveXObject("Microsoft.XMLDOM");

            dom.loadXML(xdr.responseText);
			if (xdr.responseText == "success")

				$('.holder').hide();
				$('#callback').show().append("Your feedback is appreciated, Thank You!");

				$('.holder').hide();
				$('#callback').show().append("There was an error processing your request, we're very sorry please try again later.");
			}
        };
        xdr.send(dataString);	//send the data string
    } else {	
	$.ajax({  
		type: "POST",  
		crossDomain: true,
		url: "http://proxy1.car.org/detect/mail.php",  
		data: dataString,  
		success: function() {
			$('.holder').hide();
			$('#callback').show().append("Your feedback is appreciated, thank you!");
		},
		error:function(){
			$('.holder').hide();
			$('#callback').show().append("There was an error processing your request, we're very sorry please try again later.");
		}
});
} 
return false;

function validateEmail(email) { 
  // http://stackoverflow.com/a/46181/11236 
    var re = /^(([^<>()[\]\\.,;:\s@\"]+(\.[^<>()[\]\\.,;:\s@\"]+)*)|(\".+\"))@((\[[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\.[0-9]{1,3}\])|(([a-zA-Z\-0-9]+\.)+[a-zA-Z]{2,}))$/;
    return re.test(email);
}
