var searchBox = function() {

    var sPlaceholderText = '';
    var oToolTipDiv = null;

    return {
    
        focus : function() {
        
            if (sPlaceholderText == '') {
                sPlaceholderText = document.getElementById('HeaderSearchText').value;
            }

            if (document.getElementById('HeaderSearchText').value == sPlaceholderText) {
                document.getElementById('HeaderSearchText').value = '';
            }
        
        } , 
        
        blur : function() {
        
            if (document.getElementById('HeaderSearchText').value == '') {
                document.getElementById('HeaderSearchText').value = sPlaceholderText;
            }
        
        } , 

        tooltip : function() {
	        
	        var id = 'HeaderSearchTooltip';
	        var speed = 10;
	        var timer = 20;
	        var endalpha = 95;
	        var alpha = 0;
	        var ie = document.all ? true : false;
	        var sTooltipHtml = ' \
	            <p class="top">Enter any search criteria or keywords to search the site.</p> \
	            <p class="mid">To Locate a Dealer</> \
	            <p class="bot">Enter your 5 digit zip code to locate a dealer near you.</p> \
	            ';

	        var fade = function(d){

		        var a = alpha;
		        if((a != endalpha && d == 1) || (a != 0 && d == -1)){
			        var i = speed;
			        if(endalpha - a < speed && d == 1){
				        i = endalpha - a;
			        }else if(alpha < speed && d == -1){
				        i = a;
			        }
			        alpha = a + (i * d);
			        oToolTipDiv.style.opacity = alpha * .01;
			        if (!ie) {
			            oToolTipDiv.style.filter = 'alpha(opacity=' + alpha + ')';
			        }
		        }else{
			        clearInterval(timer);
			        if(d == -1){oToolTipDiv.style.display = 'none'}
		        }
	        }

        	
	        return {
		        show:function(){
			        if(oToolTipDiv == null){
				        oToolTipDiv = document.createElement('div');
				        oToolTipDiv.setAttribute('id',id);
				        document.body.appendChild(oToolTipDiv);
				        oToolTipDiv.innerHTML = sTooltipHtml;
			        }
			        oToolTipDiv.style.opacity = 0;
			        if (!ie) {
				        oToolTipDiv.style.filter = 'alpha(opacity=0)';
				    }
				    var rect = document.getElementById('HeaderSearchField').getBoundingClientRect();
                    oToolTipDiv.style.top = rect.bottom + 10 + 'px';
			        oToolTipDiv.style.left = rect.left - 100 + 'px';
			        oToolTipDiv.style.display = 'block';
			        clearInterval(timer);
			        timer = setInterval(function(){fade(1)},timer);
		        },
		        hide:function(){
			        clearInterval(timer);
			        timer = setInterval(function(){fade(-1)},timer);
		        }
	        };
        }
            
    }

}();

    function GlobalNavigationSearchSite(){
      if (document.getElementById("HeaderSearchText").value.replace(/ /g,'').length > 0) {
        // If 5 digits, do a dealer locator search

            top.location = "http://automobiles.honda.com/search-results.aspx?question="+escape(document.getElementById("HeaderSearchText").value);
        }
        
      }
      else {
      alert("You must enter a question before submitting");
      }

				// if cookie exists

					offset += search.length
					// set index of beginning of value

					// set index of end of cookie value

			}

var xmlHttp = false;


function AddTrackingTagsToPage (url) {

    // Create http object
    try {
        xmlHttp=new XMLHttpRequest();
    } catch (e) {
        try {
            xmlHttp=new ActiveXObject("Msxml2.XMLHTTP");
        } catch (e) {
            xmlHttp=new ActiveXObject("Microsoft.XMLHTTP");
        }
    }

    if (xmlHttp) {
        xmlHttp.onreadystatechange = ProcessTrackingTags;
        xmlHttp.open ('GET', url, true);
        xmlHttp.send (null);
    }

}

function ProcessTrackingTags() {

    var sUserTrackingHTML = '';

    if (xmlHttp.readyState == 4) {

        if (xmlHttp.status == 200) {

            var sUserTrackingHTML = xmlHttp.responseText;

	        ParseTrackingHTML(sUserTrackingHTML);
                    
        } else {
                // The request could not be fulfilled.
        }
    }

// This function will separate out the HTML and JavaScript from the HTML block.
// The JavaScript code will be passed into an eval() statement.
// The HTML code will be added to a new DIV.
function ParseTrackingHTML(sValue) {

		var sTempValue = sValue;

		var sTempString = sTempValue.toLowerCase();
		
        // Pull JavaScript code out of HTML block
		while (sTempString.indexOf("<scr" + "ipt") > 0) {
		
			var iIndex1 = sTempString.indexOf("<scr" + "ip" + "t");
			var iIndex2 = iIndex1 + sTempString.substr(iIndex1).indexOf(">") + 1;
			var iIndex3 = sTempString.substr(iIndex1).indexOf("</s" + "cr" + "ipt>");
			
			var sJS = sTempValue.substr(iIndex2   , iIndex3 - (iIndex2 - iIndex1));

            // document.write won't work, so we need to replace it with a call to CreateDivAndWriteContent
            sJS = sJS.replace(/document.write/g , 'CreateDivAndWriteContent');
            // Get rid of comment tags
            sJS = sJS.replace("<!--" , "");
            sJS = sJS.replace("//-->" , "");

            // Execute JS code
			eval(sJS);

            // Remove this JS block from sTempValue to look for next instance
			sTempValue = sTempValue.substr(0 , iIndex1) + '' + sTempValue.substr(iIndex1 + iIndex3 + 9);

			var sTempString = sTempValue.toLowerCase();			
		
		}

    // What's remaining from sTempValue is just HTML. Write it to a <div>
    CreateDivAndWriteContent(sTempValue);
		
}
	
// This function will dynamically create a div, populate it with passed in content, 
//  and append it to the current document.
function CreateDivAndWriteContent(sText) {

    var oNewDiv = document.createElement('div');
    oNewDiv.innerHTML = sText;
    document.body.appendChild(oNewDiv);

}
