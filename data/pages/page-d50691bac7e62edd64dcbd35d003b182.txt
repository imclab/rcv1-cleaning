        function GetCategoryContents(divId)
        {
            function OnCompleteCallback(args){DataReturned(divId, args);}
            SCPPublic.Services.CatalogService.ReturnChildrenCategories(divId, OnCompleteCallback, null, null);
            ChangeLinkToCollapse(divId);
        }
        
        function DataReturned(id, args)
        {      
               var markup = "<div id=\"" + id + "_children\" style=\"padding-left:10px;vertical-align:middle;line-height:20px;white-space:nowrap;\" />"
               for(var i=0;i<args.length;i++)
                 markup += AddCategoryMarkup(args[i]);
               
               $get(id).innerHTML += markup;
        }
        
        function AddCategoryMarkup(Category)
        {
        
            var image = "<image style='float:left;width:20px;' src=\"/vimages/spacer.gif\" border='0'/>";
            var href="javascript:ShowCategoryIsEmpty()";
             if(Category.HasInventory)

             }else
             {
                if(Category.Expandable)
                {
                    image="<image style='float:left;width:20px;' src='" + expandImageSrc + "' border='0'/>";

                }

            return "<div style=\"clear:both;white-space:nowrap;\" id=\"" +Category.ID+ "\">" + anchor + "</div>";
        }

        }
        
        function ChangeLinkToCollapse(divId)
        {
           $get("link_"+divId).href = "javascript:CollapseCategory('" + divId + "')";
            $get("link_"+divId).innerHTML = $get("link_"+divId).innerHTML.replace(expandImageSrc, collapseImageSrc);
                
        }
        
        function ChangeLinkToExpand(divId)
        {
            $get("link_"+divId).href = "javascript:ExpandCategory('" + divId + "');";
            $get("link_"+divId).innerHTML = $get("link_"+divId).innerHTML.replace(collapseImageSrc, expandImageSrc);
        }
        
        function ExpandCategory(divId)
        {
            if($get(divId + "_children"))

        }
