			if (!assetRotator.isEmpty(item.override) && (showThisSlide === true)) {
				if (item.override instanceof Array) {
					assetRotator.overrideArr.concat(item.override);
				} else {
					assetRotator.overrideArr.push(item.override);
				}
			}

			/* Check if this slide is in ContextManager.promotions */

			if (item.promotion) {
				showThisSlide = false;
				if (!assetRotator.isEmpty(promoList)) {
					jQuery.each(promoList, function (e, pList) {
						if (pList.title === item.promotion) {
							showThisSlide = true;
							return false;
						} else {
							showThisSlide = false;
						}
					});
				}
			}

			/* Check if this slide is in ContextManager.siteGroups */

			if (item.siteGroup) {
				showThisSlide = false;
				if (!assetRotator.isEmpty(groupList)) {
					jQuery.each(groupList, function (g, gList) {
						if (gList === item.siteGroup) {
							showThisSlide = true;
							return false;
						} else {
							showThisSlide = false;
						}
					});
				}
			}

			if (showThisSlide === false) {
				assetRotator.config.promos[i] = null;
			} else {
				assetRotator.promoArr.push(item);
			}
		});

		/* Check for overrides */

		if (assetRotator.overrideArr.length > 0) {
			jQuery.each(assetRotator.config.promos, function (i, item1) {
				if (jQuery.inArray(assetRotator.overrideArr, item1) > -1) {
					item1 = null;
				}
			});
		}
		assetRotator.loadHtml();
	},
	checkDate : function (start_date, end_date) {
		"use strict";

		if (this.isEmpty(start_date)) {
			start_date = this.dateToday;
		} else {
			start_date = this.scrubDate(start_date);
		}

		if (this.isEmpty(end_date)) {
			end_date = this.dateToday;
		} else {
			end_date = this.scrubDate(end_date);
		}

		if ((this.dateToday < start_date) || (this.dateToday > end_date)) {
			return false;
		}
		return true;
	},
	scrubDate : function (d) {
		"use strict";
		var year = d.split("/"),
			date;

		if (year[2].length < 4) {
			year[2] = "20" + year[2];
		}

		date = new Date(Number(year[2]), Number(year[0]) - 1, Number(year[1]));
		return (date);

	},

	preLoadImages : function (imgs) {

	init : function (uniqueId) {
		"use strict";

			assetRotator.trimArray();

	trimArray : function () {
		"use strict";
		var assetRotator = this,
			promoList = ContextManager.getPromotions(),
			groupList = ContextManager.getSiteGroups(),
			inventoryCheck = ContextManager.isNewInventory;

		/* Get a copy of the config object */
		assetRotator.config = jQuery.extend(true, {}, imgRotatorConfig2);

		jQuery.each(assetRotator.config.promos, function (i, item) {

			var showThisSlide = assetRotator.checkDate(item.startDate, item.endDate),
				selection = item.newOrUsed;

			/* Check to see if this is a "new inventory" promo */
			if (!assetRotator.isEmpty(item.newOrUsed)) {

				if (inventoryCheck === "true" && selection === "new") {
					showThisSlide = true;
				} else if (inventoryCheck === "false" && selection === "used") {
					showThisSlide = true;
				} else {
					showThisSlide = false;
				}
			}

			/* Check if promo contains override, if so add to the override array */

			if (!assetRotator.isEmpty(item.override) && (showThisSlide === true)) {
				if (item.override instanceof Array) {
					assetRotator.overrideArr.concat(item.override);
				} else {
					assetRotator.overrideArr.push(item.override);
				}
			}

			/* Check if this slide is in ContextManager.promotions */

			if (item.promotion) {
				showThisSlide = false;
				if (!assetRotator.isEmpty(promoList)) {
					jQuery.each(promoList, function (e, pList) {
						if (pList.title === item.promotion) {
							showThisSlide = true;
							return false;
						} else {
							showThisSlide = false;
						}
					});
				}
			}

			/* Check if this slide is in ContextManager.siteGroups */

			if (item.siteGroup) {
				showThisSlide = false;
				if (!assetRotator.isEmpty(groupList)) {
					jQuery.each(groupList, function (g, gList) {
						if (gList === item.siteGroup) {
							showThisSlide = true;
							return false;
						} else {
							showThisSlide = false;
						}
					});
				}
			}

			if (showThisSlide === false) {
				assetRotator.config.promos[i] = null;
			} else {
				assetRotator.promoArr.push(item);
			}
		});

		/* Check for overrides */

		if (assetRotator.overrideArr.length > 0) {
			jQuery.each(assetRotator.config.promos, function (i, item1) {
				if (jQuery.inArray(assetRotator.overrideArr, item1) > -1) {
					item1 = null;
				}
			});
		}
		assetRotator.loadHtml();
	},
	checkDate : function (start_date, end_date) {
		"use strict";

		if (this.isEmpty(start_date)) {
			start_date = this.dateToday;
		} else {
			start_date = this.scrubDate(start_date);
		}

		if (this.isEmpty(end_date)) {
			end_date = this.dateToday;
		} else {
			end_date = this.scrubDate(end_date);
		}

		if ((this.dateToday < start_date) || (this.dateToday > end_date)) {
			return false;
		}
		return true;
	},
	scrubDate : function (d) {
		"use strict";
		var year = d.split("/"),
			date;

		if (year[2].length < 4) {
			year[2] = "20" + year[2];
		}

		date = new Date(Number(year[2]), Number(year[0]) - 1, Number(year[1]));
		return (date);

	},
