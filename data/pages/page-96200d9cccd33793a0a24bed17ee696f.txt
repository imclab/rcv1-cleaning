<script language="javascript" type="text/javascript">
function changeAction() {

	var selectedScope = "";
	
	for (i=0;i<document.searchForm.scope.length;i++) {
		if (document.searchForm.scope[i].checked) {
	//		alert(document.searchForm.scope[i].value);
			selectedScope = document.searchForm.scope[i].value;
		}
	}
	
	if (selectedScope == "site") {
		document.searchForm.method = "GET";
		document.searchForm.action = "http://www.csi.edu/search.asp";
		//document.searchForm.action = "http://www.google.com/u/southernidaho"; -- Old Search Location
	}
	else if (selectedScope == "directory") {
		document.searchForm.name.value = document.searchForm.q.value;
		document.searchForm.method = "POST";
		document.searchForm.action = "http://www.csi.edu/facultyAndStaff_/directories/PersDirOption.asp";
	}
	else if (selectedScope == "web") {
		document.searchForm.domains.value = "";
		document.searchForm.sitesearch.value = "";
		document.searchForm.method = "GET";
		document.searchForm.action = "http://www.google.com/search";
	}

	return true;
	
//	myalert = alert("i = " + i + "\n selectedScope = " + selectedScope + "\n formLength = " + document.searchForm.scope.length + "\n Method = " + document.searchForm.method + "\n Action = " + document.searchForm.action);
}
