			function updateDate(createdDate){
				if(typeof(createdDate) === "string")
					createdDate = new Date(createdDate);
				var currDate = new Date(), updateTime = "",
				currDateNum = currDate.getDate(),
				createdDateNum = createdDate.getDate(),
				currHoursNum = currDate.getHours(),
				createdHoursNum = createdDate.getHours(),
				currMinuteNum = currDate.getMinutes(),
				createdMinuteNum = createdDate.getMinutes();

				if(currDateNum > createdDateNum){
					updateTime = (currDateNum - createdDateNum) + "d";
				} else {
					if(currHoursNum > createdHoursNum){
						if(currMinuteNum > createdMinuteNum){
							updateTime = (currHoursNum - createdHoursNum) + " h " + (currMinuteNum - createdMinuteNum) + " m";
						} else {
							updateTime = (currHoursNum - createdHoursNum - 1) + " h " + ((currMinuteNum + 60) - createdMinuteNum) + " m";
						}
					} else if(currMinuteNum > createdMinuteNum){
						updateTime += (currMinuteNum - createdMinuteNum) + " m";
					} else if(currMinuteNum == createdMinuteNum){
						updateTime = "Just Now";
					}
				}
				return updateTime;
			}
		})
