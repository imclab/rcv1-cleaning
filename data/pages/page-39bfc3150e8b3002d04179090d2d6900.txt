							{
								$items.removeClass('active');
								$searchInput.val(searchLast);
							}
						}
						if (e.keyCode == 40) // down
						{
							e.preventDefault();
							if ($items.filter('.active').length == 0)
							{
								$items.eq(0).addClass('active');
								$searchInput.val($('a', $items.eq(0)).text());
							}
							else if (idx < $items.length-1)
							{
								$items.removeClass('active');
								$items.eq(idx+1).addClass('active');
								$searchInput.val($('a', $items.eq(idx+1)).text());
							}
						}
						if (e.keyCode == 13) // enter
						{
							e.preventDefault();
							if ($('.active', self).length)
							{
								$searchInput.val($('.active', self).find('a').text());
							}
							$searchInput.closest('form').submit();
						}
					}
					else
					{
						searchLast = null;
					}
					if (e.keyCode == 27) // esc
					{
						$searchInput.blur();
					}
				});

				$searchInput.blur(function(){
					searchLast = null;
				});

				// focus search box in navigation bar on forward slash
				$('body').keydown( function ( e ) {
					var ignoreTags = [ 'input', 'textarea' ];
					var currentTag = document.activeElement.tagName.toLowerCase();
					if (e.keyCode == 191 && $.inArray(currentTag, ignoreTags) == -1) // forward slash
					{
						e.preventDefault();
						$searchInput.focus();
					}
				});

				$searchLabel.mousedown(function(e){
					e.preventDefault();
					e.stopPropagation();
					$searchInput.focus();
				});
				$searchLocation.mousedown(function(e){
					e.preventDefault();
				});
				$searchResult.mousedown(function(e){
					e.preventDefault();
				});
				if ( ! isNav)
				{
					$searchLocation.show();
				}
			});
		};

		});
