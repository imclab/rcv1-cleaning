<script language="javascript" type="text/javascript">
    // Globals
    var csDefaultPartNumberText = "Enter Part Number",
        csDefaultBrandText = "Enter Brand",
        csDefaultModelText = "Enter Model #",
        csDefaultKeywordText = "Enter Keyword";

    // Globals for TypeAhead
    var csTAGLSObject = {
        searchType: "glsByFilter",
        count: 20,
        minCharLength: 2,
        orderChannelID: 37,
        catalogID: 15
    };
    var csTABrandObject = {
        searchType: "glsBrandsFinder",
        count: 20,
        minCharLength: 2,
        orderChannelID: 37,
        catalogID: 15
    };
    var csTAModelObject = {
        searchType: "glsModelFinders",
        count: 20,
        minCharLength: 2,
        orderChannelID: 37,
        catalogID: 15
    };
    
    // Typeahead styling objects
    var csTASNormal = {
            "width" : "225px",
            "margin" : "5px 0 0 -2px"
        },
        csTASShort = {
            "width" : "172px",
            "margin" : "5px 0 0 -2px"
        };
    
    
    function submitCSForm() {
        var searchtype = $("#csSearchTypeDropDown").val(),
            searchform = $("#CombinedSearchBarForm");
        switch (searchtype) {
            case "partnumber":
                if(!validateComboForm($("#csPartNumberTextBox"),"Please enter a Part Number", csDefaultPartNumberText, 2, 200)) {
                    return false;
                }
                $("#csGLSSearch").val($("#csPartNumberTextBox").val());
                $("#csGLSTabType").val("Parts");
                break;
            case "brandmodel":
                if(!validateComboForm($("#csBrandModelBrandTextBox"), "Please enter a Brand", csDefaultBrandText, 2, 200)) {
                    return false;
                }
                if (!validateComboForm($("#csBrandModelModelTextBox"),"Please enter a Model", csDefaultModelText, 2, 200)) {
                    return false;
                }
                $("#csBrandName").val($("#csBrandModelBrandTextBox").val());
                $("#csModelNumber").val($("#csBrandModelModelTextBox").val());
                break;
            case "brand":
                if(!validateComboForm($("#csBrandTextBox"), "Please enter a Brand", csDefaultBrandText, 2, 200)) {
                    return false;
                }
                $("#csBrandName").val($("#csBrandTextBox").val());
                break;
            case "model":
                if (!validateComboForm($("#csModelTextBox"),"Please enter a Model", csDefaultModelText, 2, 200)) {
                    return false;
                }
                $("#csGLSSearch").val($("#csModelTextBox").val());
                $("#csGLSTabType").val("1");
                break;
            case "keyword":
                if (!validateComboForm($("#csKeywordTextBox"),"Please enter a Keyword", csDefaultKeywordText, 2, 200)) {
                    return false;
                }
                $("#csGLSSearch").val($("#csKeywordTextBox").val());
                break;
            default:
                break;
        }
        removeExcessFields(searchtype);
    }
    
    function csSetSearchType() {
        var searchtype = $("#csSearchTypeDropDown").val(),
            searchform = $("#CombinedSearchBarForm");
        csClearFields();
        csResetText();
        csHideAll();
        switch (searchtype) {
            case "partnumber":
                searchform.attr('action', '/SearchResults.aspx');
                $("#csReferPageType").val("PMN");
                $("#csPanePartNumber").show();
                break;
            case "brandmodel":
                searchform.attr('action', '/BrandSearch.aspx');
                $("#csReferPageType").val("BRM");
                $("#csPaneBrandModel").show();
                break;
            case "brand":
                searchform.attr('action', '/BrandSearch.aspx');
                $("#csReferPageType").val("BRD");
                $("#csPaneBrand").show();
                break;
            case "model":
                searchform.attr('action', '/SearchResults.aspx');
                $("#csReferPageType").val("MOD");
                $("#csPaneModel").show();
                break;
            case "keyword":
                searchform.attr('action', '/SearchResults.aspx');
                $("#csReferPageType").val("KEY");
                $("#csPaneKeyword").show();
                break;
            default:
                searchform.attr('action', '/SearchResults.aspx');
                $("#csReferPageType").val("GLS");
                break;
        }
    }
    
    function removeExcessFields(searchtype) {
        switch (searchtype) {
            case "partnumber":
                $("#csBrandName").removeAttr("name");
                $("#csModelNumber").removeAttr("name");
                break;
            case "brandmodel":
                $("#csGLSSearch").removeAttr("name");
                $("#csGLSTabType").removeAttr("name");
                $("#csTabAllSwitch").removeAttr("name");
                $("#csTabPartsSwitch").removeAttr("name");
                $("#csTabModelsSwitch").removeAttr("name");
                break;
            case "brand":
                $("#csGLSSearch").removeAttr("name");
                $("#csGLSTabType").removeAttr("name");
                $("#csModelNumber").removeAttr("name");
                $("#csTabAllSwitch").removeAttr("name");
                $("#csTabPartsSwitch").removeAttr("name");
                $("#csTabModelsSwitch").removeAttr("name");
                break;
            case "model":
                $("#csBrandName").removeAttr("name");
                $("#csModelNumber").removeAttr("name");
                break;
            case "keyword":
                $("#csBrandName").removeAttr("name");
                $("#csModelNumber").removeAttr("name");
                break;
            default:
                break;
        }
    }

    function csClearFields() {
        $("#csGLSSearch").val("");
        $("#csGLSTabType").val("");
        $("#csBrandName").val("");
        $("#csModelNumber").val("");
    }
    
    function csResetText() {
        $("#csPartNumberTextBox").css('color','#676868').val(csDefaultPartNumberText);
        $("#csBrandModelBrandTextBox").css('color','#676868').val(csDefaultBrandText);
        $("#csBrandModelModelTextBox").css('color','#676868').val(csDefaultModelText);
        $("#csBrandTextBox").css('color','#676868').val(csDefaultBrandText);
        $("#csModelTextBox").css('color','#676868').val(csDefaultModelText);
        $("#csKeywordTextBox").css('color','#676868').val(csDefaultKeywordText);
    }

    function csHideAll() {
        $("#csPanePartNumber").hide();
        $("#csPaneBrandModel").hide();
        $("#csPaneBrand").hide();
        $("#csPaneModel").hide();
        $("#csPaneKeyword").hide();
    }

    function csBindDefaultText(objToBind, textToBind) {
        objToBind.focus(function() {
            if (objToBind.val() == textToBind) {
                objToBind.val("");
                objToBind.css('color','#000');
            }
        }).blur(function() {
            if ((objToBind.val() == '') || (objToBind.val() == textToBind)) {
                objToBind.val(textToBind);
                objToBind.css('color','#676868');
            }
        });
    }
