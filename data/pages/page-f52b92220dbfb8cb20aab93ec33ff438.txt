     var timerInterval;
     var highlightTime;
     var secondsRemaining;

     $(document).ready(function () {
         secondsRemaining = parseInt(jQuery("#secondsRemaining").val());
         if (secondsRemaining !== -1)
             startCartTimer();
         highlightTime = parseInt($('#warningSeconds').val());
         //highlightTime = 20;
     });



     function startCartTimer() {
         $('#countDownTimer').append('<span class="countdown_row countdown_amount"></span>');
         timerInterval = setInterval(function () {

             if (secondsRemaining < 0) {
                 secondsRemaining = 0;
             }

             var hangingSeconds = secondsRemaining % 60;
             var minuteSeconds = secondsRemaining - hangingSeconds;
             var minutesRemaining = minuteSeconds / 60;
             var formattedHangingSeconds;
             var formattedMinutesRemaining;

             if (hangingSeconds < 10) { formattedHangingSeconds = "0" + hangingSeconds; }
             else { formattedHangingSeconds = hangingSeconds; }

             if (minutesRemaining < 10) { formattedMinutesRemaining = "0" + minutesRemaining; }
             else { formattedMinutesRemaining = minutesRemaining; }

             $('.countdown_amount').html(formattedMinutesRemaining + ":" + formattedHangingSeconds);
             if (secondsRemaining <= highlightTime) { cartTimerShouldHiglight() }
             if (secondsRemaining === 0) { cartTimerDidEnd(); }
             secondsRemaining--;

         }, 1000);
     }

     function cartTimerShouldHiglight() {
         $('.countdown_amount').parents('.expirationTime').addClass('highlight');
     }

     function cartTimerDidEnd() {
         clearInterval(timerInterval);
     }

        var drawerTabs = $('#homeDrawers > ul a');

        var supportsOpacity = Modernizr.opacity;

        $(function () {
            var changeDrawerEvent = Modernizr.touch ? 'click' : 'mouseover';
            drawerWrapper.height('0');

            if (!supportsOpacity) {
                drawers.css('display', 'none');
            }

        function changeDrawer() {
            if (!$(this).parent().hasClass('tabOn')) {
                var visible = $('.visible');
                var divToShow = $('#' + $(this).attr('href').replace('#', ''));
                var newHeight = divToShow.outerHeight();
                $('.tabOn').removeClass('tabOn');
                $(this).parent().addClass('tabOn');
                if (visible.length) {
                    if (!supportsOpacity) {
                        visible.removeClass('visible').hide();
                    }
                    else {
                        visible.removeClass('visible').stop().animate({ opacity: 0 }, dur, ease);
                    }

                }

                if (!supportsOpacity) {
                    divToShow.addClass('visible').show();
                }
                else {
                    divToShow.addClass('visible').stop().animate({ opacity: 1 }, dur, ease);
                }
                drawerWrapper.stop().animate({ height: newHeight }, dur, ease);
            }
            return false;
        }

        function closeDrawers() {
            $('.tabOn').removeClass('tabOn');
            drawerWrapper.stop().animate({ height: 0 }, dur, ease);

            if (!supportsOpacity) {
                $('.visible').removeClass('visible').hide();

            }
            else {
                $('.visible').removeClass('visible').stop().animate({ opacity: 0 }, dur, ease);
            }
            return false;
        }

        function initFeatureRotator() {
            $('#featureContent').cycle({ before: updateFeature, fx: 'fade', /*pagerEvent: 'mouseenter.cycle',*/speed: dur, /*timeout: 0,*/easing: ease, pager: '#featureLinks',
                pagerAnchorBuilder: function (idx, slide) {
                    return '#featureLinks li:eq(' + idx + ') a';
                }
            });
        }

        function updateFeature() {
            var currentLinkIndex = $(this).index();
            var content = $('#featureContent > div:eq(' + currentLinkIndex + ')').find('.content').html();
            $('#featureContentBox').html(content);
        }
