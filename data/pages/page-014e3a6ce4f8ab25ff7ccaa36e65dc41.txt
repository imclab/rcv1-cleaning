function getCookie(){

  if(document.cookie){

    index = document.cookie.indexOf(cookie_name);
    putCookie("userLoc","",2005);

    if (index != -1){

      temp = document.cookie.split(';');
      crumb = new Array();
      i=0;

        crumb[i] = temp[name].split('=');
