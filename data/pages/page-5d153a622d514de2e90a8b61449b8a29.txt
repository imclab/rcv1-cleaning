		var results = regex.exec( window.location.href );

	function validZip(zip)
{
	zip=zip.toUpperCase();
	
	if (zip.match(/^[A-Z][0-9][A-Z][0-9][A-Z][0-9]$/)) 
	{
		return true;
	}
	
	return false;
}

	function menuManualSelection(oMenu, itemPosition) {
		oMenu.clearActiveItem();
		oMenu.getItem(itemPosition).cfg.setProperty("selected", true);
		if (oMenu.cfg.getProperty("autosubmenudisplay")) {
			oSubmenu = oMenu.getItem(itemPosition).cfg.getProperty("submenu");
			if (oSubmenu) {
				oSubmenu.show();
			}
		}
	}

    function getNavFileName(carid, over)
    {
        var fileName;

        switch (parseInt(carid))
        {
            case 1:
                fileName = "/suzuki/assets/cms/btnCNsx4Hatch.png";
                break;
            case 2:
                fileName = "/suzuki/assets/cms/btnCNsx4Sedan.png";
                break;
            case 3:
                fileName = "/suzuki/assets/cms/btnCNgrandVitara.png";
                break;
            case 4:
                fileName = "/suzuki/assets/cms/btnCNsx4Crossover.png";
                break;
            case 5:
                fileName = "/suzuki/assets/cms/btnCNkizashi.png";
                break;
            case 6:
                fileName = "/suzuki/assets/cms/btnCNgvUrban.png";
                break;
        }

        var pureFileName = fileName.split(".")[0];
        var fileType = fileName.split(".")[1];

        if (lang == "fr-CA")
            pureFileName = pureFileName + "_fr";
        if (over)
            pureFileName = pureFileName + "_over";

        fileName = pureFileName + "." + fileType;
        return fileName;
    }

        function closeCarPopup(e, carid)
        {
            var navCar = document.getElementById("navcar" + carid);
            var navpopup = document.getElementById("navtxt");
            var foundCarNav = false;

            if (!e) var e = window.event;
            var tg = (window.event) ? e.srcElement : e.target;
            var reltg = (e.relatedTarget) ? e.relatedTarget : e.toElement;

            while (tg.nodeName != "BODY")
            {
                if (tg == navCar)
                {
                    foundCarNav = true;
                    break;
                }
                else if (tg == navpopup)
                    break;

                tg = tg.parentNode;
            }

            while (reltg.nodeName != "BODY")
            {
                if (foundCarNav)
                {
                    if (reltg == navpopup)
                        break;
                }
                else if (reltg == navCar)
                    break;
                reltg = reltg.parentNode;
            }


            if ((tg == navCar && reltg != navpopup) || (tg == navpopup && reltg != navCar))
                actuallyClosePopup(carid);
        }

        function actuallyClosePopup(carid)
        {
            writetxt(0);
            var navImage;
            navImage = getNavFileName(carid, false);
            document.getElementById("navcar" + carid).style.zIndex = 0;
            document.getElementById("navcar" + carid).src = navImage;
        }
