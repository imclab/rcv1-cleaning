                        zTglc(flyInEl, 'hidden');
                        addEventListeners();
                    }
                }
            }
            function setNextPage() {
                var els = [],
                selector,
                isSelected = false;
                
                if (document.querySelectorAll) {
                    for (var i = 0, len = nextPageSelectors.length; i < len; i++) {
                        selector = nextPageSelectors[i];
                        if (document.querySelector(selector)) {
                            els = document.querySelectorAll(selector);
                            finalizeFlyIn(els);
                            if (nextPage.href && nextPage.title) {
                                break;
                            }
                        }
                    }
                }
                else {
                    if (gEI('articlebody') && getElementsByClassName('bnav', gEI('articlebody'))[0]) {
                        if (getElementsByClassName('next', getElementsByClassName('bnav', gEI('articlebody'))[0])[0]) {
                            if (getElementsByClassName('next', getElementsByClassName('bnav', gEI('articlebody'))[0])[0].getElementsByTagName('a')) {
                                els = getElementsByClassName('next', getElementsByClassName('bnav', gEI('articlebody'))[0])[0].getElementsByTagName('a');
                                finalizeFlyIn(els);
                                if (nextPage.href && nextPage.title) {
                                    isSelected = true;
                                }
                            }
                        }
                    }
                    if (!isSelected && gEI('resources') && getElementsByClassName('.lkbx', gEI('resources'))) {
                        els = getElementsByClassName('lkbx', gEI('resources'));
                        for (var i = els.length - 1; i >= 0; i--) {
                            if (els[i].getElementsByTagName('a')) {
                                finalizeFlyIn(els[i].getElementsByTagName('a'));
                                if (nextPage.href && nextPage.title) {
                                    isSelected = true;
                                }
                            }
                        }
                    }
                    if (!isSelected && gEI('rel') && gEI('rel').getElementsByTagName('a')) {
                        els = gEI('rel').getElementsByTagName('a');
                        finalizeFlyIn(els);
                        if (nextPage.href && nextPage.title) {
                            isSelected = true;
                        }
                    }
                    if (!isSelected && gEI('related') && gEI('related').getElementsByTagName('a')) {
                        els = gEI('related').getElementsByTagName('a');
                        finalizeFlyIn(els);
                        if (nextPage.href && nextPage.title) {
                            isSelected = true;
                        }
                    }
                }
            }
            function insertContent() {
                flyInLink.setAttribute('href', nextPage.href);
                flyInLinkTitle.innerHTML = nextPage.title;
            }
            function addEventListeners() {
                addEventSimple(window, 'scroll', handleFlyIn);
                addEventSimple(flyInCloseBtn, 'click', function (e) {
                    if (e.preventDefault) {
                        e.preventDefault();
                    }
                    else {
                        e.returnValue = false;
                    }
                    destroy();
                });
            }
            function destroy() {
                zTglc(flyInEl, 'hidden');
                removeEventSimple(window, 'scroll', handleFlyIn);
            }
            return {
                init: function () {
                    prepareFlyIn();
                }
            };
