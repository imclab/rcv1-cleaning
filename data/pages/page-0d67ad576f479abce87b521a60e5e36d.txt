 function LoadPrice()
{
    var FromId='ctl00_MainContent_CategoryInfo1_ctl00_NarrowSearchCtrl_hdnFromPrice';
    var ToId='ctl00_MainContent_CategoryInfo1_ctl00_NarrowSearchCtrl_hdnToPrice';
    var FromPrice=document.getElementById("txtPriceFrom").value;
    var ToPrice=document.getElementById("txtPriceTo").value;
    if (VerifyRange(FromPrice,ToPrice))
    {
        //if "from" price is greater than  "to" price swap the values
        if(eval(FromPrice)> eval(ToPrice) &&  eval(FromPrice)> 0 && eval(ToPrice) > 0 )
         {
                document.getElementById(FromId).value=document.getElementById("txtPriceTo").value;
                document.getElementById(ToId).value=document.getElementById("txtPriceFrom").value;
         }
         else
         {
                document.getElementById(FromId).value=document.getElementById("txtPriceFrom").value;
                document.getElementById(ToId).value=document.getElementById("txtPriceTo").value;
         }

    __doPostBack();
    }
}

//This will search based on entered keyword
function GetEnteredValue(HdnSelectedValobj, txtSearchobj, hdnSeeMoreobj)
{
    var SearchValue = document.getElementById(txtSearchobj).value;
    var len = SearchValue.length;
    if(len == 0)
    {
        alert('Please enter a keyword in search criteria.');
        return false;
    }
    SearchValue = FormatSearchValue(SearchValue);
    document.getElementById(txtSearchobj).value = SearchValue;
    var SelectedIds = "";
    if(document.forms[0].elements.length != null)
	{
        for(i=0; i < document.forms[0].elements.length; i++)
        {
           if(document.forms[0].elements[i].type == "checkbox")
           {
                if(document.forms[0].elements[i].checked == true)
                {
                    SelectedIds += document.forms[0].elements[i].id + "#";
                }
            }
        }
    }
    len = SelectedIds.length;
    if(len != 0)
    {
        document.getElementById(HdnSelectedValobj).value = SelectedIds;
        document.getElementById(hdnSeeMoreobj).value = "";
    }
    return true;
}

//This function puts validations on the price range
function VerifyRange(PriceFrom,PriceTo)
{

   if(PriceFrom!= "" || PriceTo !="" )
   {
       //if "from" price is not having null value
          if(PriceFrom!="")
        {
            if(!isNumeric(PriceFrom))
             {
                alert('Please enter valid price');
                document.getElementById("txtPriceFrom").focus();
			    return false;
             }
             if(PriceFrom < 0)
             {
                 alert('Please enter valid price');
                 document.getElementById("txtPriceFrom").focus();
			    return false;
             }

        }
       //if "to" price is not having null value
         if(PriceTo !="")
         {
             if(!isNumeric(PriceTo))
             {
                 alert('Please enter valid price');
                 document.getElementById("txtPriceTo").focus();
			    return false;
             }

             if(PriceTo < 0)
             {
                 alert('Please enter valid price');
                 document.getElementById("txtPriceTo").focus();
			    return false;
             }
         }
   }
   return true;
}

    function ValidateDOB(input) {

        var validformat = /^\d{2}\/\d{2}\/\d{4}$/ //Basic check for format validity
        var returnval = false
        if (!validformat.test(input.value))
            return false;
        else { //Detailed check for valid date ranges
            var monthfield = input.value.split("/")[0]
            var dayfield = input.value.split("/")[1]
            var yearfield = input.value.split("/")[2]
            var dayobj = new Date(yearfield, monthfield - 1, dayfield)
            if ((dayobj.getMonth() + 1 != monthfield) || (dayobj.getDate() != dayfield) || (dayobj.getFullYear() != yearfield))
                returnval = false;
            else
                returnval = true
        }
        if (returnval == false) input.select()
        return returnval
    }
