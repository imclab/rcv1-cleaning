function ed_genDartFile(adtype) {
    if ((adtype == "box" || adtype == "spotlight")) {
        if (ed_req_model) {
            return(ed_req_make + ".forums.dart");
        } else if (ed_req_make) {
            return("forums.make.dart");
        } else {
            return("forums.dart");
        }
    } else {
        return("forums.dart");
    }
}

function ed_genDartParams(adtype) {
    var params = "";
    if (adtype == "spotlight" && ed_req_model) {
        params += "mdl=" + ed_req_model + ";";
    } else {
        if (ed_req_make) {
            params += "make=" + ed_req_make + ";";
        }
        if (ed_req_model) {
            params += "mdl=" + ed_req_model + ";";
        }
    }
    for (var i = 0; i < ed_req_category_list.length; i++) {
        params += "theme=" + ed_req_category_list[i] + ";";
    }
    for (var i = 0; i < ed_req_type_list.length; i++) {
        params += "type=" + ed_req_type_list[i] + ";";
    }
    return(params);
}

        params += ":type_" + ed_req_type_list[i];
    }
    return(params);
}
