			def = $(this).attr('data-default');
		} else {
			def = $(this).val();
		}
		$(this).data('initial_value',def);

			if ($(this).val() == def) {
				this.value = '';
			}
		});
		$(this).blur(function() {
			if (this.value.length == 0)
				this.value = def;
		});
