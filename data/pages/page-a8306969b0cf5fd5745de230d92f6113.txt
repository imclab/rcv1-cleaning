        var length = 3;
        var count = 0;
        var videoList = Y.one('#video_list_container ul');
        var thumbWidth = 0;
        if (videoList) {
            var videoThumb = videoList.one('li');
            var thumbWidth = videoThumb ? parseInt(videoThumb.getComputedStyle('width'), 10) : 0;
            thumbWidth += 4;
            var videoListWidth = (length * thumbWidth)+'px';
            videoList.setStyle('width', videoListWidth);
        }

        var nextBtn = Y.one('#video_list_next');
        var prevBtn = Y.one('#video_list_prev');

        if (nextBtn) {
            nextBtn.on('click', function(){
                count += 1;
                var dist = '-' + (count * thumbWidth) + 'px';

                videoList.transition({
                    duration: 0.5,
                    left: dist
                });
                if ((count + 3) >= length) {
                    this.hide();
                }
                prevBtn.show();
            });
        }

        if (prevBtn) {
            prevBtn.hide().on('click', function(){
                count -= 1;
                var dist = '-' + (count * thumbWidth) + 'px';
                if (count == 0) {
                    this.hide();
                    dist = "0px";

                videoList.transition({
                    duration: 0.5,
                    left: dist
                });
                nextBtn.show();
            });
