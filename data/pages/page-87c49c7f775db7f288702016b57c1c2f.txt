                //resize images
                function resizeImg() {
                    var maxWidth = 500;
                    for(var i=0; imgEle=document.getElementsByTagName('img')[i];i++){
                        if (imgEle.width > maxWidth){ imgEle.height = Math.round(((maxWidth)/imgEle.width)*imgEle.height); imgEle.width = (maxWidth); }
                        imgEle.style.visibility='visible';
                    }   
                }

                        }                                                       
                    });
                    return false;
