var prevId = '1';

//declare function to auto call panel
function callPanel()
{
	var newId = (prevId < 3) ? (parseInt(prevId) +1) : 1;
	showPanel(newId);
}

{

	//alert(prevId);

	jQuery('#slide'+id).fadeIn("slow");
