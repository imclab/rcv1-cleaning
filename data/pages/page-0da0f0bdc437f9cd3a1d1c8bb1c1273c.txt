jQuery(document).ready(function() {
  var axel = Math.random() * 8;

  jQuery.ajax({
    url: './getVideoPlayerXML.do?axel=' + axel,
    data: {},
    success: function(data) {
      var json = jQuery.xmlToJSON(data);
      if (json.RootName == 'allEntries') {
        getVideoPlayerXMLResponseHandler(json);
      } else {
        videoPlaylistLoadError();
      }
    },
    error: function() {
      videoPlaylistLoadError();
    }
  });

  function videoPlaylistLoadError() {
    jQuery('div.videoPlayer').html('<p class="videoPlaylistLoadError">We apologize but the videos are currently being refreshed. Please check back later.</p>');
  }
  
  function getVideoPlayerXMLResponseHandler(json) {
    var playList = '<ul>';
    var count = 0;
    var videoThumbServer = 'http://www.scottsdalelexus.com/VideosPage/nationals/small_thumbs/';
    
    var entries = [];
    for (var i = 0; i < json.entries.length; i++) {
      entries = entries.concat(json.entries[i].entries)
    }
    
    var videoName, videoThumb, videoPath, videoDuration;
    
    for (var i = 0; i < entries.length; i++) {
      if (i < 11) {
        for (var j = 0; j < entries[i].entry.length; j++) {
          switch (entries[i].entry[j].key) {
            case 'pretty_name': videoName = entries[i].entry[j].Text; break;
            case 'small_thumb': videoThumb = entries[i].entry[j].Text; break;
            case 'video': videoPath = entries[i].entry[j].Text; break;
            case 'duration': videoDuration = entries[i].entry[j].Text; break;
            default: /* console.log(entries[i].entry[j]); */ break;
          }
        }
        if (videoThumb.indexOf('http') == -1) {
          videoThumb = videoThumbServer + videoThumb;
        }
        playList+= '<li class="' + videoPath + '"><img src="' + videoThumb + '" alt="' + videoName + '" />' + videoName + '</li>';
      } else {
        continue;
      }
    };

    playList+= '<li class="viewAll"><a href="/VideosPage"><img src="' + ContextManager.getUrl() + 'HomePage_Cagney/videoPlayer/videos_viewAll.gif" alt="View All Videos" /></a></li>';

    playList += '</ul>';

    jQuery('div.playlist ul li').hover(function() {
        jQuery(this).addClass('hover');

        jQuery(this).removeClass('hover');
