<script>
/*
 * options.langDropdownJSON 				 - JSON - list of languages available for this product
 * options.attributeJson 						 - JSON - list of available attributes
 * options.initialVisibileAttributes - INT - Number of products to show at load time, if present will enable hide/show functionality
 * options.insertAttributeContainer  - FUNCTION - Callback to place the container on the page - Is passed the container as a jQuery object
 * options.containerElement					 - STRING - Markup to build the container
 * options.langSort									 - ARRAY - Order of country codes to sort the language drop down
 * 5661869, 4383813 
 * TODO: This is ready to be externalized
 */
os.ProductFeatures = function (options) {
	
	var attributeList = null;
	var initialVisibileAttributes;
	var isOpen;
	var toggleVisibleAttributes;
	var sortedLanguageDropdownJSON;

	options = options || {};
	initialVisibileAttributes = parseInt(options.initialVisibileAttributes);

	if( initialVisibileAttributes > 0 ){
		isOpen = false; //start closed
		toggleVisibleAttributes = true;
	}

	return {
		init: function(){
			var _this = this;

			jQuery(document).ready(function() {
				var json = options.attributeJSON; 
				sortedLanguageDropdownJSON = _this.sortLanguages(options.langDropdownJSON, options.langSort);
				_this.setAttributeContainer(options.containerElement);
				_this.setAttributeList( json );
			});
		},

		$attributesContainer: null,

		//create container and call insert fn to place it in the dom
		setAttributeContainer: function(containerElement){
			var $container;

			if(containerElement){
				$container = jQuery(containerElement);
			}
			
			if(!$container || !$container.length){
				$container = jQuery('<div id="productAttributesContainer"></div>');
			}

			this.insertAttributeContainer($container);
			this.$attributesContainer = $container;
		},

		insertAttributeContainer: function($container){
			if(typeof options.insertAttributeContainer === 'function'){
				options.insertAttributeContainer($container);
			}
			else {
				jQuery('#details_descMisc').after( $container );
			}
		},

		//saves the data and triggers the render fn() if the data is not empty
		setAttributeList: function(json){
			var rowLimit = (!isOpen && options.initialVisibileAttributes) ? options.initialVisibileAttributes : 0;
			// Verify the returned JSON has some data.
			if(!jQuery.isEmptyObject(json)){
				attributeList = json;
				this.renderFull(rowLimit);
			}
		},

		fetchJsonFeatObject: function(lang, callback){
			var _this = this;
			jQuery.ajax({
				url: '/jsonAttributeTranslation.html?productId=' + os.Simplicity.productId + '&languageCode=' + lang,
				dataType: 'json',
				success: function(data) {
					if(typeof callback === 'function'){
						callback( data );
					}
				}
			});
		},

		sortLanguages: function(langs, sortArr){
			//TODO: clean this method up
			var languageList = langs.languageList || [];
			var notInSortList = [];
			var sort = sortArr;
			var item;
			var sortedIndex;
			var sortedResult;

			if(typeof options.sortLanguages === 'function'){
				return options.sortLanguages( langs );
			}
			else if( !sortArr || !sortArr.length ){
				return langs.languageList;
			}
			else {
				sortedResult = new Array(sort.length);

				for(var i=0, l=languageList.length; i<l; i++){
					item = languageList[i];
					sortedIndex = sort.indexOf(item.code);
					if(sortedIndex > -1){
						sortedResult[sortedIndex] = item;
					}
					else {
						notInSortList.push(item);
					}
				}
				return sortedResult.concat(notInSortList);
			}
		},

		renderFull: function(limit){	
			var _this = this;	
			var prod_attr_list = attributeList;						
			var insertAttributeMarkup = options.insertAttributeMarkup;
			var markup = this.getFullMarkup(prod_attr_list, limit, sortedLanguageDropdownJSON);

			if (markup.length) { 
				this.$attributesContainer.html(markup);
				this.afterRenderFull();
			}
		},
		getFullMarkup: function(prod_attr_list, limit, prod_attr_dropdown){
			//TODO: doT or handlebars template
			var rowCount = this.getRowCount( prod_attr_list.attributes );
			var attributesToShow = ( limit !== undefined ) ? limit : options.initialVisibileAttributes;
			var attributeListMarkup = this.buildAttributeList( prod_attr_list.attributes, parseInt(attributesToShow) );
			var buildToggle = (toggleVisibleAttributes === true 
				&& rowCount > options.initialVisibileAttributes );

			if( attributeListMarkup.length ){
				// Create full table using otag.	
				return this.buildTable(prod_attr_dropdown, prod_attr_list, attributeListMarkup, buildToggle);
			}
			//if there are no attributes there is nothing to return
			return false;
		},
		getRowCount: function(attributes){
			var rowCount = 0;

			for(var index in attributes) {
				rowCount++;
			}
			return rowCount;
		},		
		buildToggle: function(){
			if(isOpen === true){
				return '<div class="attr-toggle">Less<i class="attr-is-open"></i></div>';
			}
			else {
				return '<div class="attr-toggle">More<i></i></div>';
			}
		},
		buildAttributeList: function(attributes, rowLimit){
			//TODO: doT or handlebars template
			var markup = '<ul>';
			var rowCount = 0;

			//normalize limit - 
			if(rowLimit === NaN || rowLimit < 0){
				rowLimit = 0;
			}
			// LOOP OVER THE OBJECT, WRITE LI ROWS FOR EACH AND THE DIV FOR THE LABEL.				
			for(var index in attributes) {
				markup += '<li><div class="attributeLabel">'+ index +'</div><div class="attributeData">'; 
				// LOOP OVER THE VALUES, WRITE ALL VALUES INTO A SINGLE DIV.
				for(var i=0; i<attributes[index].length; i++) {
					if (i >=1) {
						markup += ', ';
					}
					markup += attributes[index][i];
				}	
				// CLOSE THE DIV AND LI OPENED EARLIER.
				markup += '</div></li>';
				// 0 or below is the same as no limit
				if(rowLimit > 0 && rowCount+1 === rowLimit){
					break;
				}
				rowCount++;
			}

			markup += '</ul>';

			return (rowCount) ? markup : '';
		},
		buildTable: function(languageList, attributeJson, attributeMarkup, buildToggle){
			//TODO: doT or handlebars template
			// Create dropdown table using otag.	
			var activeItem ='';
			var	table_markup = '';
			var languageItem;

			attributeMarkup = attributeMarkup || '';

			table_markup += '<div id="productIntlAttr"><div id="productAttributes" class="panel-e hollow"><div class="hd"><h4>' + attributeJson.title + '</h4>'
				
			if (languageList && languageList.length > 0){
				table_markup += '<span id="prod-attr-dropdown"><select id="attr-dropdown-select">';
				for(var i=0; i<languageList.length; i++) {
					languageItem = languageList[i];
					if(languageItem && attributeJson.languageCode==languageItem.code){
						activeItem = "selected=selected";
					}
					if(languageItem){
						table_markup += '<option  title="' + languageItem.name + '"  ' + activeItem + ' value="' + languageItem.code + '">' + languageItem.nativeName + '</option>';
					}
					activeItem = "";
				}
				table_markup += '</select></span>';
				
				if (attributeJson.tooltip != ""){
					table_markup += '<span class="attr-icon-trigger info-icon" data-tt-code="' + attributeJson.languageCode + '" data-tt-msg="' + attributeJson.tooltip + '"></span>'
				}
			}
			
			table_markup += '</div><div class="bd">';
				//insert attribute markup - typically a ul
				table_markup += attributeMarkup;
			// LOOP COMPLETE - CLOSE THE UL AND WRITE THE RESULTS TO THE DOM.
			table_markup += '</div></div></div>';

			if(buildToggle === true){
				table_markup += this.buildToggle();
			}

			return table_markup;
		},
		//use this to run plugins now that the markup is in the dom
		afterRenderFull: function(){
			var _this = this;
			jQuery('#prodDescTab').addClass('hasAttributes');
			this.attrRunToolTip();
			jQuery('#attr-dropdown-select').on('change', function(){
				_this.fetchJsonFeatObject(this.value, function(data){
					_this.handleLanguageChange(data)
				});
			});

			jQuery('.attr-toggle').on('click', function(){
				_this.toggleAttributes();
			});
		},
		handleLanguageChange: function(json){
			this.setAttributeList(json);
			s.linkTrackVars = "eVar34";
			s.eVar34 =  json.languageCode + '-' + os.Simplicity.productId;
			s.tl(this,'o','Translate Selected');
		},
		toggleAttributes: function(){
			if(isOpen === true){
				isOpen = false;
				this.renderFull(options.initialVisibileAttributes);
			}
			else {
				isOpen = true;
				this.renderFull(0); //0 shows ALL attributes
			}
		},
		attrRunToolTip: function(){
			os.ToolTips('.attr-icon-trigger', {
				offset: {x: 1,y: 2},
				toolTipContent: function() {
					var _this = this,
					defaultMsg = 'Key product features have been translated for your convenience.',
					data = _this[0].attributes['data-tt-msg'],
					code = _this[0].attributes['data-tt-code'],
					msg = (data && data.value != 'undefined') ? data.value : defaultMsg,
					arCode = (code.value != 'ar') ? 'style="text-align:left"' : 'style="text-align:right"',
					$tt = jQuery('<div class="info-tooltip attributes-tooltip" ' + arCode + '>' + msg + '</div>');

					return $tt;
				}
			}); 
		}
	}.init();
};

var collectionHelpers = {
	getCollectionItemMarkup: function( product ){
		return new Handlebars.SafeString( Handlebars.partials.collectionItemPartial( product ) );
	},
	imageUrl: function(){
		if(os.Otags.image_url){
			return os.Otags.image_url;
		}
		else{
			return "http://ak1.ostkcdn.com";
		}
	},
	headerText: function( product ){
		if(product.products.length > 1){
			return "Coordinating Items";
		}
		else{
			return "Coordinating Item";
		}
	},
	getOptions: function( optionListObj, proId ){

		var options = [];
		for(var i=0, l=optionListObj.length; i<l; i++) {
			options.push(optionListObj[i]);
		}

		//IE 7/8 read the options length differently
		if($.browser.msie && $.browser.version < 9) {
		
			if( options.length <= 2){
				return collectionHelpers.hiddenInput( options[0] );
			}
			else {
				return collectionHelpers.selectOptions( options );
			}
		}
		else {
			if( options.length <= 1){
				return collectionHelpers.hiddenInput( options[0] );
			}
			else {
				return collectionHelpers.selectOptions( options );
			}
		}

	},
	selectOptions: function( options ){

		//TODO: this could probably be cleaned up to use a handlebars partial

		var html = ['<select class="optionId">'];
		var currentOption;
		html.push('<option value="">Option</option>');

		for(var i=0, l=options.length; i<l; i++){
			currentOption = options[i];
			html.push('<option value="' + currentOption.id + '" data-option-qty="' + currentOption.quantity + '" data-option-price="' + currentOption.price + '">' + currentOption.description + ' - ' + currentOption.price + '</option>');
		}

		html.push('</select>');

		return new Handlebars.SafeString( html.join('') );

	},
	getPrice: function( optionListObj ){

		var price = '';
		var option;

		for( var option in optionListObj){
			option = optionListObj[option];
			if(option && option.price){
				price = option.price;
				break;
			}
		}

		return price;

	},
	hiddenInput: function( data, id ){
		//<input name="addid4318180" value="5860072" type="hidden">
		return new Handlebars.SafeString('<input class="optionId" type="hidden" value="' + data.id + '" name="optionId">');
	},
	addToCart: function( productId, optionId, qty, type, callback, warrantyId ){
	//var cartUrl = '../api/cart/cartLayer.json';
	var cartUrl = '/api/cart/cartLayer.json';

		var json = {"changes":[
				{
					"optionId": optionId,
					"quantity": qty,
					"type": type || "optionAdd"
				}

			]}

			if( warrantyId && parseInt( warrantyId ) > 0 ){
				//4133535
				json.changes.push( {"type":"warrantyAdd","warrantyProductId":warrantyId,"parentOptionId":optionId} );
			}


		jQuery.ajax({
			type: 'POST',
			dataType: 'json',
			contentType: 'application/json',
			cache: false,
			data: jQuery.toJSON( json ),
			url: cartUrl,

					if(data){
						data.addedId = productId;

			}

os.CSEWidget = function(apiUrl, enableTracking, renderMarkup, productPrice){

	var API_URL = apiUrl;
	var $wrapper = jQuery('<div id="#cse-placeholder"></div>');
	var testGroup = parseInt(testGroup,10);
	var productId = parseInt(os.Simplicity.productId);

	if( !$wrapper.length || isNaN(productId) || !productPrice ) return false;

	return {
		init: function(){
			productPrice = this.formatPrice(productPrice);
			this.registerHelpers();
			this.getData();
			return this;
		},

		getData: function(next){

			var _this = this;

			jQuery.ajax({
				url: _this.buildUrl(),
				dataType: 'json', 
				success: function(data){
					_this.handleResults(data);
				},
				error: function(){
					if(window.console && console.error) console.error('Problem with CSE request.');
				}
			});

		},

		formatPrice: function(price){
			price = (price) ? parseFloat( price.replace(/\$/,'') ) : 0;
			if( isNaN(price) ){
				return 0; //map pricing
			}
			return price;
		},

		validatePricing: function(products){

			var validatedProducts = products.slice(0);
			var retailers;
			var indexToRemove = [];
			var proPrice = productPrice;

			var hasValidPrice = function(retailerPrice, price){
				return ( retailerPrice > price );
			};

			if(validatedProducts.length && validatedProducts[0].retailers.length){
				retailers = validatedProducts[0].retailers;

				for(var i=0; i<retailers.length; i++){
					if( !hasValidPrice(this.formatPrice(retailers[i].price), proPrice) ){
						indexToRemove.push(i);
					}
				}
				if(indexToRemove.length){
					for(var j=indexToRemove.length-1; j>=0; j--){
						if(retailers[indexToRemove[j]]) retailers.splice(indexToRemove[j],1);
					}
				}

		  }

		  return validatedProducts;

		},

		template1: Handlebars.templates.cseGroup1,

		template2: Handlebars.templates.cseGroup2,

		buildUrl: function(){

			/* Required: productId
			 *
			 * Optional: productName
			 * 			 brand
			 *			 modelNumber
			 * 			 upc
			 *			 maxResults
			 */

			var cseUrl = API_URL;
			return cseUrl.replace(/\#\{productId\}/, productId);

		},

		formatUri: function(str){
			return encodeURIComponent( this.removeWhiteSpace( str) );
		},


		removeWhiteSpace: function( str ){
			//can't use trim because of ie7/8
			return ( str.length ) ? str.replace(/^\s+|\s+$/g, "") : null;
		},

		registerHelpers: function(){
			//used by group2 
			Handlebars.registerPartial('cseGroup1Partial', this.template1);
			Handlebars.registerHelper('formatCseTitle', this.formatCseTitle);

		},

		formatCseTitle: function(originalTitle){

			if(originalTitle && originalTitle.length && originalTitle.length > 87){
				return new Handlebars.SafeString(originalTitle.substring(0,85) + '...');
			}

			return new Handlebars.SafeString(originalTitle);

		},

		handleResults: function(data){
			//cseA
			var productsData = this.validatePricing(data.products);


			if( !data 
				|| (data && productsData && !productsData.length ) 
				|| (data && productsData && productsData.length && !data.products[0].retailers.length) ){
					//no products/retailers so don't apply template
					return false;
			}

			if(typeof renderMarkup === 'function'){
				jQuery.proxy(renderMarkup,this,$wrapper, data)(); //immediately invoke proxy
			}
			else {
				this.defaultRenderMarkup($wrapper, data);
			}


			if(enableTracking) this.trackResults( data.products );
			this.applyListeners($wrapper);
		},

		//default render markup if it is not specified in config
		defaultRenderMarkup: function($wrapper, data){
			jQuery('#prod_mainRight').addClass('hide-co-offers');
			$wrapper.html(this.template1(data));
		},

		//Positions modal(group 2) below title/price and above short description
		setWrapperPosition: function( $elm ){
			var offset = jQuery('#prod_mainCenter').find('.Oprice').position();

			var addOffset = function(offset, value){
				return offset = parseInt( offset ) + value + 'px';
			}

			$elm.find('#cse-modal').css({ top: addOffset(offset.top, 30), left: offset.left });

		},

		applyListeners: function($elm){
			var _this = this;

			$elm.on('click', '.modal-close', function(){
				$wrapper.fadeOut(200);
			});

			if(enableTracking){
				$elm.on('click', 'a', function(){
					//track click
					s.tl(this, 'e', 'CSE Widget Click');
				});
			}

		},

		//fire pixel if we have more than 0 retailers
		trackResults: function( products ){

			if(products.length && products[0].retailers.length && window.s){
				s.tl(this, 'o', 'CSE Has Results', {
					events: 'event43'
				});
			}

		}


	}.init();
};
