                        $("#hidden_security_question").val($(data).text());
                        $("#spanSecurityQuestion").html($(data).text());
                        $("#divForgot").hide();
                        $("#divAnswer").show();
                    } else if (resultText.match('error-no-question-false')) {
                        $("#wrongEmailError").html("The email address entered was not found.");
                    }
                },
                error: function(response) {
                    // TODO --> HIDE ENTIRE CONTROL

                }

            });
        });

                        $("#spanYourEmail").html($("#hidden_forgot_email").val());

                    },
                    error: function(response) {
                        // TODO --> HIDE ENTIRE CONTROL
                        alert("Error Response:\r\n" + response.responseText);
                    }

                });
            }
        });

        $("#btnEmailPassword").click(function() {
            if ($("#hidden_forgot_email").val()) {
                emailPassword($("#hidden_forgot_email").val());
            }
        });

            hideDivs();
            $("#divForgot").show();
        });
