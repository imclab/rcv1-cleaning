function trimString (str) {
  while (str.charAt(0) == ' ')
    str = str.substring(1);
  while (str.charAt(str.length - 1) == ' ')
    str = str.substring(0, str.length - 1);
  return str;
}

function validEmail(email)
{
    var vAtSym    = email.indexOf('@')
    var vPeriod   = email.lastIndexOf('.')
    var vSpace    = email.indexOf(' ')
    var vLength   = email.length - 1   // Array is from 0 to length-1

    if (vAtSym < 1 ||                     // '@' cannot be in first position
    	vPeriod <= vAtSym + 1 ||          // Must be atleast one valid char btwn '@' and '.'
    	vPeriod == vLength  ||            // Must be atleast one valid char after '.'
    	vSpace  != -1)                    // No empty spaces permitted
	{
        return false;
    }
	return true;
}

               pars += '&referenceRegistryNumber=' + document.orderForm.referenceRegistryNumber.value;

		{
			method:'post',
			postBody:pars,
			onSuccess:function(transport){

			var validate = eval('('+ transport.responseText +')');				
			Omniture_onClick_RegistryAdd(regId, pidsDetails);

			}

			}

			}
		}

function createItemQuantityListURL(itemQuantityList){
   var createItemQuantityListURL = 'quantityList=';
   for(var i=0; i<itemQuantityList.length; i++){
      
      createItemQuantityListURL=createItemQuantityListURL+itemQuantityList[i];
      if(i<(itemQuantityList.length-1)){
      createItemQuantityListURL=createItemQuantityListURL+'&quantityList=';
      }
   }
   return createItemQuantityListURL;
}

function getValueFromPidElement(pidElement,element){
	var idList = pidElement.split("|");
	var sku = idList[1];
	var pid = idList[0];
	if(element.indexOf('PID')>=0){
		return pid;
	}else{
		return sku;
	}
}

function checkAvail()
{
	if(!isInStoreProduct)
	{
		var prod_n_sku = document.orderForm.prod_1.options[document.orderForm.prod_1.selectedIndex].value;
		if (prod_n_sku.indexOf("|") > -1)
		{
			var arrayOfSkus = prod_n_sku.split("|");
			var pid = arrayOfSkus[0];
			var skuToShow = arrayOfSkus[1];
			for(var i = 0; i < itemMap.length; i++)
			{
			    if (itemMap[i].sku == skuToShow && itemMap[i].pid == pid)
				{
					if(itemMap[i].avail != "" && itemMap[i].avail != "null" && ( itemMap[i].avail.indexOf("OUT_OF_STOCK") > -1 ||itemMap[i].avail.indexOf("DISCONTINUED") > -1 ||itemMap[i].avail.indexOf("NOT_AVAILABLE") > -1))
					{
						forOutOfStock();
					}
					/**
					  * TRU Wanted PMO 49027 on hold now.Commenting the changes.
					  *
					else if(itemMap[i].avail != "" && itemMap[i].avail != "null" && itemMap[i].avail.indexOf("DISCONTINUED") > -1 )
					{
					    forDiscontinued();
					}**/
					else if(itemMap[i].avail != "" && itemMap[i].avail != "null" && (itemMap[i].avail.indexOf("ADVANCED_SALE_LIMITED") > -1 || itemMap[i].avail.indexOf("ADVANCE_SALE") > -1))
					{					   
						forAdvanceSale();
					}
					else if(itemMap[i].avail != "" && itemMap[i].avail != "null" && itemMap[i].avail.indexOf("RELEASE_DATE") > -1 && "" != "")
					{
						forPreOrder("");
					}
					else 
					{
						resetStatus();
						flag1 = false;
					}
				}
			}
		}
		else
		{	
			resetStatus();
			flag1 = false;
		}
	}
}
