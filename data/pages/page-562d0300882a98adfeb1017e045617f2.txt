jQuery(document).ready(function(){

		pageBody = jQuery("body");
			
	function exeBackgroundCssFallback(){
		var src,
			imgPageBackgroundFallback = jQuery("img.pageBackgroundFallback");
			pageBody.addClass("is-ie"+browserversion);
			src = imgPageBackgroundFallback.css("background-image");
			src = src.replace('url("','').replace('")','');
			imgPageBackgroundFallback.attr("src", src);
	}
	
	jQuery(window).one('backgroundCSSLoaded', function(e) {
		if(jQuery.browser.msie && browserversion < 9) {
			exeBackgroundCssFallback();
