function findCurrentNav() {
	var currLoc = window.location.href;
	var hardNavLinks = $("#hard-coded-nav li");
	var topNavSelected = false;

	if ( currLoc != "/" ) {
		hardNavLinks.each(function() {
			var url = $(this).find("a").eq(0).attr("href");
			var targetUrl = url;

			if ( url != null && url.indexOf("#") >= 0 ) {
				targetUrl = url.split("#")[0];
			}				

			if ( targetUrl != null && currLoc.indexOf(targetUrl) >= 0 ) {
					$(this).addClass("current selected");
					if ( !topNavSelected ) {
						$(this).parents("li").addClass("current selected");
					}
					topNavSelected = true;
			}
		});
	}
}

function addBreadCrumbs() {

	// Get active nav and its parents
	var activeNavItem = $("li.current.selected:last");
	var parentTrace = activeNavItem.parents("li");
	var breadcrumbs = "";
	
	// Set class on all active nav and parents
	parentTrace.each(function() {
		var parentItem = $(this).find("a:first");
		var parentUrl = parentItem.attr("href");
		var parentText = $(this).find("a:first").html();
		breadcrumbs += "<a href='" + parentUrl + "'>" + parentText + "</a><span> > </span>"
	});
	
	if ( parentTrace != null && parentTrace.length >= 0 ) {
		var activeNavText = activeNavItem.find("a:first").html();
		var pageTitle = $(".content-item.page h1:eq(1)").html();

		if ( activeNavText != null ) {
			$(".breadcrumb-container").html( "<a href='/'>Home</a> > " + breadcrumbs + activeNavText);
		}
	}
}
