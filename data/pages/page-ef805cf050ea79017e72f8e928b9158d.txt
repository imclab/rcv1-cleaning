	function checkHeaderForm(aform) {
		var inputField = $j(aform).children('[type=text]');
		var inputText = $j.trim(inputField.val());
		var defaultText = $j(inputField).attr('title');
		var valid = true;
		var errMsg = "";
		if(inputText === ""){
			valid = false;
			errMsg = "Please enter a search term.";
		} else if (inputText === defaultText) {
			valid = false;
			errMsg = "Please enter a valid search term.";
		}

		if(!valid) {
			alert(errMsg);
			try {
				inputField.focus();
				inputField.select();
				inputField.val('');
			} catch (err) {
				// Failure is a no-op
			}
		}

		return valid;
	}

	
	function setGlobalSearchActionDestination (url, show) {
		$j('#global-search').attr('action', url);
		$j('.no-result-global-search').attr('action', url);
		if (show == '') {
			$j('#idShow').attr('disabled', true);
			$j('#idShow').attr('value', '');
			$j('#nrShow').attr('disabled', true);
			$j('#nrShow').attr('value', '');
		} else {
			$j('#idShow').attr('disabled', false);
			$j('#idShow').attr('value', show);
			$j('#nrShow').attr('disabled', false);
			$j('#nrShow').attr('value', show);
		}
	}
